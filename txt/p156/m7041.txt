


	
	מחברות עמנואל: מחברת עשרים
	




אָמַר הַמְחַבּר:

אַחֲרֵי אֲשֶר רָאָה הַשָּׂר הַמּוּסָרִים הַנִּזְכָּרִים

וְהַתְּפִלוֹת הַמְהֻלָּלוֹת

נִפְלָא עֲלֵיהֶם בְּתַכְלִית הַפְּלִיאָה

וַיֹּאמֶר אֵלַי: רָאִיתִי יָדְךָ בַשִׁירִים וּבַמְּלִיצוֹת חֲזָקָה וְנוֹרָאָה

וּלְךָ לָשׁוֹן מְדַבֶּרֶת גְּדוֹלוֹת כְּאִלּוּ הָיִיתָ בִּימֵי אַנְשֵׁי הַנְּבוּאָה

וְלֹא רָאִיתִי שִׁירָה כְּשִׁירְךָ עֹז אוֹזֶרֶת

וְכִמְעַט הַגְּבוּל עוֹבֶרֶת

רַק שִׁירָה רְאִיתִיהָ לְכָל הַשִּׁירוֹת חוֹדֶרֶת

וְהִיא עַל עַמּוּדֵי הַשֵּׂכֶל גֻּלָּה וְכוֹתֶרֶת

וּבְרֹאשׁ מְשׁוֹרְרֵי הַזְּמָן עֲטֶרֶת תִּפְאֶרֶת

וּתְחִלָּתָהּ עוּרִי מְלִיצָתִי בְּרֹב תִּפְאֶרֶת

עוּרִי לְשׁוֹן זָהָב וְהָאַדֶּרֶת

רְאִיתִיהָ הֲרֵי יָמִים וְשָׁנִים

בְּיַד אֲנָשִׁים מְבִינִים

יִשָּׂאוּהָ עַל כָּתֵף יִסְבְּלוּהָ

וְעַל רֹאשׁ שִׂמְחָתָם יַעֲלוּהָ

וְכָל רוֹאֶהָ לִקְרָאתָהּ יֶחֱרַד

וְחָשַׁבְתִּי לְרֹב יָפְיָהּ כִּי בָאָה מִסְפָרַד

וְלוּ מְצָאתִיהָ לְרֹב חִשְׁקִי בָהּ וַאֲשֶר אוֹתָהּ אֶאֱהַב

נָתַתִּי מִשְׁקָלָהּ זָהָב

וְלוּ רָאִיתָ צַחוּת לְשׁוֹנָהּ

אָז תְּבָרֵךְ הַמְחַבֵּר בְּשֵׁם אֲדֹנָי רִאשׁוֹנָה

וְלוּ תֵרֵד לְעֹמֶק עִנְיָנֶיהָ

וְתִרְאֶה סַפִּירֶיהָ וּפְנִינֶיהָ

תִּבְרַח מִפָּנֶיהָ

וַיְהִי כֶּאֱמֹר הַשָׂר כָּזֹאת לֹא אַחַת וְלֹא שְׁתַּיִם

עַד אֲשֶר לְקוֹלוֹ צָלְלוּ הָאָזְנַיִם

לֹא יָכֹלְתִּי לְהִתְאַפֵּק וְשָׂחַקְתִּי עַד אֲשֶר דָּמְעוּ הָעֵינָיִם

וְכִרְאוֹת הַשָּׂר רֹב שַׂחֲקִי

אֲשֶר לֹא הָיָה מֵחֻקִּי

כִּמְעַט בּוֹשׁ וְחָפַר

וְחָשַׁב לְרֹב שַׂחֲקִי כִּי לִבִּי אֶת בְּרִיתוֹ הֵפַר

וַיִּפֶן בְּחֵמָה אֵלַי וַיֹּאמַר

חֵי אָנִי אִם רָאִיתָ הַשִּׁיר אֲשֶר אֹמַר

מִפַּחְדוֹ לִבְּךָ סָמַר

וְעֵינֶיךָ מִבְּכִי חֳמַרְמָר

וְלֹא תִתְמַרְמֵר כַּאֲשֶר תִּתְמַרְמָר

וָאַעַן וָאֹמַר אֲדֹנִי אַחֲרֵי אֲשֶר הַשִּׁיר הַהוּא תֶאֱהָבֶנּוּ

וְרֹאשׁ כָּל הַשִּׁירִים אֲשֶר רָאִיתָ תַּחְשְׁבֶנוּ

מַה לַּעֲשׂוֹת לָאִישׁ אֲשֶר לְךָ יִכְתְּבֶנוּ

וַיֹּאמֶר הַשָּׂר הִנְנִי נִשְׁבָּע בַּאֲשֶר עָשָׂה אֶרֶץ וְחוּצוֹת

כִּי הָאִישׁ אֲשֶר יָשִׁית בְּנַפְשׁוֹ עֵצוֹת

וְאֵלַי יִכְתְּבֶנוּ אֶעֱשֶׂה לוֹ מִזָּהָב מִשְׁבָּצוֹת

וְהַלְבֵּשׁ אוֹתוֹ מַחֲלָצוֹת

וָאַעַן וָאֹמַר חַיֶּיךָ הַשָּׂר אוֹתִי תַלְבִּישׁ בִּגְדֵי חֹפֶשׁ

כִּי הַיּוֹם אֶתֵּן לְלִבְּךָ נֹפֶשׁ

וְעַתָּה הִכּוֹן וְהָכֵן לְךָ בִּגְדֵי חֶמְדָּה

כִּי אִתִּי הַשִּׁירָה הַחֲמוּדָה

וַיָּפָג לֵב הַשָּׂר כִּי לֹא הֶאֱמִין

וָאַעַן וָאֹמַר חַי אָנִי אִם אִשׁ לְהַשְׂמִיל וּלְהַמִין

כִּשְׁמֹעַ הַשָּׂר עָלַז לִבּוֹ

וַתְּחִי רוּחוֹ בְקִרְבּוֹ

וַיֹּאמֶר אֵלָי בַּאֲשֶר אָהַבְתִּי תַחְמֹד

מְהֵרָה חוּשָׁה אַל תַּעֲמֹד

וְהָבֵא לִי הַשִּׁיר אֲשֶר אָהַבְתִּי

כּי לִרְאוֹתוֹ תָאַבְתִּי

וָאַעַן וָאֹמַר הִנֵּה לְךָ הַשָּׂר הַשִּׁיר אֲשֶר תִּשְׁאָלֵהוּ

הַרְחֵב פִּיךָ וַאֲמַלְאֵהוּ





וָאֹמַר:





עוּרִי מְלִיצָתִי בְּרֹב תִּפְאֶרֶת

עוּרִי לְשׁוֹן זָהָב וְהָאַדֶּרֶת

הִתְעוֹרְרִי הִתְנַעֲרִי מַה תֵּלְכִי

עַל מַעֲרֶכֶת שִׁיר כְּמוֹ חִגֶּרֶת

מַה תֵּזְלִי / לִמְאֹד לְשַׁנּוֹת טַעֲמֵךְ

בֹּאִי וּמַה זֶה אַתְּ כְּמִתְנַכֶּרֶת

חֶרֶב תְּבוּנָתִי וּמַה לָךְ תִּשְׁקְטִי

חֶרֶב חֲלַל גָּדוֹל וְהַחוֹדֶרֶת

מַה תֵּשְׁבִי בֵּין מִשְׁפְּתֵי תַעְרֵךְ וּמַה

לָךְ תִּירְאִי וַתֵּלְכִי קוֹדֶרֶת

אִי יַם תְּבוּנָתִי הֲתָנוּס מִפְּנֵי

שִׁירַת אֲבִיגְדוֹר הוֹדְךָ זוֹכֶרֶת

יַרְדֵּן תְּעוּדוֹת מַה לְךָ תִּסֹּב אֲחוֹ-

רַנִית וְדִמִּיתָ לְיָם כִּנֶרֶת

הָיוּ זְמִירַיִךְ חֲקוּקִים עַל פְּנֵי

תֵבֵל בְּעֵט בַּרְזֶל וְגַם עוֹפֶרֶת

עוּרִי מְלִיצָתִי וְאֵלָיו דַבְּרִי

פָּנִים בְּפָנִים אַל תְּהִי סוֹרֶרֶת

וּקְחִי בְיָדֵךְ שִׁיר וּפָנָיו קַדְּמִי

כִּי יֵשׁ וְיֵשׁ אֶל פָּעֳלֵךְ מַשְׂכֹּרֶת

כָּזֹאת וְכָזֹאת תֹּאמְרִי אֵלָיו אֲנִי

חוֹלַת אֲהָבִים טוּבְךָ שׁוֹחֶרֶת

דּוֹדִי נְגִידִי קַח בְּרָכָה הוּבְאָה

מִקִּיר לְבָבִי לָךְ מְאֹד נוֹהֶרֶת

יִהְיוּ לְרָצוֹן מַעֲנֵי אַף יִזְּלוּ

כַּטַּל וְשִׂיחִי יֶעֱרַב כִּקְטֹרֶת

דּוֹדִי בְּחַנְתִּיךָ צְרַפְתִּיךּ בְּכוּר

מִבְחַן מְצָאתִיךָ בְּיַד גּוֹבֶרֶת

אֶזְרַע בְּךָ שִׁירַי וּמֵאָה אֶמְצְאָה

עַל כָּל אֲשֶר נַפְשִׁי בְךָ שׁוֹעֶרֶת

לֹא יָדְעָה נַפְשִׁי אֲבָל הַיּוֹם לְבַד

כַּמָּה וְכַמָּה חַרְבְּךָ דוֹקֶרֶת

תִּלְאֶה יְחִידָתִי לְהָכִיל גָּדְלְךָ

הִיא מִמְּךָ תִבְרַח וְלָךְ חוֹזֶרֶת

קַמְתָּ לְרוֹעֶה נֶאֱמָן דּוֹר לֶאֱסוֹף

הַצּוֹלְעָה גַּם לַחֲבוֹשׁ נִשְׁבֶּרֶת

בָּךְ יִמְצְאוּ אִישִׁים יְסוֹדוֹת הָאֱמוּן

לוּ תִהְיֶה כָּל הָאֱמֶת נֶעְדֶּרֶת

תִּמְשֹׁךְ בְּחַכַּת שִׁירְךָ לִבִּי וְעַל

לֵב כָּל נְדִיב עַם תִפְרְשָׂה מִכְמֹרֶת

פֶּצַע אֲהָבֶיךָ לְכָל דּוֹר נֶאֱמָן

אָכֵן נְשִׁיקַת בִּלְתְּךָ נֶעְתֶּרֶת

הָמוּ לְךָ מֵעַי וְגַם הָמוּ לְךָ

הַלֵּב וְהַכָּבֵד וְהַיֹּתֶרֶת

נַפְשִׁי בְנַפְשֶׁךָ צְרוּרָה נִקְשְׁרה

אוּלָם בְּכָל דּוֹד בִּלְתְּךָ גוֹעֶרֶת

אֶשְׁאַף כְּעֶבֶד צֵל וְדִמִּיתִי לְמוֹ

הָרָה אֲשֶׁר כָּל יוֹם וְלֵיל סוֹפֶרֶת

נַפְשִׁי בְּךָ תָגִיל צְבִי חִשְׁקִי וּבָךְ

מִתְהַלֲלָה תָמִיד וּמִתְפָּאֶרֶת

תִּבְחַר לְהִסְתּוֹפֵף בְּבֵיתָךְ הַצְּבִי

מִדּוּר בְּבֵית בִּלְתָּךְ וּמִתְגּוֹרֶרֶת

תַּבִּיט תְּמוּנָתָךְ בְּעֵין הִדּוּר אֲבָל

הִיא מֵרְאוֹת זוּלָתְךָ עִוֶּרֶת

מִיּוֹם אֲשֶׁר נַפְשִׁי יְדָעָתָךְ בְשֵׁם

שָׁבָה לְרֹב הָאַהֲבָה שִֹבֹּרֶת

הִתְנַצְּלָה עֶדְיָה לְנוּדָהּ וַתְּהִי

מִיּוֹם פְּרִידָה כָאֳנִי סֹעֶרֶת

לָשׁוּב לְיַבֶּשֶׁת מְנוּחָה אֶל מְחוֹז

חֶפְצָהּ בְּכָל לִבָּהּ הֲכִי חוֹתֶרֶת

לוּלֵי אֲשֶׁר לָשׁוּב מְהֵרָה יִחֲלָה

אָז יוֹם פְּרִידָה הָיְתָה נִגְזֶרֶת

דּוֹדִי בְטֶרֶם בּוֹא זְמִירֶיךּ הֲכִי

צָרָה יְחִידָתִי כְּמוֹ מַבְכֶּרֶת

אָכֵן בְּבוֹאָם הָיְתָה נַפְשִׁי כְזֹ-

הַר הָרְקִיעִים זָהֳרָהּ עַל לְבָבִי וַעֲנַדְ-

תִּים כַּעֲנָקִים לִי עֲלֵי גַרְגֶּרֶת

מַה יָקְרָה שִׁירָתְךָ הַשָּׂר וּמָה

נָעְמָה אֱלֵי לִבִּי וּמַה נֶּהְדֶּרֶת

שִׁירָתְךָ הַשָּׂר נְגִידִי תַּחֲרִישׁ

בַּדִּים וְאוֹתוֹת קֹסְמִים מֵפֶרֶת

הַשּׁוֹכְנִים עָפָר לְרֵיחָהּ יִחְיוּ

אַף הָרְפָאִים מִשְּׁאוֹל עוֹרֶרֶת

מָשְׁכָה בְיָפְיָהּ הַלְּבָבוֹת וַתְּהִי

שִׁירַת אֱנוֹשׁ זוּלָתְךָ מַמְזֶרֶת

הִיא לַתְּבוּנָה אֵם וְכִמְעַט מַעֲמַד

שֵׂכֶל חֲשַׁבְתִּיהָ וְגַם נִבְצֶרֶת

הִיא הָיְתָה אָמוֹן לְמוֹ דַעַת וְאֵם

לַשִּׁיר וְלַמִּקְרָא וְלַמָּסֹרֶת

הִיא נֹזְלָה גִשְׁמֵי נְדָבוֹת הֵן כְּעֵין

שַׁחַק וְטַל אוֹרוֹת הֲכִי מַמְטֶרֶת

הָיוּ מְלִיצוֹת הַמְּלִיצִים כַּחֲלוֹם

הַבָּא בְרֹב עִנְיָן וְהִיא פוֹתֶרֶת

תִּדְמֶה לְבִכּוּרָה בְרֵאשִׁיתָה וְאַךְ

שִׁירַת נְדִיבִים דָּמְתָה בַקָּרֶת

רִפְאוּת לְשָרְרִי הִיא וְגַם שִׁקוּי לְעַצְ-

מוֹתַי וְשִׁיר זוּלָתְךָ מַמְאֶרֶת

תּשְׂחַק לְיוֹם אַחְרוֹן וְתָבוּז יִקְהַת

עַמִּים בְּגַאְוָתָהּ וּמִשְׂתָּרֶרֶת

עַל כָּל דְּרָכֶיהָ מְאוֹרוֹת נָגְהוּ

יָקוּם כְּהַיוֹם אֶת אֲשֶׁר גוֹזֶרֶת

מֵרֹאשׁ יְנִיקוֹת הַמְּלִיצָה קָטְפָה

רִק מֵחֲרָזִים לָקְחָה צַמֶּרֶת

יִתְעוֹרֲרוּ כָל יוֹרְדֵי בוֹר לוּ עֲלֵי-

הֶם שֵׁש פְּעָמִים תִּהְיֶה גוֹהֶרֶת

מִי מוֹשְׁכוֹת שִׁירָתְךָ יִפְתַּח וּמִי

זֹאת מַעֲדַנוֹת זָהֳרָה קוֹשֶׁרֶת

תִּפְתַּח בְּשִׁיר חָדָשׁ וְאֵין סֹגֵר וּמִי

יוֹאֵל וְיִפְתַּח כַּאֲשֶׁר סוֹגֶרֶת

לָהּ יִקְּדוּ שָׂרִים וְיִכְרָעוּ וְלָהּ

יִשְׁתַּחֲווּ לוּ הָיְתָה נִכֶּרֶת

תּוֹצִיא לְאוֹר כָּל תַּעֲלֻמָה וַתְּהִי

דֶּלֶת וּמַפְתֵּחַ לְכָל מִסְגֶּרֶת

טוֹב עוֹלֲלוֹת שִׁירָתְךָ דוֹר מִבְּצִיר

שִׁירַת נְדִיבֵי עָם וּבָם עֹצֶרֶת

חֶמְדָה יְחִידָתִי סְפִיחֶךָ וְאֶת

עִנְבֵי נְזִירֶיךָ הֱיוֹת בּוֹצֶרֶת

תִּשְׁאַב בְּשָׂשׂוֹן מֵי עֲיָנוֹת שִׂכְלְךָ

אַף שִׁבֳּלֵי חַסְדָּךְ תְהִי קוֹצֶרֶת

עוֹרֵךְ לְהוֹד שִׁירָךְ זָמִיר זוּלָתְךָ

עוֹרֵךְ לְקוֹל צִפּוֹר אֲתוֹן נוֹעֶרֶת

הַשִּׂיר לְךָ יֵאוֹת וְאָכֵן בִּלְתְּךָ

בַּשִּׁיר כְּגַנָּב בָּא בְּתוֹך מַחְתֶּרֶת

לָשׁוֹן אֲשֶר תָּקוּם לְמִשְׁפָּט אִתְּךָ

תַּרְשַׁע בְּעֵת בָּהּ תִּגְעֲרָה מִגְעֶרֶת

תַּעַשׂ לְךָ בָּתִּים בְּשִׁירִים נִמְרְצוּ

עֵת הַמְּלִיצִים יַעֲשׂוּ מִנְהֶרֶת

עֶרְכָּךְ לְמוּל כָּל שָׁר כְּעֶרֶךְ הַצְּבִי

רוֹעֶה בְשׁוֹשַנִּים לְמוּל חַזֶּרֶת

הָיָה תְמוֹל הַשִּיר דְּבָרוֹ נֶעֱלָם

אַף מַעֲלָתוֹ מֵאֱנוֹשׁ נִסְתֶּרֶת

גַּם הַמְּלִיצָה נִסְתְּרָה לֹא יָדְעָה

אִישׁ הָיְתָה הָעֹבְרִים שׂוֹכֶרֶת

עַד שִׂכְלְךָ בָא עַד מְלוֹן קִצָּהּ וְגַם

קוֹרֵא וּפוֹשֵׁט בָּהּ כְּמוֹ אִגֶרֶת

יָעֹד יְעַדְתָּהּ עוֹד וְגַם הִנֵּה לְךָ

הָרָה וְיֹלֶדֶת וְגַם דֹּגֶרֶת

מַה נִפְלָאָה אַהֲבָתָךְ הַצְּבִי

הָיְתָה בְתוֹךְ לִבִּי כְּאֵשׁ בּוֹעֶרֶת

שָׁרְשָׁהּ בְּתוֹךְ לִבִּי וְהִנֵּה עַל שְׂפַת

לָשׁוֹן כְּאַפִּיק נַחֲלָה נִגֶּרֶת

לֹא יָכְלָה עוֹד הַצְפּינוֹ חִשְׁקְךָ

שָׁמָּה לְשׁוֹנִי לִהְיוֹת דּוֹבֶרֶת

קַח נָא נְשִׂיא הַדּוֹר תְּשׁוּרָה מִידֵי

עַבְדָּךְ אֲשֶר נַפְשׁוֹ בְךָ נִקְשֶׁרֶת

תִּצְמָא לְךָ נַפְשׁוֹ וְתַפְלִיא לַעֲשׂוֹת

הַטּוֹב בְּעֵינֶיךָ וּמִתְמַכֶּרֶת

כִּי יָדְךָ יַרְדֵּן תְּכוּנוֹת תַּעֲמִיד

אַף יַם תְּעוּדוֹת רֹם בְּעֹז פּוֹרֶרֶת

לוּ תֹאמְרָה שֶׁמֶשׁ בְּגִבְעוֹן דֹּם וְיָ-

רֵחַ לְקוֹלָךְ שָׁמְרוּ מִשְׁמֶרֶת

תִּבְחַר וְתִקְרַב אֵת מְלִיצָתִי כְמוֹ

עוֹלָל עֲלֵי יַד אַהֲרֹן נִקְטֶרֶת

תָּשִׁיר בְּשִׁיר חָדָשׁ לְךָ וּלְהַחֲזִיק

בָּךְ קוֹרְאָה בִשְׁמָךְ וּמִתְעוֹרֶרֶת

הֵא שִׁיר יָסַדְתִּיו מִפְּנִינֵי מַאֲמָר

לִהְיוֹת עֲלֵי רֹאשָׁךְ לְהוֹד וַעְטֶרֶת


  


וַיְהִי כִּשְׁמֹעַ הַשָּׂר הַשִּׁיר וּדְבָרָיו הַיְקָרִים

פָּשַׁט אֶת בְּגָדָיו וְלָבַשׁ בְּגָדִים אֲחֵרִים

וְהִלְבִּישֵׁנִי1 בְּגָדָיו הַחֲמוּדוֹת וּמְעִילָיו

וּכְתֹנֶת הַפַּסִּים אֲשֶר עָלָיו

וַיְכַסֵּנִי מֶשִׁי וַיִּנְעָלֵנִי תָּחַשׁ

וַיֹּאמַר בֶּאֱמֶת הָיָה מְחַבֵּר זֶה הַשִּׁיר חֲכַם חֲרָשִׁים וּנְבוֹן לָחַשׁ

וְכָל מְשׁוֹרְרֵי הַזְּמַן בִּלְעָדָיו אָכְלוּ פְּרִי כָחַשׁ

וּבְחֵפֶץ גָּדוֹל הָיִיתִי מַבִּיט תְּמוּנָתוֹ

אִם הוּא תָמִים כְּמַעֲלָתוֹ

מִי יִתֵּן יָדַעְתִּי וָאֶמְצָאֵהוּ אָבֹא עַד תְּכוּנָתוֹ

וָאֹמַר אֶל הַשָּׂר אֱמֹר נָא נֵזֶר תִּפְאַרְתִּי

אִם אַרְאֶךָּ הָאִישׁ אֲשֶר חִבְּרוֹ מַה מַשְׂכֻּרְתִּי

וַיֹּאמֶר הַשָּׂר הִנְנִי נִשְׁבָּע בֶּאֱמוּנָתִי

כִּי תִהְיֶה נֶאֱמֶנֶת לְךָ אַהֲבָתִי

וְאַתָּה תִהְיֶה עַל בֵּיתִי

וְהָאִישׁ הַהוּא עַל שִׁכְמִי אֶשָּׂאֶנּוּ

וַעֳטָרוֹת לִי אֶעֶנְדֶנּוּ

כִּי הוּא יִהְיֶה אִישׁ סוֹדִי וַחֲמוּדִי

חַגִּי חָדְשִׁי וְשַׁבַּתִּי וְכָל מוֹעֲדִי

וְיַחְדָּו נִהְיֶה שַׁעֲשׁוּעִים יוֹם יוֹם

וּבוֹ נִמְצָא לְפִיד יוֹם פִּדְיוֹם

וָאַעַן וָאֹמַר דַּע אֲדֹנִי הַשָּׂר כִּי הָאִישׁ אֲשֶר אַתָּה שׁוֹאֵל

הוּא הַנּוֹשֵׂא דֶגֶל אַהֲבָתְךָ עִמָּנוּאֵל

הַמְדַבֵּר אֵלֶיךָ פָּנִים בְּפָנִים

אֲשֶר נִרְאוּ עַל פִּיו אֵלֶה הַנִּצָּנִים

וְעַל פִּיו יִסְעוּ וְעַל פִּיו חוֹנִים

חַיּוֹת הַשִּׁיר וְהָאוֹפַנִּים

כִּשְׁמֹעַ הַשָּׂר דְּבָרַי חִבְּקַנִי וּנְשָׁקַנִי

וְכַאֲשֶׁר יִדְבַּק הָאֵזוֹר אֶל מָתְנֵי אִישׁ אֵלָיו הִדְבִּקַנִי

וַיַּרְכִּיבֵנִי בְּמִרְכֶּבֶת הַמִּשְׁנֶה

וַיּוֹסֶף אֵת כָּל אֲשֶׁר נָתַן אֵלַי לְמִשְׁנֶה

וַיְהִי בִּהְיוֹתִי בְּבֵית הַשָּׂר שׁוֹתֶה מַמְתַּקִּים וְאוֹכֵל מַשְׁמַנִּים

וְלוֹבֵשׁ שָׁנִי עִם עֲדָנִים

וְהִנֵּה שְלֹשָׁה אֲנָשִׁים מִדְיָנִים

מִגֹּאֲלֵי הַשָּׂר וַאֲחֻזַּת מֵרֵעָיו

וּמֵחֶבְרַת מְיֻדָּעָיו

בָּאוּ אֶל בֵּית הַשָּׂר וַיִּהְיוּ שָׁם

הָאֶחָד אֲשֶׁר הוּא אָשֹׁם אָשָׁם

כִּי הָיוּ לוֹ בִגְדֵי חֹפֶשׁ לְרִכְבָּה

וּכְדֵי לְהַכְאִיב נַפְשִׁי וּלְהַעֲצִיבָהּ

הָיָה שֹאֵל מִמֶּנִּי בְּגָדַי וּמִנְעָלַי

וְאַדַּרְתִּי וּמְעִילַי

וְאִם הָיִיתִי שֹאֵל מִמֶּנּוּ דָבָר יְמָאֵן אֵלָי

וְהַשֵּׁנִי יִקַּח בְּגָדַי בַּסֵּתֶר וַיַּלְבִּישֵׁם

וְהַשְּׁלִישִׁי כָאָשָׁם אַכְזָרִי הוּא וְלֹא יְרַחֵם

וּבְיָד חֲזָקָה יִקָּחֵם

וַאֲנִי הָיִיתִי בְכָל זֶה מַחֲרִישׁ

וְהַיָּגוֹן בְּלִבִּי יַשְׁרִישׁ

כְּנָחָשׁ יִשּׁוֹךְ וּכְצִפְעוֹנִי יַפְרִישׁ

עַד אֲשֶר נִרְאוּ בְּפָנַי שְׂעִפֵּי הַדְּאָגָה

וְדָלִפָה נַפְשִׁי מִתּוּגָה

וַיְהִי הַיּוֹם מְצָאַנִי הַשָּׂר סַר וְזָעֵף וּכְסָלַי מָלְאוּ נִקְלֶה

וַיֹּאמֶר אֵלַי מַדּוּעַ פָּנֶיךָ רָעִים וְאַתָּה אֵינְךָ חוֹלֶה

וָאֹמַר אֶל הַשָּׂר אֲדֹנִי וּמַדּוּעַ לֹא יֵרְעוּ פָנַי

וּבְגָדַי הַחֲמוּדוֹת מַחֲמַד עֵינַי וַעֲדָנַי

נְתוּנִים לְעַם אַחֵר וְרוֹאוֹת עֵינַי

וְלוּלֵי יָרֵאתִי מַעֲלָתְךָ אָמַרְתִּי בְשִׂירִי

עַל כָּל הַלּוֹבְשִׂים מַלְבּוּשׁ נָכְרִי

דִּבְרֵי בוּז וְקָלוֹן וּנְבָלוֹת

יַעֲשׂוּ בָהֶם רֹשֶׁם יוֹתֵר מִפְּצָעִים וַחֲבָלוֹת

כִּי לִי עַל אַנְשֵׁי עַוְלוֹת

לָשׁוֹן מְדַבֶּרֶת גְּדוֹלוֹת

וַיֹּאמֶר הַשָּׂר אַשְׁבִּיעֲךָ בְּאַהֲבָתִי הַיְקָרָה

לְחַבֵּר עַל כָּל אֶחָד מֵהֶם שִׁירָה

וּקְבַע שִׁיר כָּל אֶחָד מֵהֶם בְּדַלְתֵי חַדְרוֹ

וּכְתֹב עָלָיו כָּכָה יֵעָשֶׂה לָאִישׁ אֲשֶר לֹא יַחְפֹּץ הַשָּׂר בִּיקָרוֹ

וָאָקוּם בְּמִצְוַת הַשָּׂר וָאֲחַבֵּר שְׁלֹשָׁה שִׁירִים

בְּעִנְיָן אֶחָד מְדַבְּרִים

וְקָבַעְתִּי כָּל אֶחָד מֵהֶם בְּדַלְתֵי חַדְרוֹ

לְמַעַן יַכִּיר כָּל אֶחָד מֵהֶם מַעֲלָתוֹ וְקִצּוּרוֹ

כִּי לֹא כְּצוּרִי צוּרוֹ

וְיַכִּיר מֵהַיּוֹם אֲשֶר עַד כֹּה לֹא הִכִּירוֹ

וְזֶה הַשִּיר אֲשֶר חִבַּרְתִּי בְּדִבְרֵי בוּז וּנְבָלָה

עַל הָאִישׁ אֲשֶר הָיָה לוֹקֵחַ בְּגָדַי בְּהַשְׁאָלָה

לִהְיוֹת לוֹ לְשַׁמָּה לְחֶרְפָּה וְלִקְלָלָה

וְרָבְצָה בוֹ כָּל הָאָלָה


  


נָבָל אֲשֶר תִּשְׁאַל וְלֹא תַשְׁאִיל

וּבְהוֹן אֲחֵרִים תִּהְיֶה נֶעְזָר

תַּחְמֹל עֲלֵי צֹאנָךְ וְאֵת כִבְשַׂת

אִישׁ רָשׁ תְּזַבֵּחַ וְתִתְאַכְזָר

לוּ יֻתְּנוּ עָרְלוֹת בְּהַשְׁאָלָה

אָז תִּהְיֶה בּוֹעֵל בְּעָרְלַת זָר


  


וְזֶה הַשִּׁיר אֲשֶר חִבַּרְתִּי עַל הָאִישׁ אֲשֶר בַּסֵּתֶר יִקָּחֵם

וְהוּא אַכְזָרִי וְלֹא יְרַחֵם


  


רְאוּ נָבָל יְהִי רַע אַחֲרִיתוֹ

וְשָׁמַיִם יְגַלּוּ אֵת עֲוֹנוֹ

אֲשֶר לוֹ צֹאן וְכִבְשַׂת רָשׁ יְאַוֶּה

לְקַחְתָּהּ לוֹ בְּחֶמְלָתוֹ לְצֹאנוֹ

וְלוֹ עָרְלָה וּמִיִרְאַת בְּלוּתָהּ

יְאַו לִבְעֹל בְּמוֹ עָרְלַת שְׁכֵנוֹ


  


וְזֶה הַשִּׁיר אֲשֶר חִבַּרְתִּי עַל הָאִישׁ בַּעַל רִיב וּמָדוֹן

אֲשֶר הָיָה עוֹשֶׂה בְּזָדוֹן

נָבָל אֲשֶר כְסוּתִי וּמִנְעָלַי

תִּלְבַּשׁ וְתִנְעַלֵם לְחֶרְפָּתִי

הוֹאֵל בְּזֹה נֶפֶשׁ מְתָעֵב גּוֹי

לִבְעֹל חֲשׁוּקָתָךְ בְּעָרְלָתִי


  


וַיְהִי כַּאֲשֶר אָכְלוּ הָאֲנָשִׁים הָהֵם

מִפְּרִי דַרְכָּם וַיִשְׂבְּעוּ מִמּוֹעֲצוֹתֵיהֶם

וַיְקָרְבוּ כַתַּנּוּר לִבָּם בְּאָרְבָּם

וַיֹּאמְרוּ אִישׁ אֶל אָחִיו הָבָה נִתְחַכְּמָה

וְנַעֲשֶׂה מֵהָאִישׁ הַזֶּה בַּעַל הַשִׁירִים נְקָמָה

כִּי אִם גָּבְרָה יָדוֹ עָלֵינוּ בְשִׁירוֹת חֲמוּדוֹת

הֲלֹא לָנוּ בְחָכְמוֹת נִכְבָּדוֹת וּבַתְּעוּדוֹת

וּבְטֶבַע הַמְּצִיאוּת וּבִקְצַת סוֹדוֹת

עֶשֶׂר יָדוֹת

וְעַתָּה נָבֹא לְנַסּוֹתוֹ בְחִידוֹת

וְנִשְׁאָלֵהוּ בְחָכְמַת הַטֶּבַע

וּמַה שֶאַחַר הַטֶּבַע

וּבְמַהוּת הַזְּמָן וְהַתְּנוּעָה וְהָרֵיקוּת וְהַמָּקוֹם

אוּלַי חֶרְפָּתֵנוּ נִקֹּם

וַיִּוָעֲדוּ שְׁלֹשֶת הָאֲנָשִׁים הַנִּזְכָּרִים

וַיָּבֹאוּ לְיוֹם נוֹעָד כְּאִישׁ אֶחָד חֲבֵרִים

וַאֲנִי הָיִיתִי אָז עִם הָשָּׂר בְּמוֹשַׁב זְקֵנִים וְישִׁישׁים

וְשׁוֹפְטֵי אֶרֶץ וְחַכְמֵי חֲרָשִׁים

וַיֹּאמְרוּ הִכּוֹן וְהָכֵן לְךָ כִּי לְבַעֲבוּר נַסּוֹת אוֹתְךָ בָאנוּ

וָאַעַן וָאֹמַר אֶל מִי מִכֻּלָּנוּ

וַיֹּאמְרוּ אֵלֶיךָ הַמְשׁוֹרֵר הַסּוֹרֵר

וּמְדָנִים מְעוֹרֵר

כִּי תִשְׂתָּרֵר עָלֵינוּ גַּם תִּשְׂתָּרֵר

אָז אָמַרְתִּי אֶל הַנְּכָאִים

אֲרוּרִים הַבָּאִים

חֵי אָנִי אַחֲרֵי אֲשֶר לֹא שָׂבְעָה נַפְשְׁכֶם בָּרָעוֹת

הַיּוֹם אֶפְרַע בָּכֶם פְּרָעוֹת

וַיִּגַּשׁ הָרִאשׁוֹן וַיַּחֲשׁב כִּי חֶרְפָּתוֹ וְחֶרְפַּת חֲבֵרָיו יִקֹּם

וַיִּשְׁאָלֵנִי בְּמַעֲמַד כֻּלָּם עַל מַהוּת הַמָּקוֹם

וָאַעַן וָאֹמַר לְהָתֵל בּוֹ

וּלְהַכְאִיב לֵב חֲבֵרָיו וְלִבּוֹ

דְּבַר שְׁאֵלָתְךָ הַנִּבְזֶה מִבֶּטֶן מִי יָצָא

כִּי תִשְׁאַל עַל דָּבָר בִּלְתִּי נִמְצָא

וַיִּצְעַק הָאִישׁ לְיָרְאֵנִי וּלְהַבְהִילֵנִי

וַיֹּאמַר אֱלֹהִים מִזֶּה הַדַּעַת יַצִּילֵנִי

אֵיךְ תּוּכַל לְהַכְחִישׁ הַמֻּרְגָּשׁ

וְלַאֲמִיתּוּת דָּבָר לֹא תִגַּשׁ

מִי יוּכַל לְהַכְחִישׁ הַדָּבָר הַנִּרְאֶה לָעָיִן

וְלַחְשֹׁב מִן הַיֵּשׁ אָיִן

וָאַעַן וָאֹמַר חַיֵּיכֶם חֲשַׁבְתֶּם כִּי נִמְלַטְתֶּם מִן הַפַּחַד וּנְפַלְתֶּם בַּפַחַת

כִּי בְאָמְרִי אֵין הַמָּקוֹם נִמְצָא אָשִׂים טַעֲנָתִי נוצַחֲת

כִּי אִם הוּא נִמְצָא אוֹ מֻשְׂכָּל אוֹ מֻרְגָּשׁ וְזֹאת חֲלֻקָּה מֻכְרַחַת

וּמִן הַשֶּׁקֶר שֶׁיִּהְיֶה הַמָּקוֹם מֻשְׂכָּל

כִּי הַמֻּשְׂכָּל הַשִּעוּר וְהַכַּמּוּת מִמֶּנּוּ יֻסְכָּל

וְאִם הוּא מֻרְגָּשׁ אֵיךְ הַגֶּשֶׁם לְהִכָּנֵס בְּגֶשֶׁם יוּכָל

אִם כֵּן אֵין הַמָּקוֹם מָצוּי

וּבְהֶכְרַח מַאֲמָרִי רָצוּי


  


וָאֶשָּׂא מְשָׁלִי וָאֹמַר:


  


הַמָּקוֹם מֻשְׂכָּל אוֹ מֻרְגָּשׁ

מִן הַשֶּׁקֶר לִהְיוֹת מֻשְׂכָּל

כִּי הַמֻּשְׂכָּל הַשִּׁעוּר אוֹ

הַכַּמּוּת מִמֶּנּוּ יֻסְכַּל

אִם הוּא מֻרְגָּשׁ אֵיךְ הַגֶּשֶׁם

לִכָּנֵס בַּגֶּשֶׁם יוּכַל


  


וְהִנֵּה לְךָ מוֹפֵת אַחֵר

אִם תִּהְיֶה הָאֱמֶת בּוֹחֵר

הוּא שֶׁצָרִיךְ אַתָּה לָדַעַת אִישׁ חֶרְמִי

כִּי כָל נִמְצָא גַשְׁמִי צָרִיךְ אֶל מָקוֹם גַּשְׁמִי

אִם כֵּן הַמָּקוֹם נִמְצָא גַּם הוּא צָרִיךְ אֶל מָקוֹם אַחֵר

עוֹבֵר לַסּוֹחֵר

וְהָאַחֵר לְאַחֵר

וְכֵן יֵלֵךְ לְלֹא תַכְלִית וָחֵקֶר

וּמְצִיאוּת גוּפִים בַּעֲלֵי תַכְלִית לְלֹא תַכְלִית שֶׁקֶר


  


וָאֶשָּׂא מְשָׁלִי וָאֹמַר:


  


כָּל נִמְצָא גַשְׁמִי צָרִיךְ אֶל

מָקוֹם גַּשְׁמִי יִמָּצֵא בוֹ

אִם כֵּן אִם הַמָּקוֹם נִמְצָא

הוּא צָרִיךְ מָקוֹם לִמְסִבּוֹ

גַּם יֵלֵךְ זֶה אֶל לֹא תַכְלִית

הוּא נִמְנָע עַל שָׁוְא מַצָּבוֹ

אִם כֵּן אין הַמָּקוֹם נִמְצָא

עַתָּה אֵפוֹא דוֹד מַה טִבּוֹ


  


וַיִּגַּשׁ הַשֵּׁנִי וַחֲנִית שְׁאֵלָתוֹ יָרַק

יְנוֹפֵף יָדוֹ וְשִׁינָּיו חוֹרֵק

וְאֵש מִפִּיו זוֹרֵק

וַיִּשְׁאָלֵנִי בְדִבְרֵי הָרֵיקוּת לְהוֹרוֹת כִּי הוּא רֵיק

וּבִרְאוֹתִי כִּי הוּא בְּעֵינָיו אַרְכוֹן

עֲנִיתִיו כַּאֲשֶר עָנִיתִי לַחֲבֵרוֹ הֵפֶך הַנָּכוֹן

וְאָמַרְתִּי כִּי הָרֵיקוּת נִמְצָא מְצִיאוּת חָזָק

יִתְנוֹעֲעוּ בוֹ הַהֲוָיוֹת רָצוֹא וָשׁוֹב כְּמַרְאֶה הַבָּזָק

וְהָיָה כִּי יִפָּלֵא מִמְּךָ דְבָרִי מִמְּעוֹן בִּינָתְךָ הַשְׁקֵף

כִּי בִּלְעֲדֵי הָרֵיקוּת לֹא יִהְיֶה נִמְצָא גַלְגַּל יוֹצֵא הַמֶּרְכָּז וְגַלְגַּל הַהֶקֵּף


  


וָאֶשָּׂא מְשָׁלִי וָאֹמַר:


  


הָרֵיקוּת נִמְצָא גַּם גַּלְגַּל

עַל דָּבָר לֹא קַיָּם יַקֵּף

מוֹפֵת זֶה הַגַּלְגַּל יוֹצֵא

הַמֶּרְכָּז גַּם גַּלְגַּל הֶקֵּף


  


וַיִּגַּשׁ הַשְּׁלִישִׁי וְהוּא הָיָה בַּעַל חָכְמַת הַלִּמּוּדִים

וַיֹּאמַר אֶרְאֶה דְבָרֶיךָ בָתִּים לְבַדִּים

וְטַעֲנוֹתֶיָך עֲקֻדִּים נְקֻדִּים

הַיּוֹם תִּהְיֶינָה חִידוֹתַי לְךָ לְשָׂטָן

וְתַחַת אֲשֶר חָשַׁבְתָּ לִהְיוֹת גָּדוֹל תִּהְיֶה בְעֵינֶיךָ קָטָן

וַיִּשְׁאָלֵנִי חִידָה נִפְלָאָה

בְּפַז הַחֵן מְסֻלָּאָה

וְכַאֲשֶׁר אוֹתָהּ שָׁמַעְתִּי

כִּמְעַט אָחוֹר נִרְתַּעְתִּי

וְלוּלֵא יָדַעְתִּי

מֵחָכְמַת הַתְּכוּנָה מַה שֶיָּדַעְתִּי

אָז בִּיוֵן מְצוּלַת הַשְּׁאֵלָה טָבַעְתִּי

וְקָלוֹן מִכָּבוֹד שָׂבַעְתִּי

וְזֹאת הִיא הַשְּׁאֵלָה הַמְהֻלָּלָה

אֲשֶרשָׁאַל בַּעַל הָאֲבֵלָה

אָמָר אֱמֹר נָא הַמְשׁוֹרֵר אוֹיְבֵנו

הַזּוֹרֵק הָאֵשׁ בָּעִיר לְעֵינֵינוּ

וְעַל מִצְחוֹת אֲנָשִׁים נִכְבָּדִים הִתְוִיתָ תָו

מַה נִשְׁתַּנָּה צֶדֶק לִהְיוֹת מוֹרֶה עַל בָּתָּיו

אָז הַתְמַהְמַהְתִּי כְרֶגַע וְאָמַרְתִּי אֵין זֶה מִמִּפְלְאוֹת הַכְּתָב וְהַמִּכְתָּב

וְלָכֵן הַסְתֵּר נִצָּנֶיךָ כִּי הִנֵּה הַסְּתָו

וְלוּ הִשְׂכַּלְתָּ אָז יָדַעְתָּ כִּי כֵן כּוֹכָב כֹּתֵב הַכָּף כֹּתַב בַּתָּיו בֵּית תָּיו


  


וָאֶשָּׂא מְשָׁלִי וָאֹמַר:


  


אָמְרוּ מַה נִשְׁתַּנָּה צֶדֶק

לִהְיוֹת מוֹרֶה הוּא עַל בָּתָיו

אָמַרְתִּי כֵּן כּוֹכָב כֹּתֵב

הַכַּף כֹּתֵב בַּתָּיו בֵּית תָּיו


  


כִּשְׁמֹעַ הָאִישׁ אָבְדוּ עֶשְׁתֹּנוֹתָיו

וּבִרְאוֹתוֹ כִּי לֹא הוֹעִילוּהוּ חָכְמוֹתָיו

בָּטַח בְּחִידוֹתָיו

וַיֹּאמַר וּמַה הוּא הַכֶּלֶב אֲשֶר לֹא יֶחֱרַץ לְשׁוֹנוֹ

וְשָׂם בֵּין כּוֹכָבִים קִנּוֹ

יִנְשֹׁךְ וְלֹא יִזְעָף

יָרוּץ וְלֹא יִיגַע יֵלֵךְ וְלֹא יִיעַף

נִלְאוּ הָרִים שְׂאֵתוֹ כַלְכֵּל

וְאֵשׁ מִפִּיו תֹּאכַל

הַלַּיְלָה לוֹ לְיוֹמָם

וַאֲשֶר יִזְרֹבוּ וַיִּצְמְתוּ בְחֻמּוֹ נִדְעֲכוּ מִמְּקוֹמָם

וָאַעַן וָאֹמַר הוּא כוֹכַב הַכֶּלֶב

אֲשֶר לֹא יֹאכַל לֹא דָם וְלֹא חֵלֶב


  


וָאֶשָּׂא מְשָׁלִי וָאֹמַר:


  


כֶּלֶב קְרָא לַכְּלָבִים וְאֶל דָּג

שׁוֹכֵן בְּלֵב-יָם לְדָגִים לְשָׂטָן

כֶּלֶב קְרָא עוֹד בְּשִׁתּוּף לְכוֹכָב

מַזְהִיר אֲשֶר בֵּין תְּאוֹמִים וְסַרְטָן


  


וְכַאֲשֶׁר הָיָה שָׁם כֶּלֶב מְשֻׁתָּף לְאֵלֶּה הַשְּׁלֹשָׁה

כֵּן הָיָה סַרְטָן מְשֻׁתָּף לְדָג וּלְמַזָּל וּלְמַחֲלָה קָשָׁה


  


וָאֶשָּׂא מְשָׁלִי וָאֹמַר:


  


סַרְטָן נִקְרָא שֵׁם הַנֶּגַע

גַּם שֵׁם זֶה עַל דָּג הִתְגַּלְגָּל

עוֹד אֶל צוּרָה מֵאַרְבָּעִים

וּשְׁמֹנֶה צוּרוֹת הַגַּלְגָּל


  


וְהִנֵה לְךָ שִׁיר אֶחָד עַל זֶה הָעִנְיָן

לְמַעַן תֵּדַע כִּי הַשִּׁיר וְהַחָכְמָה לִי לְקִנְיָן


  


וָאֶשָּׂא מְשָׁלִי וָאֹמַר:


  


קְרָא סַרְטָן לְסַרְטָנֵי יְאוֹרִים

וְאֶל נֶגַע בְּשֵׁם סַרְטָן מְקֹרָא

וְכֵן הוּא שֵׁם לְמַזָּל בֵּין תְּאוֹמִים

וּבֵין אַרְיֵה הֲכִי יִשְׁכֹּן וְיִירָא


  


וַיְהִי כִּשְׁמֹעַ הָאֲנָשִׁים מַעֲנַי

וְשִׁירֵי רְנָנַי

זָחֲלוּ מֵחֲמַת פְּתָנַי

וְלֹא יָכְלוּ לַעֲנוֹת אוֹתִי כִּי נִבְהֲלוּ מִפָּנַי

וְהַזְּקֵנִים וְהַחֹרִים

הָיוּ שִׁירוֹתַי מְפָאֲרִים

וְרַק מֵאֹדוֹתַי מְדַבְּרִים

וַיְהִי בְתוֹכֵינוּ אִישׁ

הָיָה לוֹ בֵן מַחְפִּיר וּמַבְאִישׁ

וְהָיָה נְעִים קוֹל וּשְׁמוֹ אַבְרָהָם

יַעֲמֹד בֵּין הָאֲנָשִׁים כְּאִיש נִדְהָם

וְהָיָה לְאָלָה בְקֶרֶב עַמָּיו

גֶּבֶר לֹא יִצְלַח בְּיָמָיו

וְנָתַן לוֹ אָבִיו אִשָּׁה וּשְמָהּ אֶסְתֵּר

וּמַעֲשֶׂיהָ כְמַעֲשֶׂה זֶרֶשׁ אָכוֹל וְהוֹתֵר

תִּסְתַּתֵּר בְּעֶת אָכְלָהּ

וְתַשְׁמִיעַ בַּחוּץ קוֹלָהּ

שִׁית זוֹנָה וּנְצוּרַת לֵב

תְּבַשֵׁל גְּדִי בַּחֲלֵב

וַיֹּאמֶר אֵלַי הָאִישׁ חַיֶּיךָ אָבִיהָ שֶׁל כַּלָּתִי חָשַׁב בְּדַעְתּוֹ כִּי רִמָּנִי

בְּתִתּוֹ בִתּוֹ לִבְנִי וְכָזֹאת וְכָזֹאת הֶרְאָנִי

וְגַם אָמְנָה כִּי הֶלְאָנִי הֱשִׁמָּנִי

וּלְמַעֲצֵבָה הִשְׁכִּיבָנִי

כַּאֲשֶר מַטָּרָה לְחִצֵּי בִתּוֹ הִצִּיבָנִי

אָמְנָם לֹא רִמַּנִי

אֶלָּא רִמִּיתִיו

וְלֹא הֶלְאַנִי

אֶלָּא הֶלְאֵתִיו

וְלֹא הוֹגִיעַנִי

אֶלָּא הוֹגַעְתִּיו

כִּי נָתַתִּי לְבִתּוֹ עֵגֶל יָרָבְעָם

חֶרְפַּת אָדָם וּבְזוּי עָם

וּבְחֵפֶץ גָּדוֹל הָיִיתִי שׁוֹלֵחַ לוֹ שִׁירָה

תִּהְיֶה מִזֶּה הָעִנְיָן מְדַבְּרָה

אֲשֶר לְתִתִּי לוֹ בְשׂרָה

וָאַעַן וָאֹמַר חַיֶּיךָ פָּגַעְתָּ הָאִישׁ אֲשֶר אַתָּה מְבַקֵּש

הִנְנִי בְשִׁירִי אוֹדִיעֶנּוּ כִּי אַתָּה נִפְתָּל אִם הוּא עִקֵּש

וּבְפֹעַל כַּפָּיו רָשָׁע נוֹקֵשׁ

וְאִם הִשְׁקְךָ קֻבַּעַת כּוֹס חֲמָתוֹ

לָקַח מִיָּדְךָ כִּפְלַיִם בְּכָל חַטָּאתוֹ


  


וָאֶשָּׂא מְשָׁלִי וָאֹמַר:


  


בִּתָּךְ צְבִי חֵן הָיְתָה לִי טֹרַח

אֶסְתֵּר חֲשַׁבְתִּיהָ וְהִנָּה זֶרֶשׁ

חַי הַבְּרִית תַּחַת מְמוּכָן תֶּרֶשׁ

תַּתִּי וְתַחַת אַהֲרֹן לְךָ קֹרַח


  


אַבְרָם חֲשַׁבְתוֹ אַב הֲמוֹן הוּא תֶרַח

וּלְבִתְּךָ תַּתִּיו מְרַגֵּל חֶרֶשׁ

בִּתָּךְ עֲלֵי יִפְעָתְךָ דוֹד פֶּרֶשׁ

וּקְרָא שְׁמָהּ אִם לֹא בְרִיעָה סֶרַח


  


חֵי נַפְשְׁךָ דוֹדִי בְּיוֹם הֻלֶּדֶת

שָׂטָן בְּקִרְבֶּךָ שְׁמוֹנִים אֶלֶף

רֵעָיו סְבִיבוֹת אָהֳלַי נִצָּבוּ


  


נַפְשִׁי בְמִדָּתָךְ לְךָ מוֹדֶדֶת

אִם אִתְּךָ מִרְמוֹת וְאִתִּי חֵלֶף

מִרְמָתְךָ מִרְמוֹת לְרִיב נִצָּבוּ


  


וַיֹּאמְרוּ הָאֲנָשִׁים הַיּוֹשְׁבִים שָׁם בֶּאֱמֶת בֶּן הָאִישׁ הַלָּזֶה

הִכַּרְנוּהוּ כְבָר לְנִבְזֶה

וְאָסַף כָּל מִדּוֹת רָעוֹת כְּעוֹרֵב

אֶלָּא שֶׁקּוֹלוֹ עָרֵב

וְאִם שָׁלַל מִמֶּנּוּ הַזְּמָן מַעֲלוֹת גְּדוֹלוֹת

חָלַק לוֹ הַמַּעֲלָה מִן הַמַּעֲלוֹת הַמְהֻלָּלוֹת

וָאַעַן וָאֹמַר דְּעוּ אֶל נָכוֹן כִּי נְעִימוּת קוֹלוֹ

הוּא אוֹת אֱמֶת עַל חֶסְרוֹן שִׂכְלוֹ

כִּי לִהְיוֹתוֹ נָבוּב קוֹלוֹ יֶעֱרָב

וְיָבֵשׁ מְקוֹר שִׂכְלוֹ וּמַעְיָנוֹ חָרָב

וְנָפַל מִמֶּנּוּ רָב

כַּאֲשֶר נִרְאֶה הַתֻּפִּים וְהַכִּנּוֹרוֹת וְהַחֲלִילִים

וְהַנְּבָלִים וְהַמְנַעַנְעִים וְהַצֶּלְצְלִים

לוּלֵא הֱיוֹתָם חֲלוּלִים לֹא הָיוּ לָהֶם קוֹלוֹת נְעִימִים

וְהָיוּ כְלָבִים אִלְמִים

וְאָמַר הֶחָכָם דְּבָרוֹ אֵין לִשְׁכֹּחַ

כָּל עֲרֵבֵי הַקּוֹל הֵם חַסְרֵי מֹחַ


  


וָאֶשָּׂא מְשָׁלִי וָאֹמַר:


  


דּוֹדִים רְאוּ כִּנּוֹר וְנֵבֶל תֹּף נְבוּב

לוּחוֹת וְקוֹלָם יֶעֱרַב לִשְׂמֹחַ

לוּלֵי חֲלִילוּתָם אֲזַי יִדְמוּ כְלָ-

בִים אִלְּמִים לֹא יוּכְלוּ לִנְבֹּחַ

עַל זַה וְעַל כֵּן יֹאמְרוּ הַמּוֹשְׁלִים

כִּי כָל עֲרֵבֵי קוֹל חֲסֵרַי מֹחַ


  


וַיְהִי כִּרְאוֹת אַנְשֵׁי הַמַּעֲמָד

כִּי שִׁירִי לֹא יִסָּפֵר וְלֹא יִמָּד

הֵכִין כָּל אִישׁ שְׁאֵלָתוֹ

וְעָרֲךְ כָּל אֶחָד וְאֶחָד חִידָתוֹ

אָז לָבַשׁ שִׂכְלִי בִּגְדֵי תִפְאַרְתּוֹ

וְהֶרְאָה לַכֹּל יִפְעָתוֹ וְיִקְרָתוֹ

וָאֹמַר לְעֵינֵי הַשֶּׁמֶשׁ וּלְעֵינֵי הַשַּׂר אֲדֹנִי

הִנְנִי נִשְׁבָּע בִּגְאוֹנִי

כִּי לֹא אָשׁוּב אֶל בֵּיתִי וְאֶל מִשְׁכָּנִי

עַד אָפִיק שְׁאֵלַת כָּל שׁוֹאֵל

וְאֶהְיֶה לַשִּׁיר וְלַמְּלִיצָה גוֹאֵל

לְמַעַן יֵדַע כָּל מִתְגָּאֵל

כִּי לִי הַחָכְמוֹת בְּכֹחַ אֵל

וְהַשֵּׂכֶל לִשְׁכֹּן בְּאָהֳלֵי יוֹאֵל

וְהָיוּ מֻטּוֹת כְּנָפָיו מְלֹא רֹחַב אַרְצִי עִמָּנוּאֵל

אֲנִי טֶרֶם אֲכַלֶּה לְדַבֵּר

וְהִנֵּה אִישׁ עוֹבֵר

מִבְּנֵי יִשְׂרָאֵל בָּא

וַיֹּאמַר אַחֲרֵי אֲשֶר רְאִיתִיךָ נְבוּאוֹת הַשִּׁיר נִבָּא

וְהַשֵּׂכֶל בָּרַח מִמְּקוֹמוֹ וּבְכֵלֶיךָ נֶחְבָּא

חַבֵּר לִי שִׁיר עַל אַהֲבַת יְעֵלָה עֲדִינָה

לֹא רְאִיתִיהָ הֲרֵי שְׁתֵּים עֶשְׂרֵה שָׁנָה

וְאַחֲרֵי בְלוֹתִי הָיְתָה לִי עֶדְנָה

כִּי רָאִיתִי אוֹתָה אֶמֶשׁ בַּחֲלוֹם

וְאָמַרְתִּי בְּלִבִּי הִנְנִי עַתָּה בְעֵמֶק הַשָּׁלוֹם

וְהָיוּ רַעְיוֹנַי תְּמֵהִים

כְּאִיּלוּ נִפְתְחוּ הַשָּמַיִם

וְרָאִיתִי מַרְאוֹת גְּבֹהִים

וְכַאֲשֶׁר הֱקִיצוֹתִי וְלֹא מָצָאתִי הַחֲשׁוּקָה

הוֹדִי נֶהְפַּך עָלַי לְמַשְׁחִית וְנַפְשִׁי שׁוֹקֵקָה

וְשַׁבְתִּי קוֹדֵר גּוֹנֵחַ

אַחֲרֵי הֱיוֹתִי בַחֲלוֹמִי עָלֵז וְשָׂמֵחַ

וָאַעַן וָאֹמַר הִנֵּה הַשִּׁיר אֲשֶר תִּרְצֶה

וְהוּא מַבְרִיחַ אֶת חֲלוֹמְךָ מִן הַקָּצֶה אֶל הַקָּצֶה


  


וָאֶשָּׂא מְשָׁלִי וָאֹמַר:


  


מֶה עָרְבָה שֵׁנָה בְּחָלוֹם הֶרְאֲתָה

אֵלַי צְבִיָּה לֹא רְאִיתִיהָ הֲרֵי

שָׁנִים שְׁתֵּים עֶשְׂרֵה וְכִמְעַט אַחֲרָי-

הָ כָלְתָה נַפְשִׁי עֲדֵי כִּי סָפְתָה


  


וּבְעֵת חֲזִיתִיהָ בְחָזוֹן חָדְתָה

נַפְשִׁי וְנִפְתָּחוּ לְפָנַי שַׁעֲרֵי

עֵדֶן עֲדֵי נִגְלוּ לְפָנַי שַׁחֲרֵי

אוֹר מַלְאֲכֵי מָרוֹם וְרוּחִי רָפְתָה


  


שַׁחַר הֱעִירַנִי וְרוּחִי יֲהֲלֹם

יִפְעַת פְּנֵי חַמָּה בְאוֹרוֹ נִסְתְּרָה

יִפְעַת צְבִיַּת חֵן וְגִזְרַת יַהֲלוֹם


  


אוֹרוּ פְנֵי תֵבֵל וְנַפְשִׁי קוֹדְרָה

שָׁבָה בְּשׁוּרָה כִּי תְמוּנָה בַחֲלוֹם

רָאָה וְלָכֵן בַּהֲקִיצָהּ נֶעְכְּרָה


  


וַיַּרְא אִישׁ מֵאַנְשֵׁי הַמַּעֲמָד הַשִּׁיר אֲשֶר חִבַּרְתִּי

וְכִי מִקְרֵי הָעוֹלָם בְּשִׁיר עָרֵב סִפַּרְתִּי

וַיֹּאמַר גַּם אֲנִי לֹא אֶחְשֹׂךְ פִּי

דַּע שָׂרִי וְאַלוּפִי

כִּי לִי אִשָּׁה שְׁחַרְחֹרֶת

וּמֵאַהֲבַת הַחֹשְׁקִים שִׁכּוֹרֶת

וְלָהּ מַרְאָה בֵּין יָדֶיהָ

תַּשְׁקִיף תָּמִיד בְּעַד חַלּוֹנֶיהָ

וְתִקְרַע בַּפּוּךְ עֵינֶיהָ

תִּתְכַּבֵּס בַּנֶּתֶר וְתֵרָבֶּה לָהּ בּוֹרִית

וְתִסָּתֵר עִם הַחֹשְׁקִים בְּנַחַל כְּרִית

וְנִשְׁמַת חִשְׁקָהּ כְּנַחַל גָּפְרִית

וְעִם הַמַּרְאָה כָרְתָה בְרִית

וּבְכָל עֵת אֶמְצָאֶנָה בֵין יָדֶיהָ

כְּאִלוּ לֹא תוּכַל לִחְיוֹת בִּלְעָדֶיהָ

וּמֵרֹב אֲשֶר פָּנֶיהָ בְסֹלֶת תַּלְבִּין

תִּמְצָא בְרַקָּתָהּ תָּמִיד הַסֻּבִּין

וְהִיא יוֹדַעַת סֵפֶר וְתִקְרָא הַשִּׁירִים וְתָבִין

וְעַתָּה חַבֵּר שִׁירָה נִפְלָאָה

וַאֲנִי אֶכְתְּבֶנָּה סְבִיב הַמַּרְאָה

לְמַעַן תִּקְרָאֶנָּה וְתֵבוֹשׁ וְתִכָּלֵם

וְתִהְיֶה לָהּ חֶרְפּתַ עוֹלָם

וָאַעַן וָאֹמַר חַיֶּיךָ אֲנִי אֲחַבֵּר שְׁנֵי שִׁירִים

וְתִהְיֶינָה מֵאֹדוֹת הַיְּעֵלָה מְדַבְּרִים

וּשְׁנֵיהֶם תִּכְתֹּב סְבִיב הַמַּרְאָה

הַחִיצוֹן יְשַׁבַּח הַנָּאָה

וְהַפְּנִימִי יְגַנֶּה הַצּוֹלֵעָה וְהַנַּהֳלָאָה

וּבִהְיוֹת אִשְׁתְּךָ שָׁם נִשְׁקְפָה

תִּשְׂבַּע בְּחֶרְפָּה וּגְדוּפָה

אוּלַי תִּהְיֶה לְמַחֲלָתָהּ תְּרוּפָה

וָאָקוּם וָאֲחַבֵּר שִׁירָה עֲרֵבָה

כְּאִילּוּ תְדַבֵּר הַמַּרְאָה אֶל הַיָּפָה הַנִּשְׁקָפָה בָהּ


  


וְהִיא זֹאת:


  


כִּי יָגִיל כָּל חֹשֵׁק בִּי כָל

עָפְרַת חֵן תִּתֵּן עֵינֶיהָ

בִּי תַעְלֶה אֶת נֵרוֹת אוֹרָהּ

תָּאִיר אֶל עֵבֶר פָּנֶיהָ


  


וְעוֹד דִּבַּרְתִּי דִּבְרֵי חֶרְפָּה

אֶל הַכְּעוּרָה אֲשֶרבָּה מִתְיַפָּה


  


וְהִיא זֹאת הַשִּׁירָה:


  


מַה לָךְ פֹּה שִׁפְחָה נֶחְרֶפֶת

וּבְעַד אֶשְׁנַבִּי נַשְׁקֶפֶת

לֹא תַלְבִּינִי שַׁחְרוּת עוֹרֵךְ

עַד תַּלְבִּינִי אֶת הַזֶּפֶת


  


וַיִּשְׂמַח הָאִישׁ בִּרְאוֹתוֹ הַשִּׁירִים

וְיִהְיוּ יָדָיו לְכָתְבָם סְבִיב הַמַּרְאָה מְהִירִים

וְכִרְאוֹת הָאִשָּׁה הַשִּׁיר בָּהּ מְצַחֵק

וַתַּשְׁלֵךְ הַמַּרְאָה מִיָּדָהּ וַתֵּלֵךְ וַתֵּשֶׁב לָהּ מִנֶּגֶד הַרְחֵק

עוֹד זֶה מְדַבֵּר וְזֶה בָּא וַיֹּאמַר

הִנֵּה לְשָׁלוֹם מַר לִי מַר

כִּי נִקְרֵיתִי הַיּוֹם בְּחֻפַּת שְׁלֹמֹה יְדִידְיָה

וּמַעֲשֶׂה שֶהָיָה כָּךְ הָיָה

כִּי תַּחַת אֲשֶר דִּימִּינוּ שֶׁיִּהְיֶה כוֹסֵנוּ רְוָיָה

וְלֶאֱכֹל בַּרְבּוּרִים אֲבוּסִים

וְלִשְׁתּוֹת יֵינוֹת עֲסִיסִים

הָיוּ לָנוּ הַקְּטָטוֹת וְהַכְּעָסִים

וְצָרוֹת כַּמַּיִם לַיָּם מְכַסִּים

אָכַלְנוּ מַעֲדַנִים מְאוּסִים

וְשָׁתִינוּ יֵין חֲמָסִים

אֲשֶר יִלְאוּ הַלְּשׁוֹנוֹת לְסַפֵּר

וְתַחַת בְּשַׂר הַצְּבָיִם הָיָה לָנוּ שָׂעִיר לְכַפֵּר

וּבְלֹא חַג הַמַּצּוֹת אָכַלְנוּ מַצּוֹת וּמְרוֹרִים

וְהָיָה לָנוּ מִשְׁתֵּה שְׁמָרִים

וְכַאֲשֶׁר הִגִּיעַ הַתּוֹר

לְבָרֵךְ הֶחָתָן בְּפֶרַח אוֹ בְכַפְתּוֹר

רָאִינוּ מַעֲשֶׂה לִסְתּוֹר

כִּי אָמַר אָבִיו אַחֲרֵי רַבִּים לְהַטּוֹת

וּבֵרְכוֹ בִּשְׁנֵים עָשָׂר פְּרוּטוֹת

וְעַל הַכֹּל הָיָה צָרוּת רְאוּת הַכַּלָּה וְעֵינַיהָ תְרוּטוֹת

וְגַבּוֹת עֵינֶיהָ מְרוּטוֹת

וְעַתָּה חַבֵּר שִׁירָה נִכְבָּדָה

תְּדַבֵּר עַל עִנְיָן זֹאת הַסְּעֻדָּה

וָאַעַן וָאֹמַר חַיֶּיךָ אֲנִי אֲחַבֵּר וּקְלוֹנוֹ אוֹדִיעוֹ

וְלֹא תִשָּׁכַח מִפִּי זַרְעוֹ

וְזֶה הַשִּׁיר אֲשֶר חִבַּרְתִּי עָלָיו

לָתֵת לוֹ כִדְרָכָיו וְכִפְרִי מַעֲלָלָיו


  


מִשְׁתֵּה חֲתוּנוֹת שְׁלֹמֹה יְדִידְיָה

אַגִיד לְךָ דוֹד לְמַעַן תְּסַפֵּר

חֵלֶף עֲסִיסָיו וּבַרְבּוּר אֲבוּסָיו

מִשְׁתֵּה שְׁמָרִים וְשָׂעִיר לְכַפֵּר

אָמְרוּ יְבֹרָך בְכַפְתּוֹר וְפֶרַח

הִקְדִּים אֲבִיהוּ וּבֵרְכוֹ בִסְנַפֵּר


  


עוֹד זֶה מְדַבֵּר וְהִנֵּה אִישׁ בּוֹכֶה לְמִשְׁפְּחוֹתָיו

מָלְאוּ זַעַם שְׂפָתָיו

וַיֹּאמֶר לִפְלוֹנִי קְרוֹבוֹ

נָפְלָה עֲטֶרֶת רֹאשׁוֹ

כִּי בְנוֹ יְחִידוֹ אֲשֶר הָיָה יְדִידוּת נַפְשׁוֹ

לְקָחוֹ אֱלֹהִים וְאֵינֶנּוּ

וְהִנֵּה הוּא יִסְפְּדֶנּוּ וִיקוֹנֶנּוּ

יִגְעַר בַּמְנַחֲמִים

וַיֹּאמַר גָּדְלָה הַחֲרָדָה וְאֵין מָקוֹם לְנִחוּמִים

כִּי בְנוֹ הָיָה נֵזֶר הַיָּמִים

וְרָצִיתִי שֶׁתְּחַבֵּר שְׁתֵּי שׁוּרוֹת בָּם אֶדְרְשֶׁנוּ

וְהַיָּגוֹן אֲשֶר בְּלִבּוֹ יַשְׁרִישֶׁנוּ

אוּלַי אוּכַל נַכֶּה בוֹ וַאֲגָרְשֶׁנּוּ

וָאָקוּם וָאֲחַבֵּר אֵלָיו זֹאת הַשִּׁירָה הַנְעִימָה

וַיְשַׁלְחֶהָ אֵלָיו וַתְּהִי לוֹ לְנֶחָמָה


  


נִכְאָב לְבִנְךָ הֲתַעְכּוֹר שְׁאֵרָךָ

פֶּרַח זְמָן קוּם חֲבֹשׁ אֵת פְּאֵרָךְ

נָתַן אֲדֹנָי לְךָ בֵן וְהוּא גַם

לָקַח יְהִי שֵׁם אֲדֹנָי מְבֹרָךְ .


  


עוֹד זֶה מְדַבֵּר וְזֶה בָּא וַיֹּאמַר

יֵש לִי תַלְמִיד אֶחָד וְהוּא בֶן הַסּוֹכֵן

אֲלַמְּדֶנּוּ לְדַבֵּר צַחוֹת וְלֹא יָבִין לְדַבֵּר כֵּן

יוֹצִיא כָּל מִלָּה מִשָּׁרְשָׁהּ וּמִיסוֹדָהּ

אֲנִי אוֹמֵר מִזְמוֹר לְתוֹדָה

וְהוּא מִזְמוֹר לְסוֹדָהּ

אֲנִי אוֹמֵר קְרָשָׁיו וְהוּא קְרָסָיו

לֹא תוּכַל הָאָרֶץ לְהָכִיל אֶת כְּעָסָיו

אֲנִי אוֹמֵר שָׁחַת

וְהוּא תָחַת

אֶרְגָּז וְיִשְׂחַק וְאֵין נָחַת

וְאֲנִי שָׁמַעְתִּי עָלֶיךָ נוֹרָאוֹת

וְעַתָּה חַבֵּר שִׁיר אֲשֶר אֶל זֶה הָעִנְיָן יֵאוֹת

וָאָקוּם וָאֲחַבֵּר שִׁיר עַל הָעִנְיָן הַנִּזְכָּר

וָאֹמַר אֵלָיו אַל תַּבֵּט אַחֲרֶיךָ וְאַל תַּעֲמֹד בְּכָל הַכִּכָּר

הִנֵּה לְךָ הַשִּׁיר אֲשֶר תֶּאֱהָב

וּמִשְׁקָלוֹ עֲשָׂרָה זָהָב


  


אֶרְגָּז וְתִשְׂחַק כְּבַד פֶּה וְאֲנִי

אֶקְרָא קְרָשָׁיו וְעַתָּה קְרָסָיו

אוּלַי אֱלֹהִים יְצַו בַּחֲלוֹם לָךְ

עֵת יֹאמְרוּ לְךָ קְרָא שִָיו קְרָא סָיוֹ


  


עוֹד זֶה מְדַבֵּר וְזֶה בָּא

וַיֹּאמַר הֵחָלְצוּ מֵאִתְּכֶם אֲנָשִׁים לַצָּבָא

כִּי תְּמוֹל בִּהְיוֹתֵנוּ בְּבֵית הַתְּפִלָּה

וְהִנֵּה יַעֲקֹב הַמְשׁוֹרֵר הָאַשְׁכְּנַזִּי בַּכְּנֶסֶת עָלָה

וְהַחַזָּן רַבִּי מְשֻׁלָּם

בִּרְאוֹתוֹ אוֹתוֹ קָפָץ פִּיו כְּאִלּוּ נֶאֱלָם

ויַעֲקֹב הַמְשׁוֹרֵר בּוֹשׁ וְנִכְלָם

וְהָיְתָה לוֹ חֶרְפַּת עוֹלָם

וְלֹא יָדַע כִּי כֵן מֵאֵילֵי אַרְצֵנוּ צֻוֵּינוּ

לְבִלְתִּי נִתְפַּלֵּל רַק עִם הַתּוֹשָׁבִים הַיְדוּעִים אֶצְלֵנוּ

וַיִּחַר לְיַעֲקֹב מְאֹד וַיֵּרֵד מִבֵּית הַתְּפִלָּה

וַיֵּלֶךְ בְּזַעַף וַיְחַבֵּר מְגִלָּה

וְכָתַב בָּהּ עָלֵינוּ דִבְרֵי לָצוֹן

לֹא הָיָה לָנוּ לְרָצוֹן

וְאָמַר כִּי לֹא הִגַּעְנוּ לְרוֹעֵי צֹאן

וְקָבַע הַשִּׁירִים אֲשֶר חִבֵּר עָלֵינוּ בָאָרוֹן

לִסְפּוֹת עוֹד עַל חֲרוֹן אַפֵּנוּ חָרוֹן

וְעַתָּה הוֹשִׁיעֵנִי נָא מִיָּדוֹ

וְחַבֵּר שִׁיר יַבְהִילֶנוּ וְיַחֲרִידוֹ

וְהַגֶּד לוֹ פִּשְׁעוֹ וּמִרְדּוֹ

וָאַעַן וָאֹמַר אַל תִּרְאוּ מֵאִשּׁוֹ וְלַפִּידוֹ

כִּי פִּתְאֹם יָקוּם אֵידוֹ

וּתְשַׂחֲקוּ לְפִידוֹ

וְאִמְרוּ לָעָם הַבּוֹכֶה לְמִשְׁפְּחוֹתָיו

אַל תִּירְאוּ חֶרְפַּת אֱנוֹשׁ וְאַל תֵּחַתּוּ מִגְדוּפוֹתָיו

וְאִם בָּא עֲלֵיכֶם בְּלַהַב חֶרֶב וּבְרַק חֲנִית

וְחָרַשׁ עַל גַּבֵּיכֶם וְהֶאֱרִיךְ מַעֲנִית

הִנְנִי מֵשִׁיב אֶת צֵל מַעֲלוֹתָיו אֲחוֹרַנִּית

וּמִפָּנַי יֵלֵךְ קְדוֹרַנִית

וּמֵאֲשֶׁר חָטָא יִסְפֹּק עַל יָרֵךְ

וְתַחַת אֲשֶר קִלֵּל יְבָרֵךְ

וְיִקְרָא לִפְנֵיכֶם אַבְרֵך

וָאָקוּם וָאֲחַבֵּר שִׁיר אֲיָּלוֹת יְחוֹלֵל

יְבָרֵךְ וִיקַלֵּל

יְפָאֵר וְיַחֲנִיף

וְיָשִׂים עַל רֹאשֹׁו צָנִיף

וְלִפְעָמִים יָד רָמָה יָנִיף

וָאֶתְּנֶהוּ אֶל הָאִישׁ הַצּוֹעֵק חָמָס

וָאֹמַר לֵך וְהַרְאֵהוּ אֶל הָאַשְׁכְּנַזִּי וְאִם לִבּוֹ כְּלֵב הָאַרְיֵה הִמֵּס יִמָּס

וְאִם הָיָה עַד כֹּה רוּחַ אֱלֹהִים רָעָה מְבַעֲתוֹ

עַתָּה תִרְאֶה כִּי יִנָּחֵם מֵרָעָתוֹ

וַיֵּלֶךְ הָאִישׁ וַיַּעַשׂ כַּאֲשֶר צֻוָּה

וַיְהִי כִּרְאוֹת הָאַשְׁכְּנַזִּי הַשִּׁירָה הַנָּאוָה

קָם וְהִשְׁתַּחֲוָה

כִּי הָיְתָה נֶחְמֶדֶת לְהַשְׂכִּיל וְלָעֵינַיִם תַּאֲוָה

וְהָיָה בְעֵינָיו לְרַאֲוָה

וּבִקֵּשׁ כַּפָּרָה מֵאֲשֶׁר חָטָא

בְּפֹעַל וּבְמִבְטָא

וַיְחַבֵּר שִׁירָה אֲרֻכָּה

וַיַהֲפֹךְ בָּהּ אֶת הַקְּלָלָה לִבְרָכָה

וְזֹאת וְהִיא הַשִּׁירָה אֲשֶר חִבַּרְתִּי אֵלָיו

לָתֵת לְיַעֲקֹב כִּדְרָכָיו וּלְהָשִׁיב לוֹ כְמַעֲלָלָיו


  


נֵר מַעֲרָבִי מַה לְךָ נִזְעַקְתָּ

לָרִיב בְּצַדִּיקִים וְלֹא צָדַקְתָּ

בַּזְתָּ וְלָעַגְתָּ לְעִיר וְאֵם בְּיִשְׂ-

רָאֵל וְשָׁרַקְתָּ וְשֵׁן חָרַקְתָּ

לָמָּה צְבִי חֵן צָחֲקָה שִׁירָתְךָ

עַל עָם חֲשָׁקוּךָ וּבָם חָשַׁקְתָּ

לא מֵאֲשֶׁר קַצְנוּ בְחִשְׁקָךְ הַצְּבִי

מִבֵּית תְּפִלָּה יוֹם תְּמוֹל נִרְחַקְתָּ

כֹּה יַעֲשֶׂה לָנוּ אֱלֹהִים חַי וְכֹה

יוֹסִיף לְעַם בָּם כָאֱזוֹר דָּבַקְתָּ

אִם לֹא בְרִדְתָּךְ הַלְּבָבוֹת יָרְדוּ

אִתָּךְ וְנָדָדוּ בְּעֵת חָמָקְתָּ

הוֹאֵל צְבִי וּזְמֹר זְמִירֶךָ וְאֵת

חִקְקֵי זְמוֹרַת זָר אֲשֶׁר חָקַקְתָּ

פֶּן יִפְגְעוּ מוֹרִים חֲצָצֵיהֶם בְּךָ

פֶּן יִזְרְקוּ בָּך כַּאֲשֶר זָרָקְתָּ


  


עוֹד זֶה מְדַבֵּר וְהִנֵּה אִישׁ צוֹעֵק עַל עַבְדוֹ חָמָס

כִּי הָיָה אֵלָיו לְמִרְמָס

וַיֹּאמַר שְׁמָעֵנִי אָבִי הַמְּזִמָּה

אֵיךְ אֲקַוֶה לִי עוֹד נֶחָמָה

וְנָתַתִּי אֶל עַבְדִּי זֶרֶת וּבִקֵּשׁ אַמָּה

הֶאֱכַלְתִּיו דְּבַשׁ וְחָלָב

וּבִקֵּשׁ מָן וּשְׂלָיו

אָמַרְתִּי אֱכֹל וּבִקֵּשׁ לִשְׁתּוֹת

נָתַתִּי לוֹ יְרָקוֹת וּבִקֵּשׁ לְפָתוֹת

אָמַרְתִּי לוּן עַל אֲבוּסִי

וּבִקֵּשׁ לָנוּחַ עַל עַרְסִי

אָמַרְתִּי סוּרָה שְׁבָה פֹּה וּבִקֵּשׁ כַּר לָשֶׁבֶת עָלָיו

הִלְבַּשְׁתִיהוּ מְעִיל וּבִקֵּשׁ רִמּוֹנֵי תְכֵלֶת עַל שׁוּלָיו

וְעַתָּה חַבֵּר עָלָיו בְּחַסְדֶּךָ

אֶת אֲשֶׁר תִּמְצָא יָדֶךָ

וָאָקוּם קַל חִישׁ וָאֲחַבֵּר עָלָיו זֶה הַשִּׁיר

וּבִרְאוֹתוֹ אוֹתוֹ אָמַר בָּרוּךְ אֲדֹנָי וַאעְשִׁיר


  


כָּל הַקּוֹנֶה עֶבֶד עִבְרִי

קוֹנֶה אָדוֹן לוֹ בִּגְבוּלָיו

נָתַתִּי אֶל עַבְדִּי זֶרֶת

בִּקֵּשׁ אַמָּה לוֹ בִּמְשָׁלָיו

אָמַרְתִּי סוּרָה וּשְׁבָה פֹּה

בִּקֵּשׁ כַּר לָשֶׁבֶת עָלָיו

אָמַרְתִּי אָכְלָה מִצֵּידִי

בִּקֵּשׁ לִשְׁתּוֹת עִם טוּב מִלָּיו

אָמַרְתִּי לִין בִּקֵּשׁ חָצֵר

בַּרְזֶל וּנְחֹשֶׁת מִנְעָלָיו

אוֹכִילֵהוּ פַּת אַשְׁקֶנּוּ

יֵין רִקְחִי בִּקֵּשׁ מָן וּשְׂלָיו

הִלְבַּשְׁתִּיהוּ אֵפוֹד וּמְעִיל

בִּקֵּשׁ רִמּוֹנִים עַל שׁוּלָיו

לָכֵן אָרִים קוֹל כַּשּׁוֹפָר

מִטַּעַם מֶלֶךְ וּגְדוֹלָיו

לִקְנוֹת עִם הָעֶבֶד מַקֵּל

וּרְצוּעָה לִפְרִי מַעֲלָלָיו


  


עוֹד זֶה מְדַבֵּר וְהִנֵּה אִישׁ נִצָּב עַל יְמִינִי

וַיֹּאמַר אֲנִי הַגֶּבֶר רָאָה עֳנִי

כִּי אֶחְמֹד עֵדֶן גַּן אֱלֹהִים

וְרַעְיוֹנֵי אֵלָיו צְמֵאִים וּכְמֵהִים

וְתֵבֵל הָאֲרוּרָה תְסִיתֵנִי

וְלַעֲפַר מָוֶת תִּשְׁפְּתֵנִי

כִּי תִּמְשְׁכֵנִי בַעֲבוֹתוֹת הַבְלֶיהָ

וְתִקְשְׁרֵנִי בַחֲבָלֶיהָ

וְאִם אָשִׂים מוּל הָעֵדֶן מְגַמַּת פְּנֵי רַעְיוֹנִי

צוֹעֶקֶת אַחֲרַי מַה לְךָ פֹּה הֶעָנִי

אֵלַי תָּבוֹא כִּי שָׂכֹר שְׂכַרְתִּיךָ בְּדוּדָאֵי בְּנִי

וְעַתָּה אֲבִי הַמּוֹעֵצָה

שָׂא מָשָׁל וּמְלִיצָה

עַל זֹאת הָעֵצָה הַיְעוּצָה

וָאָקוּם וָאֲחַבֵּר לוֹ זֶה הַשִּיר הַנִּמְרָץ

בְּפִיו חָזוֹן נִפְרָץ

וַיִּיטַב הַשִּיר בְּעֵינָיו

וַיִּישָׂא חֵן וָחֶסֶד לְפָנָיו


  


נַפְשִׁי בְּתוֹךְ קִרְבִּי תְּסֹעַר כָּאֳנִי

תִּכְסַף לְעֵדֶן גַּן לְתִּתּוֹ מַחֲנִי

מָה אֹמְרָה דוֹדִים וְתֵבֶל אֹמְרָה

שָׂכֹר שְׂכַרְתִּיךָ בְּדוּדָאֵי בְּנִי


  


עוֹדֶנוּ מְדַבֵּר עִמִּי

וְהִנֵּה בָא אִישׁ צָדִיק תָּמִים

וַיֹּאמֶר אֵלַי אֲדֹנִי יֵשׁ בִּשְׁכוּנָתִי זָקֵן מָלֵא יָמִים

וְחָשַׁב כִּי יָשִׂים קִנּוֹ בַּמְרוֹמִים

וְהִנֵּה הוּא בוֹנֶה מִקְדָּשׁוֹ כְמוֹ רָמִים

כְּאִלּוּ יַחְשֹׁב לִחְיוֹת לְעוֹלָמֵי עוֹלָמִים

וּמִן הַתַּמַהּ כִּי לֹא יוּכַל מַלֵּט מַשָׂא

וּכְלָּבִיא יָקוּם וּכְאֲרִי יִתְנַשָּׂא

בּוֹנֶה הֵיכָל רָם וְנִשָּׂא

כַּאֲשֶר בְּעֵינָיו יָשָׁר

וְסָפוּן בָּאֶרֶז וּמָשׁוּחַ בַּשָָּׁשַׁר

אוֹמֵר אֶבְנֶה לִי בֵּית מִדּוֹת וַעֲלִיּוֹת

עָבְרוּ מַשְׂכִּיּוֹת

וּמִגְדָּל בָּנוּי לְתַלְפִּיוֹת

וּמִי שֶׁלֹא רָאָה בְּעֵינָיו כָּל הַמַּעֲשִׂים

בֵּית הַמְבַשְּׁלוֹת וְאֻרְוַת הַסּוסִים

וְהַמֶּרְחָץ וּמִקְוֵה הַמַּיִם

וּמִגְדָּל וְרֹאשׁוֹ בַשָּׁמָיִם

וְטַבְּעוֹתָיו וּנְזָמָיו

וּגְבִיעָיו וְחוֹתָמָיו

וּפְנִינָיו וְיַהֲלוֹמָיו

לֹא רָאָה שִׂמְחָה מִיָּמָיו

וְאִלּוּ הָיָה לוֹ בֵן אוֹ בַת יַשְׁאִיר אַחֲרָיו

אָמַרְתִּי אַשְׁרָיו

אָמְנָם לֹא נִין לוֹ וְלֹא נֶכֶד בְּעַמָּיו

וְאֵין שָׂרִיד בִּמְגוּרָיו

וְאֵיך לֹא יֹאמַר בַּעַל נֶפֶשׁ הַשּׁוֹבֵבָה

לְמִי אֲנִי עָמֵל וּמְחַסֵּר אֶת נַפְשִׁי מִטּוֹבָה

וְאֵיךְ לֹא יַחְשׁב כִּי יִנְחֲלוּ זָרִים הוֹנוֹ וּרְכוּשׁוֹ

וְהוּא יִשָּׂא הָאֲשָׁמִים לְנַפְשׁוֹ

הֲלֹא יֵדַע כִּי מִכָּל הַהוֹן אֲשֶר צָבַר

לֹא יְלַוֵּהוּ מִמֶּנּוּ דָבָר

רַק בְּלוֹיֵי הַסְּחָבוֹת אֲשֶר בָּהֶם יִהְיֶה נִקְבָּר

וְהַבַּיִת וְהַהֵיכָל

יַנִיחַ לְמִי שֶׁיוּכָל

וּמִי יוֹדֵעַ אִם חָכָם יִהְיֶה אוֹ סָכָל

וְהוּא יִהְיֶה לְרִמָּה לְמַאֲכָל

וְלִרְאוֹת זֶה הַמַּעֲשֶׂה נִלְאֵיתִי כַלְכֵּל וְלֹא אוּכָל

וּמַה מְאֹד הָיוּ מְסַפְּרִים תְּהִלוֹתָיו

אִלּוּ בָנָה בַּשָּׁמָיִם מַעֲלוֹתָיו

וְעַתָּה אֲדֹנִי חַבֵּר שִׁיר עַל זֶה הַמַּעֲשֶׂה

וְלֹא תַּחְמֹל עָלָיו וְלֹא תְכַסֶּה

אַחֲרֵי אֲשֶר תּוֹעֶה כַשֶּׂה

וָאָקוּם וָאֲחַבֵּר עָלָיו זֶה הַשִּׁיר הַנֶעֱרָב

וַיִּשְׂמַח הָאִישׁ כְּמוֹצָא שָׁלָל רָב


  


מַה תֹּאמְרוּ עַל אִישׁ יְגוֹנוֹת נִכְּלוּ

אֵלָיו וְאוֹתוֹ הַזְּמַן יוֹם יוֹם יְכַל

כִּי יֶאֱכֹל הַיּוֹם וּמָחָר מַאֲכָל

אֵיךְ יִהְיֶה הָעֵץ קְצוֹתָיו אֻכְּלוּ


  


לֹא כֵן אֲשֶר יָמִים עֲצָתוֹ סִכְּלוּ

אֵיכָה יְחִידָתוֹ לְמִגְדַּל עֹז תְּכַל

לִבְנוֹת וּמָחָר מֵת וְלֹא הִשְׂכִּיל סְכַל

אִישִׁים אֲשֶר בַּדָּיו בְּשַׁחַת יִפְּלוּ


  


אֵיךְ יַחֲרֹץ חָרוּץ קְנוֹת חָרוּץ ולא

יָשִׁיב אֱלֵי לִבּוֹ הֲאִם נַפְשׁוֹ יְלַו

הוֹן יוֹם עֲלֵי קֶבֶר וְגָדִישׁ שׁוֹקְדָה


  


לֹא יַהֲלֹךְ עִם הָאֱנוֹשׁ רַק פָּעֲלוֹ

לֹא הוֹן וְלֹא חָרוּץ יְכַלְכֵּל מַחֲלָיו

אִם תֶּחֱטָא נֶפֶשׁ תְּהִי עַל מוֹקְדָה


  


עוֹד זֶה מְדַבֵּר וְהִנֵּה אִישׁ חֹשֵׁק

יַחְפִּיר בְּיָפְיוֹ הַשֶּׁמֶש וְאוֹרוֹ עֹשֶׁק

וַיֹּאמַר אֲדֹנִי אֲנַשֵׁק רַגְלֶיךָ

אִם נְטַשְׁתַּנִי לְנַשֵּׁק

וְתַעֲשֶׂה לִי הַדָּבָר אֲשֶר אֶשְׁאַל

וְלֹא תָשִׂים מִלָּתִי לְאַל

כִּי אֵין זוּלָתְךָ לִגְאָל

וְלָכֵן אִם תִּגְאַל גְּאָל

וָאַעַן וָאֹמַר כְּאִישׁ תָּמֵהַּ

אֲדֹנִי לָמָּה זֶה אַתָּה לִשְׁאֹל מִתְמַהְמֵהַּ

הַעֲמֵק שְׁאָלָה אוֹ הֲגְבֵּהַּ

וַיֹּאמֶר הָאִישׁ אֲדֹנִי חָשֲׁקְתִּי בְשֶׁכְּבָר אַחַת מִבְּנוֹת הַנְּדִיבִים

וַיִּהְיוּ דֹדֶיהָ לִי עֲרֵבִים

הָיְתָה לִי לְעֶזְרָה בְצָרוֹתַי

וַחֲשַׁבְתִּיהָ סְגֻלַּת כָּל אוֹצְרוֹתַי

וְעַל עֲרָבִים בְּתוֹכָהּ תָּלִיתִי כִנוֹרוֹתַי

וְעַתָּה נִתְחַלְפָה הַשִּׁטָּה

וַיֵּצֵא לִי חוֹחַ תַּחַת חִטָּה

כִּי הֲרֵי חֹדֶשׁ יָמִים נָפְלָה לַמִּטָּה

וְנֶהְפַּךְ לְאֵבֶל שִׂמְחָתָהּ

כִּי קָרָה חֳלִי בְאַחַת מֵעֵינֶיהָ וְשִׁחֲתָהּ

וּמֵאָז לֹא יָסַפְתִּי עוֹד לְדַעְתָּהּ

וְעַתָּה חַבֵּר שִׁירָה נִמְרֶצֶת

תִּהְיֶה זֶה הָעִנְיָן מְקַבֶּצֶת

וְעַל פְּרָטֶיהָ רוֹבֶצֶת

וְלֹא הִתְמַהְמַהְתִּי כִּי אִם כְּרֶגַע

וָאֹמְרָה אֵלָיו נִרְפָּא הַנֶּגַע

הִנֵּה הַשִּׁיר עַל זֶה הָעִנְיָן

וְאִם אֵין בּוֹ רֹב מִנְיָן יֵשׁ בּוֹ רֹב בִּנְיָן


  


קַצְתִּי בְחֵשֶׁק יַעֲלַת הַחֵן

לִי הָיְתָה מִשְׁעַן וּמִשְׁעֵנָה

אֵיכָה בְּחִשְׁקָהּ אֶעֱלֹז עוֹד אוֹ

לִי תִהְיֶה מַשְׁעֵן וּמָשׁ עֵינָהּ


  


עוֹד זֶה מְדַבֵּר וְהִנֵּה אִישׁ בָּא

בְּנֶפֶשׁ דוֹאֲגָה וַעֲצוּבָה

כְּאִלוּ נִשְׁבַּר אוֹ נִשְׁבָּה

וְרוּחַ הַיְגוֹנִים בּוֹ נָשְׁבָה

וַיֹּאמַר אֲדֹנִי יֵש לִי בֵּן

וּשְׁמוֹ דָוִד וְהוּא שָׁמֵן וְדָשֵׁן

יִשְׁכַּב עַל מִטּוֹת שֵׁן

וְכָל הַיּוֹם וְכָל הַלַּיְלָה יָשֵׁן

ןְלֹא יְקִיצוּהוּ הַקּוֹלוֹת וְהַלַּפִּידִים וְקוֹל הַשּוֹפָר וְהָהָר עָשֵׁן

וְלֹא בְהֵמָה תְשַׁלַח בּוֹ הַשֵּׁן

שְׁנָתוֹ עַזָּה כַמָּוֶת

וְעַפְעַפָּיו צַלְמָוֶת

לֹא יֵעוֹר מִשְׁנָתוֹ וְלוּ נִרְצַח

אוֹ אַרְיֵה טֹרַף עַצְמוֹתָיו פִּצַּח

וְלוּ תִתְעָרֵב לִישֹׁן עִם הַמֵּתִים נֶצַח

וְעַתָּה חַבֵּר עָלָיו שִׁיר צַח

וָאַעַן וָאֹמַר הִנֵּה לְךָ שִׁיר נָאֶה וְרָטוֹב

דָבָר בְּעִתּוֹ מַה טוֹב


  


שְׁנַת דָּוִד שְׁנַת עוֹלָם

שֵׁנָה מִמָּוֶת לֹא הִבְדִּיל

לוּ הִתְעָרֵב עִם הַמֵּתִים

לִישׁוֹן עִמָּם דָּוִד הִגְדִּיל


  


עוֹדֶנוּ מְדַבֵּר עִמִּי

וְהִנֵּה אִישׁ קְרָאנִי בִשְׁמִי

וַיֹּאמֶר אלי אֲדֹנִי תְּמוֹל קָדַר הַשֶּׁמֶשׁ בַּחֲצִי הַשָּׁמַיִם

וַיִּמַּס לְבָב הָעָם וַיְהִי לְמָיִם

וַיְהִי כָל הָעָם נָדוֹן לָדַעַת סִבַּת קַדְרוֹ וְחָשְׁכּוֹ

וַיֹּאמְרוּ זֶה אוֹמֵר בְּכֹה וְזֶה אוֹמֵר בְּכֹה

וַיְהִי בְתוֹךְ הָעָם אִישׁ וּשְׁמוֹ יוֹחָנָן בֶּן קָרֵחַ

וַיֹּאמֶר כִּי סִבַּת קַדְרוּתוֹ הוּא הַיָּרֵחִ

אֲשֶר תַּסְתִּיר מִמֶּנּוּ אוֹר הַשֶּׁמֶשׁ הַזּוֹרֵחַ

וְקַדְרוּת הַלְּבָנָה סִבָּתָהּ צֵל הָאָרֶץ

הַפּוֹרֵץ בְּאוֹר הַלְּבָנָה פָרֶץ

וּבִהְיוֹת שִׂכְלִי עַל חוּג זֹאת הַמַחֲשָׁבָה סוֹבֵב

רָאִיתִי אִשָּׁה עֲדִינָה בְּעַד הַחַלּוֹן נִשְׁקָפָה וַתְּיַבֵּב

וְצוּף דִּבְשָׁהּ שִׂפְתֵי יְשֵׁנִים דּוֹבֵב

לֹא נִרְאָה כְיָפְיָהּ וְהוֹדָהּ

לְמִיּוֹם סוּר אֶפְרַיִם מֵעַל יְהוּדָה

אָז נוֹדַע לִי כִּי לְרֹב הוֹדָהּ הַנֶּאְדָּר

וְהַיֳפִי וְהַחֵן אֲשֶר בְּנִשְׁמָהּ דָּר

וַאֲשֶר כָּל הַהֲדּוּרִים בָּהּ אִישׁ לֹא נֶעְדָּר

בּוֹש הַשֶּׁמֶשׁ לְפָנֶיהָ וְקָדָר

וַאָחְשֹׁב כִּי קַדְרוּת הַלְּבָנָה הִיא סִבָּתָהּ גַּם כֵּן

וְלֹא צֵל הָאָרֶץ כַּאֲשֶר יֹאמַר הַמִּסְכֵּן

וְעַתָּה חַבֵּר לִי שְׁתֵּי שִׁירוֹת

תִּהְיֶינָה מֵאֹדוֹת דְּבָרַי מְדַבְּרוֹת

וָאָקוּם וָאֲחַבֵּר שְׁתֵּי שִׁירוֹת מְהֻלָּלוֹת

דִּבְרֵי זֶה הָאִישׁ כּוֹלְלוֹת

וְזֹאת הִיא הַשִּׁירָה הָאֲיֻמָּה

הַזּוֹכֶרֶת סִבַּת קַדְרוּת הַחַמָּה


  


יֹאמְרוּ עֲלֵי שֶׁמֶשׁ אֲשֶר קָדַר

כִּי הַלְּבָנָה הָיְתָה סִבָּה

לֹא הַלְּבָנָה כִּסְּתָה אוֹרוֹ

אַך אוֹר צְבִיָּה זָהֳרוֹ כִּבָּה

עָמְדָה עֲלֵי הַתְּלִי נֶגְדּוֹ

שָׁלְלָה שְׁלַל אוֹרוֹ וְשָׁב נִשְׁבָּה


  


וְזֹאת הִיא הַשִּׁירָה הָאַחֲרוֹנָה

הַמְדַבֶּרֶת עַל סִבַּת קַדְרוּת הַלְּבָנָה


  


לֹא הַתְּלִי גָרַם וְצֵל הָאָרֶץ

כִּי הַלְּבָנָה הָיְתָה קוֹדֶרֶת

אַךְ אוֹר פְּנֵי יַעְלָה מְאוֹרָה שָׁזְפָה

כִּי הִיא בְהִלָּהּ צַר וְאוֹר חוֹדֶרֶת


  


עוֹד זֶה מְדַבֵּר וְהִנֵּה אִישׁ מְשׁוֹרֵר

וְעַל אִישׁ מְדָנִים יְעוֹרֵר

וַיֹּאמַר חֲמָסִי עַל פְּלוֹנִי אוֹיְבִי

כִּי יִקַּח הַמְּלִיצוֹת אֲשֶר לָקַחְתִּי בְקַשְׁתִּי וּבְחַרְבִּי

וְהוּא מְיַחֵשׂ אוֹתָם עַל שְׁמוֹ זֶה הַנִּכְלָם

אֲשֶר לֹא עָמַל בַּם וְלֹא גִדְּלָם

וְהִנֵּה יָדַי לְךָ פְרוּשׂוֹת

לַעֲשׂוֹת בְּאוֹתוֹת וּבְמַסּוֹת

שֶיִּבָּצֵר מִמֶּנּוּ אֵת אֲשֶׁר יָזַם לַעֲשׂוֹת

וָאַעַן וָאֹמַר חַיֶּיךָ יַד חַסְדִּי מִמְּךָ לֹא אֶקְפֹּץ

הִנֵּה לְךָ הַשִּׁיר אֲשֶר תַּחְפֹּץ

וּבוֹ תִגְדֹּר פִּרְצֶךָ

וְתַגִּיעַ אֶל מְחוֹז חֶפְצֶךָ

וְלֹא יַחֲטִיא חִצֶּךָ


  


תַּעְדֶּה לְךּ עֶדְיָּי וְתִתְיַחֵשׂ

עַל מִשְׁפְּחוֹת שִׁירַי וְתִתְיַלֶּד

הִנְנִי בְשֵׁם חָדָשׁ אֲכַנֶּךָ

קוֹרֵא אֲשֶר דָּגַר וְלֹא יָלָד


  


עוֹד זֶה מְדַבֵּר וְזֶה בָּא וַיֹּאמַר

הוֹשִׁיעָה אֲדֹנִי כִּי חָסִיד גָּמַר

וְהַשֵּׂכֶל יִישַׂן וַיֵּרָדֵם

וּפַסּוּ בַעֲלֵי הַדִּקְדּוּק מִבְּנֵי אָדָם

וָאַעַן וָאֹמַר מַה לְךָ נִרְדָּם

שֶׁבַח לַאֲדֹנָי אֱלֹהֵי יִשְׂרָאֵל

יֵשׁ לַשֵּׂכֶל וּלְחָכְמוֹת הַדִּקְדּוּק גּוֹאֵל

וַיֹּאמֶר הָאִישׁ אֲדֹנִי הִתְוַכַּחְתִּי אֶתְמוֹל עִם פְּלוֹנִי הַמְדַקְדֵּק

וּרְאִיתִיו בְּכָל דְּרָכָיו צוֹדֵק

רַק בְּמַה שֶׁאָמַר כִּי לֹא תִמָּצֵא כְתוּבָה

וָו עִקָּרִית בְּרֹאשׁ הַתֵּבָה

כִּי הִיא מְשָׁרֶתֶת בָּרֹאשׁ לְעוֹלָם

בַּמִּלוֹת וּבַפְּעָלִים וּבַשֵׁמוֹת כֻּלָּם

וַאֲנִי שָׁמַעְתִּי אוֹמְרִים אַנְשֵׁי הַחָכְמוֹת

כִּי תִמָּצֵא וָו עִקָּרִית בְּרֹאשׁ קְצָת שֵׁמוֹת

וְרָצִיתִי שֶתְּחַבֵּר שִׁיר וְתַזְכִּיר מֵהֶם אַחַת אוֹ שְׁתַּיִם אוֹ שָׁלוֹש

וָאֶשְׁלַח אֶל הַמְדַקְדֵּק אוּלַי גְּאוֹן לִבּוֹ אֶחֱלוֹש

וָאָקוּם קַל חִישׁ וָאֲחַבְּרֶנוּ

וָאֹמַר הִנֵּה לְךָ הַשִּׁיר קָחֶנּוּ

וְאֶל הַמְדַקְדֵּק שְׁלָחֶנוּ

כִּי הוּא יְעִירֶנוּ מִשְׁנָתוֹ וּבַהֲקִיצוֹ יוֹכִיחֶנוּ


  


אֵיך תִּתְמַרְמֵר דּוֹד וַתֹּאמַר

בִּמְלֶאכֶת הַדִּקְדּוּק אַפְסִי

אִם נַעְלִים וַשְׁתִּי אוֹ וַשְׁנִי

אֵיךְ נַחְבִּיא בֶּן וָפְסִי


  


עוֹד זֶה מְדַבֵּר וְהִנֵּה אִישׁ בּוֹכֶה וּמְקוֹנֵן

וְאֵלַי בְּכָל עוֹז מִתְחַנֵּן

וַיֹּאמַר אֲדֹנִי בָּאתִי בְּנֶפֶשׁ נֶעֱצֶבֶת

כִּי בָאֲרָזִים נָפְלָה שַׁלְהֶבֶת

וּמַה יַעֲשׂוּ אֲזוֹבֵי קִיר

אִם הַזְּמָן נְדִיבֵי עַמִּים יַפְקִיר

וּכְהָקִיר בַּיָר מֵימֶיהָ כֵּן רָעָתוֹ יָקִיר

וְתַחַת הֱיוֹתֵנוּ בַמַּעֲלָה הָעֶלְיוֹנָה הִנְנוּ בַמּוֹרָד

וְאֵיךְ כָּל לֵב לֹא יִשּׂם וְיֶחֶרָד

וְלֹא יִהְיֶה מִמַּאֲוַיָּו נִפְרָד

וְיוֹמֵנוּ מְאֹד רָד

כִּי בְמִשְׁפַּחַת פְּלוֹנִית הֲקְּדוֹשָׁה

מֵתוּ בְשָׁנָה אַחַת אֲנָשִׁים שְׁלֹשָׁה

נְשִׂיאֵי מַטּוֹת אֲבוֹתָם

הָיוּ עַל לֵב הַזְּמַן חוֹתָם

וְעַד מָתַי יָשִׁיב הַזְּמַן צוּר חַרְבָּם

וּמַטָּרָה לְחִצָּיו הִצִּיבָם

וְיַשְׁקִיט אֲשֶר קֵרְבוּ כַתַּנּוּר לִבָּם

בְּאָרְבָּם לִטְמֹן עֲוֹנָם בְּחֻבָּם

וְעַד מָתַי תַּכֶּה עַמִּים בְּעֶבְרָה

חֶרֶב הַזְּמַן לֹא תֵאָסֵף אֶל תַּעְרָה

וְלָכֵן בָּאתִי אֵלֶיךּ בְּנֶפֶש מָרָה

לְחַבֵּר עַל זֶה הָעִנְיָן אֲמִירָה

כִּי יָדִי תִּקְצַר מֵהוֹשִׁיעַ

וְשָׁמַעְתִּי אוֹמְרִים כִּי אוֹר שִׂכְלְךָ יוֹפִיעַ

וְלַפּוֹשְׁעִים תַּפְגִּיעַ

וְכָל לָשׁוֹן תָּקוּם אִתְּךָ לַמִּשְׁפָּט תַּרְשִׁיעַ

וָאָקוּם וָאֲחַבֵּר הַשִּיר זָהָב עֶרְכּוֹ

וַיִּקָּחֵהוּ הָאִישׁ וַיֵּלֶךְ לְדַרְכּוֹ


  


חֶרֶב זְמַן בּוֹגֵד הֲלֹא תִשְׁקֹטִי

עַד אָז בְּנֵי מוּסָר בְּאַף תִּשּׂמִּי

שִׁפְכִי חֲמָתֵך עַל עֲדַת הַבּוֹעֲרִים

מִתּוֹךְ עֲדַת הָרוֹזְנִים הֲרֹמִּי

הִתְאַפְּקִי חֶרֶב זְמָן הֵאָסְפִי

אֶל תַּעֲרֵךְ הֵרָגְעִי וָדֹמִּי


  


עוֹדֶנוּ מְדַבֵּר וְהִנֵּה אִישׁ בָּא מֵרָחוֹק

וּפִיו יְמַלֵּא שְׂחוֹק

וַיֹּאמַר הֲתִרְצוּ לָדַעַת אֹדוֹתַי

הַיּוֹם רָאִיתִי שֶׁלֹּא רָאוּ אֲבוֹתַי וַאֲבוֹת אֲבוֹתַי

כִּי בִּהְיוֹתִי בְחֻפַּת סְעַדְיָה

אָמְרוּ עֲמִיתַי יוֹנָתָן וְצִדְקִיָּה

נִרְאֶה פְנֵי הַכַּלָּה אוּלַי נִצְטַיֵּד מִיָּפְיָהּ

וְכַאֲשֶׁר קָרַבְנוּ אֵלֶיהָ

וְהֵסַרְנוּ מֵעַל פָּנֶיהָ צְעִיפֶיהָ

צַר וָאוֹר חָשַׂךְ בַּעֲרִיפֶיהָ

כִּי צָפַד עַל עַצְמָהּ עוֹרָהּ

וְחָשַׂך מִשְׁחוֹר תָּאֳרָהּ

וַנִּקְרָא שֵׁם הַמָּקוֹם הַהוּא תַבְעֵרָה

בִּרְאוֹתֵנוּ פְּנֵי הַכַּלָּה כְּשׁוּלֵי קְדֵרָה

וְכַאֲשֶׁר קָם הַחַזָּן לוֹמַר הַתְּפִלָה

פָּתַח בְּיוֹצֵר אוֹר תְּחִלָּה

צָעַקְתִּי וַהֲרִימוֹתִי קוֹל כְּחוֹלֶה

וָאֹמַר אֶל הַחַזָּן אֵיךְ תֹּאמַר יוֹצֵר אוֹר בֶּן הָאֲבֵלָה

הֲשָׁכַחְתָּ פְּנֵי הַכַּלָּה

אֲשֶׁר עָשָׂה הַזְּמַן עִמָּהּ כָּלָה

אַךְ נִבְהָלָה

אֵין אוֹר וְאֵין שִׂמְחָה וְגִילָה

רַק מְעוּף צוּקָה וַאֲפֵלָה

וּבָרַחְתִּי מִשָּם בְּנֶפֶשׁ נִבְהָלָה

וְעַתָּה חַבֵּר שִׁיר כַּאֲשֶר עִם לְבָבֶךָ

וְהָרֵק עַל הַכַּלָּה הַנִּקְלָה חַרְבֶּךְ

וָאָקוּם וָאֲחַבֵּר זֶה הַשִּׁיר עַל שְׁמוֹ

וַיֵּלֶך וַיָּשָׁב הָאִישׁ לִמְקוֹמוֹ


  


אָמְרוּ עֲמִיתַי בֹּא וְנִרְאֶה אֵת

חָתָן וְכַלָהּ הֵם בְּתוֹךְ כִּילָה

וָאֵלְכָה עִמָּם וְהִנֵּה שָׁם

כַּלָּה וְחשֶׁךְ עַל פְּנֵי כֻלָּהּ

וָאֹמְרָה בָּרוּךְ אֲשֶר הֵאִיר

לַכֹּל בְּרֹב חַסְדוֹ וְהֶחְשִׁיךְ לָהּ

קָם איש וּפָתַח אָז בְּיוֹצֵר אוֹר

אַךְ בִּי הֲכִי עָשָׁן וְאַף עָלָה

קַמְתִּי בְּזַעַף אַף וְאָמַרְתִּי

דּוֹדִים הֲשָׁכַחְנוּ פְנֵי כַלָּה

אֵין אוֹר וְאֵין מָאוֹר תְּנוּ זִמְרָה

אֶל מַעֲבִיר יוֹם מִפְּנֵי לַיְלָה


  


עוֹדֶנוּ מְדַבֵּר וְהִנֵּה אִישׁ בָּא

וַיֹּאמַר אֲדֹנִי תְּמוֹל שִׁלְשֹם

נִקְרֵיתִי בְחֻפַּת גֵרְשֹׁם

וּלְרֹעַ צוּרָתוֹ כָּל רוֹאֵהוּ יֶחֱרַד וְיִשּׂם

כִּי הוּא כְּכוּשַׁן רִשְׁעָתָיִם

עַל פְּנֵי כָל הָאָרֶץ כְּאַמָּתָיִם

רוֹבֵץ בֵּין הַמִּשְׁפָּתָיִם

וּלְחָיָיו כִּלְחִי חֲמוֹר חֲמוֹרָתָיִם

וּלְשִׁנָּיו אֵין בְּרִיחַ וּדְלָתָיִם

וּפְנֵי אִשְׁתּוֹ כְּאוֹר הַחַמָּה שִׁבְעָתָיִם

וְעֶדְיָהּ צֶבַע רִקְמָתָיִם

וְכִרְאוֹתִי יִפְעַת הַכַּלָּה סְגוֹר לִבִּי תִקְרַע

וְזֻלּוּת בַּעְלָהּ הַמְּצֹרָע

אֲשֶר בֵּין רַגְלֶיהָ כָּרַע

אָמַרְתִּי בֶּאֱמֶת אָמַר הַנָּבִיא עַל זֶה הַמְּאֹרָע

יוֹצֵר אוֹר וּבוֹרֵא חשֶׁךְ עֹשֶׂה שָׁלוֹם וּבוֹרֵא רָע

וְהָאִישׁ אֲשֶר חִבֵּר זֶה הַחִבּוּר עוֹמֵד בְּמִי שֶפָּרָע

וְעַתָּה אַשְׁבִּיעֲךָ בָּאַיָּלוֹת אוֹ בַצְּבָאוֹת

לְחַבֵּר עַל זֶה הָעִנְיָן הַשִּׁיר אֲשֶר יֵאוֹת

וָאָקוּם וָאֲחַבֵּר זֶה הַשִּיר גִּזְרַת יַהֲלוֹם

וָאֶתְּנֵהוּ אֵלָיו וַיֵּלֶךְ מֵאִתִּי בְשָׁלוֹם


  


רְאוּ חָתָן פְּנֵי כִבְשָׂן כְּכוּשָׁן

וְזִיו כַּלָּה אֲשֶר יִזְרַח וְיֵאוֹר

וְהוֹדוּ אֵל אֲשֶר גּוֹלֵל לְאוֹר מִ-

פְּנֵי חשֶׁך וְחשֶׁךְ מִפְּנֵי אוֹר


  


עוֹדֶנוּ מְדַבֵּר וְהִנֵּה אִישׁ בִּשְׁמִי קְרָאָנִי

וַיֹּאמַר אֲדֹנִי שָׁמַעְתִּי שִׁירְךָ וְהֶנְאָנִי

וְהַדָּבָר אֲשֶר עָדֶיךָ הֱבִיאָנִי

הוּא רְאוֹתִי דָבָר אֲשֶר הֶלְאָנִי

כִּי רָאִיתִי בִּשְׁכוּנָתִי אִשָּׁה עִבְרִית

דּוֹמָה בְצוּרָתָהּ לְשִׁפְחָה מִצְרִית

וְנִשְׁמָתָהּ כְּנַחַל גָּפְרִית

וּפָנֶיהָ מוֹרִיקוֹת כְּאִלּוּ הַנֶּגַע שָׁם נָפָל

שְׁקַעֲרוּרֹת יְרְקְרַקּוֹת וּמַרְאֵיהֶן שָׁפָל

לֹא עָלָה עַל לֵב אָדָם בְּדוֹרָהּ

כִּי אִישׁ יְזָּקֵק לְצוּרָה כָזוֹ כְעוּרָה

וְהִנֵּה הִיא לוֹ לִזְנוּנִים הָרָה

אֶחְשֹׁב כִּי לֹא שָׂם בָּהּ קִּנּוֹ

רַק גִּבֵּן אוֹ דַק אוֹ תְבַלּוּל בְּעֵינוֹ

אוֹ נָפַל שָׁמָּה שׁוֹר אוֹ חֲמוֹר

וְעַתָּה אֱמֹר עָלֶיהָ שִׁיר אוֹ מִזְמוֹר

וָאָקוּם וָאֲחַבֵּר שִׁיר זֶה הָעִנְיָן יְסַפֵּר

אֲשֶר לֹא יִמַּד יָפְיוֹ וְלֹא יִסָּפֵרו

וְזֶה מִפְתַּח שְׂפָתָיו:


  


מִמִּי הָרִית שִׁפְחָה מִצְרִית

מֵהַמְּצֹרָע אוֹ מִגִּבֵּן

כָּעֵת חַיָּה לָךְ כָּל בִּרְיָה

תֹּאמַר חַיָּה חוֹבֶקֶת בֵּן


  


עוֹדֶנוּ מְדַבֵּר וְהִנֵּה אִישׁ הוֹלֵךְ וְקָרֵב

וַיֹּאמֶר אֵלַי אֲדֹנִי הֶעָרֵב

יֵש לִי מוֹדַע נֶאֱמָן

חוֹמֵל עָלַי וְרַחֲמָן

יְדַעְתִּיו אֲבִי הַתְּעוּדוֹת

וְרִאשׁוֹן לְכָל סֵבֶל הַסּוֹדוֹת

וְלוֹ בְּכָל חָכְמָה עֶשֶׂר יָדוֹת

חָלַץ מִן הַזְּמָן בִּתְבוּנָתוֹ

וְיָצָא נָקִי מֵאָלָתוֹ

וְהִשִּׂיגָה יָדוֹ וּמָצָא כְדֵי גְאֻלָּתוֹ

וְכִרְאוֹת הֶהָמוֹן אֶת מַעֲלָתוֹ

וְאֶת יְקָר תִּפְאֶרֶת גְּדוּלָתוֹ

וְכִי הַשֵּׂכֶל שָׂמָהוּ מֵאַנְשֵׁי סוֹדוֹ

וְאֶל אֲצִילֵי בְּנֵי יִשְׂרָאֵל לֹא שָׁלַח יָדוֹ

אָמְרוּ כִּי מָצְאוּ בּוֹ עֶרְוַת דָּבָר

כַּאֲשֶר אֶל אַנְשֵׁי הַפִּילוֹסוֹפְיָה הִתְחַבֵּר

נוֹאֲלוֹ נִשְׂאוּ לִמְּדוּ לְשׁוֹנָם דַּבֶּר-שֶׁקֶר הַעֲוֵה נִלְאוּ

הָלְכוּ קְדוֹרַנִּית בְּקִנְאָתָם אֲשֶר בּוֹ קִנְּאוּ

וַיֵּלְכוּ אֲחוֹרַנִית וְעֶרְוַת אֲבִיהֶם לֹא רָאוּ

וְאָנֹכִי הִבְחַנְתִּי מַעֲלָתוֹ וְקִצּוּרָם

כִּי לֹא כְצוּרֵנוּ צוּרָם

וְלֹא נוֹדַע אֲלֵיהֶם עֶרְכּוֹ אֲשֶר הוּנַף וַאֲשֶר הוּרָם

הֵבִיאוּ אַדִּירֵיהֶם בַּעֲבוֹדָתוֹ צַוָּארָם

וְהִנְנִי שׁוֹאֵל מִמְּךָ בְּנֶפֶשׁ שְׁפָלָה

אַחַר בַּקָּשַׁת הַמְּחִילָה

לְחַבֵּר עַל זֶה שִׁיר כַּאֲשֶר יֵאוֹת לוֹ

וְיִכְלֹל זֶה הָעִנְיָן כֻּלּוֹ

וָאָקוּם וָאֲחַבֵּר זֶה הַשִּיר אֲשֶר אֵין עַל עָפָר מָשְׁלוֹ

וַיִּשְׂמַח הָאִישׁ כִּי רַב חֵילוֹ


  


בְּחַיֵּי הַבְּרִית טוֹב אַתְּ בְּעֵינַי

וְכָל צַדִּיק בְּךָ יָגִיל וְיַכְתִּיר

וְנִגְלוּ לִי צְפוּנוֹת וַעֲמוּקוֹת

אֲשֶרתּוֹך לִבְּךָ תַעְמִיק וְתַסְתִּיר

אֲבָל אֶשְׁמַע הֲמוֹן עָם יֹאמְרוּ לִי

יְהוּדָה נִים וְלָא נִים תִּיר וְלָא תִיר


  


עוֹד זֶה מְדַבֵּר וְזֶה בָּא וַיֹּאמַר אֲדֹנִי

רָאִיתִי בְסִפְרֵי תַּלְמַי הַיְּוָנִי

בְּדַבְּרוֹ עַל טֶבַע הַמַּזָּלוֹת אֲשֶרבַּגַּלְגַּל הַשְּׁמִינִי

שֶחָלַק כָּל מַזָּל וּמַזָּל לִשְׁלֹשָׁה חֲלָקִים

וּמָצָא כָּל הַחֲלָקִים הָהֵם בִּטְבָעִים חֲלוּקִים

הַשְּׁלִישׁ הָרִאשׁוֹן יוֹרֶה טֶבַע אֶחָד

וְהַשְּׁלִישׁ הָאֶמְצָעִי יוֹרֶה טֶבַע אַחֵר מְיֻחָד

וְהַשְּׁלִישׁ הָאַחֲרוֹן בְּטֶבַע שְׁלִישִׁי הִתְיַחָד

וְחֶלְקֵי כָּל מַזָּל וּמַזָּל נִשְׁתַּנוּ מֵחֶלְקֵי חֲבֵרוֹ

וְקָשֶׁה עָלַי לִזְכֹּר מַה שֶּׁהִזְכִּיר עַל כָּל אֶחָד מֵהֶם תַּלְמַי בְּסִפְרוֹ

וְרָצִיתִי שֶׁתְּחַבֵּר שִׁיר כָּלִיל בַּהֲדָרוֹ

יַזְכִּיר בְּקִצּוּר כָּל אֲשֶר הִזְכִּיר עֲלֵיהֶם תַּלְמַי בְּחִבּוּרוֹ

כְּדֵי שֶׁבְּיָדְעִי הַשִּׁיר עַל פֶּה אֵדָעֵם

וּבְקוֹל כָּמֹהוּ אַרְעֵם

וָאָקוּם וָאֲחַבֵּר שִׁיר מִשְּׁמֹנָה טוּרִים

הָאַרְבָּעָה הָרִאשׁונִים בְּטֶבַע רָאשֵׁי הַמַּזָּלוֹת מְדַבְּרִים

וְהַטּוּר הַחֲמִשִּׁי טִבְעֵי אֶמְצָעִיוּתָם בּוֹ נִכְלָלוֹת

וְהַשְּׁלֹשָׁה הָאַחֲרוֹנִים מְדַבְּרִים בְּטֶבַע סוֹפֵי הַמַּזָּלות

וָאֶתְּנֵהוּ אֶל הָאִישׁ וְהוּא מְסַפֵּר מִמֶּנוּ תְהִלוֹת


  


וָאֶשָּׂא מְשָׁלִי וָאֹמַר:


  


טְלֵה חָלָב אֲשֶר הוֹלִיד כְּלָלוֹ

רְעָמִים עִם בְּרָקִים הֶחֱרִישָׁךְ

וְהוּא רוּחַ וְרַעַשׁ וַעֲרָפֶל

לְךָ יִרְטַב וְחֹם יִרְעַשׁ וְיֶחְשָׁךְ

אֲוִירוֹ חָם וְחַם גַּם אַחֲרֵיהֶם

שְנַיִם דּוֹן לְבֵינוֹנִים בְּלַחְשָׁךְ

אֲשֶׁר יִרְטַב וְהוּא נִשְׂרָף בְּמוֹ אֵשׁ

וְהוּא יִרְטַב וּבֵינוֹנִי יְבִישָׁךְ

וְכָל אֶמְצַע לְבֵינוֹנִי הֲכִי דוֹן

לְבַד דָּגִים אֲשֶר יִרְטַב לְבוּשָׁךְ

וְסוֹף יִשְׂרֹף אֲשֶר יַרְגִּישׁ כְּמַרְגִּישׁ

וְרוּחַ יִהְְיֶה רָטוֹב וְנִמְשָׁךְ

וּמֵימִיַּי כְמֵימִיַּי וּפֹעַל

לְךָ רֶשַׁע אֲשֶר יַמְטִיר בְּרֹאשָׁך

וְהוּא יַמְטִיר וְאַף רוּחַ יְיַלֵּד

אֲשֶר יִשְׂרֹף וְהוּא נִקְרָא אֱנוֹשָׁךְ


  


עוֹד זֶה מְדַבֵּר וְזֶה בָּא וַיֹּאמַר אֲדֹנִי רִכְבִּי וּפָרָשִׁי

יֵש לִי רֵעַ שַׂמְתִּיו נֵזֶר לְרֹאשִׁי

אֲהַבְתִּיו אַהֲבַת נַפְשִׁי

וְחָפַצְתִּי לִהְיוֹת בְּמַאֲסַר אַהֲבָתוֹ לֹא אֵצֵא חָפְשִׁי

וְעַל נוּדוֹ הֶחְדַּלְתִּי אֶת תִּירוֹשִׁי

וּבְפָנַי יַעֲנֶה כַּחֲשִׁי

וּמִן הַיּוֹם אֲשֶׁר אָרִיתִי יַעְרַת דִּבְשׁוֹ וּמָתְקוֹ

דָּמִיתִי הֱיוֹת כָּל רֵעַ חָסֵר בְּחֻקּוֹ

וְלוּ הָיוּ דוֹדָיו דְּבֵקִים בּוֹ כְמַדָּיו אֲנִי חֲלוּקוֹ

וְלוּ הָיוּ הֵם כַּחֲלוּקוֹ

אֲנִי נַפְשׁוֹ הַשּׁוֹכֶנֶת בְּחֵיקוֹ

וּמָה אֹסִיף עוֹד לְדַבֵּר מֵאֹדוֹתָיו וְהִזִּיקַנִי וְיָסַפְתִּי בְחִשְׁקוֹ

וְעַתָּה אֲדֹנִי תְּתָאֵר זֶה הַחֵשֶׁק בְּשִׁירָךְ

לְמַעַן אֲדֹנָי אֱלֹהֶיךָ וְלִקְדוֹשׁ יִשְׂרָאֵל כִּי פֵאֲרָךְ

כִּי יָדַעְתִּי אֵת אֲשֶׁר תְּבָרֵךְ מְבֹרָךְ

וָאַעַן וָאֹמַר הִשָּׁמֵר וְהַשְׁקֵט אַל תִּירָא וּלְבָבְךָ אַל יֵרַךְ

וָאָקוּם וָאֲחַבֵּר לוֹ זֶה הַשִּׁיר הַנֶעֱרָךְ

וַיִּקֹּד הָאִישׁ וַיִּשְׁתַּחוּ לְאַפָּיו וַיֹּאמַר יְהִי שֵם אֲדֹנָי מְבֹרָךְ


  


בְּחִשְׁקִי עַל צְבִי מִבְחַר עֳפָרִים

אֲדַמֶּה כָּל צְבִי חָסֵר בְּחֻקּוֹ

וְלוּ הָיוּ חֲשׁוּקָיו בּוֹ דְבֵקִים

כְּמוֹ מַדָּיו אֲנִי לוֹ כַּחֲלוּקוֹ

וְלוּ הָיוּ כְנֵתַח מִנְּתָחָיו

אֲנִי נַפְשׁוֹ אֲשֶׁר אֶשְׁכּוֹן בְּחֵיקוֹ

וּמַה אוֹסִיף לְדַבֵּר בַּחֲשׁוּקִי

וְהִזִּיק לִי וְיָסַפְתִּי בְּחִשְׁקוֹ


  


עוֹדֶנוּ מְדַבֵּר וְזֶה בָּא וַיֹּאמַר חָנֵּנִי אֱלֹהִים וְיֵש לִי רֵעַ שִׂמְחַת גִּילִי

וּמְנַחֵם מֵשִׁיב נַפְשִׁי

אֲשֶׁר הוּא הַצָּנִיף הַטָּהוֹר עַל רֹאשִׁי

וּבְכָל חֵן טוֹב הָאֵל חֲנָנוֹ

וּמִמַּעַל לְכוֹכְבֵי אֵל שָׂם קִנּוֹ

וּבְטֶרֶם נוֹצַר בַּבֶּטֶן הֱכִינוֹ

וְנָבִיא לִבְנֵי הַזְּמַן נְתָנוֹ

וְהוּא רִאשׁוֹן אֶל הַחֹשְׁקִים

וְנוֹטֶה אָהֳלֵי הַחֵשֶׁק וִירִיעוֹתָיו מֵקִים

וְהַזְּמָן וְהַחֵשֶׁק עֲבָדָיו

וְצֹאן יָדָיו

וְלֹא הִתְעוּהוּ כַּאֲשֶר הִתְעוּ כָּל מַלְכֵי גוֹיִּים בִּלְעָדָיו

וְעַתָּה אֲדֹנִי זֶה הָעִנְיָן בְּשִׁירְךָ תְפָאֲרֵהוּ

וּבִמְחוּגַת שִׂכְלְךָ תְתָאֲרֵהוּ

כִּי אֵת אֲשֶׁר תְּאַשֵׁר אַשְׁרֵהוּ

וָאָקוּם וָאֹצִיא כָל לְמַעֲשֵׂהוּ

וָאֲחַבֵּר זֶה הַשִּיר הַנִּפְלָא

אֲשֶׁר בְּכֶתֶם אוֹפִיר לֹא יְסֻלָּא

וַיֹּאמֶר הָאִישׁ בֶּאֱמֶת זֶה הַשִּיר עַל כָּל שִׁיר מֶלֶךְ

וַיִּקַח אֶת הַשִּׁיר וַיֵּלֶךְ


  


מְנַחֵם לַזְּמָן הָיָה נְבִיאוֹ

וְהַחֵשֶׁק קְרָאָהוּ בּקִיאוֹ

וְהִתְעָה לְאֱנוֹשׁ זוּלַת מְנַחֵם

כְּהַתְעוֹת אֶל אֱנוֹש שִׁכּוֹר בְּקִיאוֹ


  


עוֹדֶנוּ מְדַבֵּר וְהִנֵּה שְׁנֵי אַחִים בָּאִים בְּלֵב נֶעֱצָב

וּמוּל פָּנַי הֶחֱזִיקוּ מַצָּב

וַיֹּאמְרוּ יֵשׁ לָנוּ אָב זָקֵן

כְּעוֹף נוֹדֵד מִקֵּן

יְעַוֵת אֵת אֲשֶׁר נְתַקֵּן

אֲנַחְנוּ לְמִשְׁמַעְתּוֹ כְפִי הַיְכֹלֶת סָרִים

וּבַעֲבוֹדָתוֹ וּבְמוּסָרָיו מִתְפָּאֲרִים

וּתְהִלָּתוֹ וּמַעֲלָתוֹ כְּחוֹבָה עָלֵינוּ מְסַפְּרִים

וְהוּא יַחְשֹׁב אוֹתָנוּ לְזָרִים

וַיִּתֵּן אוֹתָנוּ כְּסוֹרְרִים וּמוֹרִים

אֲשֶר לֹא צִוִּינוּ וְלֹא דִבַּרְנוּ וְלֹא עָלָה עַל לִבֵּנוּ

וְעַל זֶה נֶעְכַּר אָבִינוּ וְגָדֵל עָצְבֵּנוּ

וְאָבִי רְאֵה גַּם רְאֵה

אֵיךְ יַעֲלֶה עַל לְבָבֵנוּ הַנִּכְאֶה

לִנְטוֹת מִדֶּרֶךְ מוּסָרוֹ וְלַעֲנוֹת עַזּוֹת

אֲשֶר עָשָׂה לָנוּ אֶת הַנֶּפֶשׁ הַזֹּאת

וְאֵיךְ תִּשְׁכַּח נַפְשֵׁנוּ כָּל גְּמוּלָיו

בְּזָכְרֵנוּ אָז מִקֶּדֶם מִפְעָלָיו

וְכָל מַחְסוֹרֵנוּ עָלָיו

וְעַתָּה חַבֶּר לָנוּ שִׁיר בִּדְבַר צַחוֹת

אוּלַי יִרְאֶה אֲדֹנָי בְּעָנְיֵנוּ וִינַהֲלֵנוּ עַל מֵי מְנוּחוֹת

וְנִמְצָא בְשִׁירְךָ שׂבַע שְׂמָחוֹת

וָאָקוּם וָאֲחַבֵּר שִׁיר כְּפִי כֹחִי

וָאֶשְׁפֹּךְ עָלָיו רוּחִי

וַיֶעֱרַב עֲלֵיהֶם שִׂיחִי


  


אֵיכָה נַמְרֶה פִּיךָ אוֹ מִן

דֶּרֶךְ מוֹסָרָךְ הָאָב נֵט

וּבְיָדָךְ כָּל מַחְסוֹרֵנוּ

מִן הַכְּתֹנֶת עַד הָאַבְנֵט


  


עוֹדָם מְדַבְּרִים אִתִּי וַאֲנִי נֹשֵׂא מְשָׁלָי

וְהִנֵּה שְלֹשָׁה אֲנָשִׁים נִצָּבִים עָלָי

וְהָאֲנָשִׁים אַנְשֵׂי שֵׁמוֹת

נְקוּבֵי רֵאשִׁית הַחָכְמוֹת

קָרְאוּ בִשְׁמוֹתָם עֲלֵי אֲדָמוֹת

וַיֹּאמְרוּ אֵלַי הַמְשׁוֹרֵר לְעוֹלָם יִחְיֶה

מֵאֶרֶץ רְחוֹקָה בָּאוּ עֲבָדֶיךָ לְשֵׁמַע יְהוּדָה גוּר אַרְיֵה

כִּי שָׁמַעְנוּ אֶת אֲשֶׁר הוֹבִישׁ מֵי יָם הַסִּכְלוּת

וְהֵאִיר אֹפֶל עֵינֵי בְנֵי הַגָּלוּת

בְּהֵיטִיבוֹ נֵרוֹת הַלְּבָבוֹת הָעֲדִינוֹת

וּבְמַלֹּאתוֹ מִשְׁבְּצוֹת הַהֲבָנוֹת

מִיַּהֲלוֹמֵי הַסּוֹדוֹת הָעֶלְיוֹנוֹת

וּבְתִתּוֹ סַפִּירֵי סְפָרָיו כֻּלָּם

לַעֲלוֹת בָּהֶם וּלְהַשִּיג כָּל נֶעְלָם

וּמִימֵי חָכְמָתוֹ לַמַּשִׂיגִים חוֹמָה מִימִינָם וּמִשְּׂמֹאלָם

וּבַהֲסִירוֹ הָעִבְרִים הָעִוְרִים הַנִּרְפִּים

וְהַפִּסְחִים הַפּוֹסְחִים עַל שְׁתֵּי הַסְּעִפִּים

אֲשֶׁר בִּמְלַאכְתָּם מִתְרַפִּים

וְהֵסִיר אֶת הַמַּסְוֶה מֵעַל פְּנֵי הַנְּפָשׁוֹת הַשּׁוֹבְבוֹת

וְהֶעֱלָה אוֹתָם לְמַעְלָה מִגַּלְגַּלֵּי עֲרָבוֹת

עַד אֲשֶר לְכָל חַכְמֵי הַזְּמָן וּבַעֲלֶי לְבָבוֹת

יִתְּנוּ בְנוֹת הַשִּׁיר הָאֲלָפִים וְלוֹ נָתְנוּ הָרְבָבוֹת

וּמָצָא בְיוֹם חֵילוֹ עַמּוֹ נְדָבוֹת

וּבִזֵּר עַמִּים יַחְפְּצוּ קְרָבוֹת

וַיִּתְגֹּדְדוּ כְּמִשְׁפָּטָם בָּרְמָחִים וּבַחֲרָבוֹת

וְעַתָּה אֵיךְ נִדְרָשָׁה כַיּוֹם אֶת פְּנֵי חָכָם עָדִיף מִנָּבִיא

כָּרַע שָׁכַב בֵּין מִשְׁפְּתֵי הַחַכְמוֹת כְּאַרְיֵה וָכַלָבִיא

וּבְיָדֵנוּ אֵין תְּשׁוּרָה לְהָבִיא

וְלָכֵן בָּאנוּ עָדֶיךָ בְּלֵב בּוֹטֵח

בְּנֶפֶשׁ כּוֹרְעָה וּבְרַעְיוֹן מִשְׁתַּטֵּחַ

לְחַבֵּר שִׁירֵי חֶמֶד הֲדוּרִים

לְקוּחִים מִפְּנִינֵי הַמַּאֲמָרִים

וְעָשִׁיתָ אוֹתָם לְרֹאשׁוֹ עֲטָרוֹת

וְנִשְׁלָחֵם אֶל כֶּתֶר הַמְּאוֹרוֹת

וְאַחֲרֵי כֵן נִרְאֶה פָנָיו הַהֲדוּרוֹת

וְזֹאת הָיְתָה סִבַּת בּוֹאֵנוּ

וְשׂוּמֵנוּ עָדֶיךָ מְגַמַּת פָּנֵינוּ

וָאֹמַר אֶל הַנִּכְבָּדִים יִתֵּן כָּל אֶחָד מִכֶּם הָעִנְיָן

אֲשֶר יַחְפְּצֶנוּ לִיסוֹד וְעָלָיו אֶבְנֶה הַבִּנְיָן

וַיִּגַּשׁ הָאֶחָד וַיֹּאמַר אֲדֹנִי

חָפַצְתִּי לְדַמּוֹת שִׂכְלוֹ

אֶל הַשֶּׁמֶשׁ בְּהִלּוֹ

כִּי כַאֲשֶר הָיָה הַשֶּׁמֶשׁ כָּלִיל בַּהֲדָרוֹ

וְהוּא סִבַּתהָרְאוּת וְנִרְאֶה אוֹר בְּאוֹרוֹ

וּבוֹ נִרְאֵהוּ

כָּךְ שִׂכְלוֹ אֲדַמֵּהוּ

הֱיוֹתוֹ סִבָּה לְכָל שֵׂכֶל וּבוֹ נַשְׂכִּיל וְנַשְׂכִּילֵהוּ

וְיָדַעְתִּי כִּי הִקְשֵיתִי לִשְׁאֹל

אָמְנָם יָדַעְתִּי כִּי אֵין זוּלָתְךָ לִגְאֹל

וָאָקוּם קַל חִישׁ וָאֲחַבֶּר לִי זֶה הַשִּׁיר אָרְכוּ מֻטּוֹת כְּנָפָיו

וְאִם קָצְרוּ מִלָּיו רָבּוּ עֲנָפָיו

וַיִּקָחֵהוּ הָאִישׁ וַיִּבֶז לְכָל זְהָבוֹ וּכְסָפָיו

וַיִּקֹד וַיִּשְׁתַּחוּ לְאַפָּיו


  


יִפְעַת תְּבוּנָתָךְ וְנֵר הִלָּךְ אֲדַ-

מֶּה הַשָּׁר לְאוֹר שֶׁמֶשׁ וְאַמְשִׁילֵהוּ

כִּי כַאֲשֶר שֶׁמֶשׁ בְּסִבַּת הָרְאוֹת

נִרְאֶה בְּאוֹרוֹ אוֹר וּבוֹ נִרְאֵהוּ

כֵּן שִׂכְלְךָ סִבָּה לְשֵׂכֶל כָּל יְצוּר

וּבְשִׂכְלְךָ נַשְׂכִּיל וְנַשְׂכִּילֵהוּ


  


וְאַחֲרָיו הַשֵּׁנִי נִגָּש

וַיֹּאמַר אֲדֹנִי רָצִיתִי לְתָאֲרוֹ

כִּי נִפְרַד מֵהַמֻּרְגָּשׁ

וְאֶל הָעֲרָפֶל אֲשֶר שָׁם האֱלֹהִים נִגַּש

וְחָשַׁב בְּשִׂכְלוֹ גַּאֲוַת הַזְּמָן נְסִיכוּת

וַיִּקְרָא שְׁמוֹ אַלוֹן בָּכוּת

וּפָשַׁט מְעִילֵי הַתְּמוּנוֹת וְלָבַשׁ מַלְאֲכוּת

עַד אֲשֶר עֵינֵי אַנְשֵׁי הַזְּמָן מֵהַשִּׂיג הוֹדוֹ עָשָׁשׁוּ

כִּי לְמַדַּי לֹא הִתְקַדָּשׁוּ

וּמֵעַבְדוּת הַמֶּלֶךְ זָקֵן וּכְסִיל לֹא חֻפָּשׁוּ

וַיֵּדְעוּ כִּי עֲרוּמִים הֵם וַיִּתְבֹּשָׁשׁוּ

וְנִגַּש יְהוּדָה לְבַדּוֹ אֶל אֲדֹנָי וְהֵם לֹא יִגָּשׁוּ

וָאָקוּם וָאֲחַבֵּר זֶה הַשִּׁיר וָאֶתְּנֶהוּ אֵלָיו

וְנִרְצָה לוֹ לְכַפֵּר עָלָיו


  


יִפָּלְאוּ לִבּוֹת בְּלָבְשָׁךְ מַלְאֲכוּת

עַד כִּי לְהַשִׂיג הוֹדְךָ נֹאָשׁוּ

תַּפְשִׁיט בְּשִׂכְלָךְ מַחֲמַדִּים אֶת עֲדִי

צוּרוֹת טְהוֹרוֹת יַעֲטוּ לָבָשׁוּ

רָאוּ הֲלִיכוֹת שִׂכְלְךָ וַיֶּחֱזוּ

כִּי הֵם עֲרוּמֵּי הוֹד וְהִתְבּוֹשָׁשׁוּ

בָּזוּ לְשִׂכְלָם לֹא עֲזָרָם לַעֲלוֹת

לֵאוֹר בְּאוֹר עֶלְיוֹן וּבוֹ נִבְאָשׁוּ

קֶרֶן מְאוֹר הִלָּךְ וְאֵיכָה יֶחֱזוּ

הוֹדָךְ בְּעֵינֵיהֶם וְהֵם עָשָׁשוּ

אֵיךְ יַעֲמֹד לִבָּם בְּגִשְׁתָּךְ לַעֲרָ-

פֶל שָׁם אֱלֹהִים חַי וְלֹא יִגָּשׁוּ


  


וַיִּגַּשׁ אֵלַי הַשְּׁלִישִׁי

וַיֹּאמַר רָצִיתִי רִכְבִּי וּפָרָשִׁי

לִרְמֹז אֶל יְהוּדָה גוּר אַרְיֵה

אֲשֶר אָמַרְנוּ בְּצִלּוֹ נִחְיֶה

לֶאֱסוֹף נְפוּצוֹת יְהוּדָה וְנִדְחֵי יִשְׂרָאֵל לְכַנֵּס

וְלַעֲשׂוֹת שׂרְפֵי סְפָרָיו וְלָשׂוּם אוֹתָם עַל נֵס

יִרְאוּ אוֹתָם נְשׁוּכֵי הַזְּמָן וְיִחְיוּ

וּמִנַּהֲרֵי נַחֲלֵי עֶדְנֵיהֶם יִרְוְיוּ

וְאַחֲרֵי מוֹת קְדוֹשִׁים יִהְיוּ

כִּי הוּא לְבַדּוֹ בַּמַּשָּׂא יָשׂוֹר

וְיַהֲפֹךְ עֵמֶק פְּגָרֵינוּ לְנַחַל הַבְּשׂוֹר

וָאָקוּם וָאֲחַבֶּר לוֹ זֹאת הַשִּׁירָה הָעֲרֵבָה

וַיִּשְׂמַח הָאִישׁ בִּרְאוֹתוֹ יָפְיָהּ וְטוּבָהּ

וַיֹּאמֶר אֵלַי הַיּוֹם נִרְצָע לְאַהֲבָתְךָ שַׂמְתָּנִי

אֶמְצָא חֵן בְּעֵינֶיךָ אֲדֹנִי כִּי נִחֲמְתָּנִי

וַיְבָרְכוּ אוֹתִי בְּשֵׁם אֵל עֵילוּם

וַיֵּלְכוּ מֵאִתִּי בְשָׁלוֹם


  


חוּשָׁה מְשִׁיחַ אֵל עֲשֵׂה שָׂרָף וְשִׂים

עַל נֵס לְהַחְיוֹת בּוֹ נְשׁוּכֵי הַזְּמָן

יָתֵד אֲשֶׁר נִתְקַע בְּמָקוֹם נֶאֱמָן

נֶשֶׁר מָלֵא נוֹצָה וּבַעַל מַעֲשִׂים

הֶן-הוּא כְּרוּב מִמְשַׁח כְּנָפָיו פּוֹרְשִׂים

נֵצֶר כְּלִיל יֹפִי וְנֶטַע נַעֲמָן

אַף זִכְרְךָ עָרֵב כְּנֹפֶת צוּף וּמַן

נֵרָךְ וְהֵלָּךְ כָּל חֲדָרִים חוֹפְשִׁים


  


עוּרָה גְאוֹן יַעֲקֹב וְלָמָּה תֶחֱרַד

עַל דּוֹדְךָ יִדֹּר וְהוֹלֵךְ וַחֲסוֹר

הַנֵּה מְשִׁיחַ אֵל בְּעָצְמָתוֹ יְרַד


  


לִבּוֹ כְגִבּוֹרִים וְרִכְבּוֹ יֶאֱסוֹר

יַהְפֹךְ וְאִם הַיּוֹם מְאֹד קָצָר וְרָד

עֵמֶק פְּגָרֵינוּ לְנַחַל הַבְּשׂוֹר


  


אֲנִי טֶרֶם אֲכַלֶה לְדַבֵּר

וְהַשִׁירִים הָעֲרֵבִים לְחַבֵּר

וְהִנֵּה אִישׁ מַכֶּה אִישׁ עִבְרִי מֵאֶחָיו וַיִּמְחָצֵהוּ

וַיֶּאֱחֹז בְּעָרְפּוֹ וַיְפַצְפְּצֵהוּ

וּפֶרֶץ עַל פְּנֵי פֶּרֶץ יִפְרְצֵהוּ

וְאֶל בֵּית הַבּוֹר יְרִיצֵהוּ

וּמַדָּיו יַפְשִׁיטֵהוּ

וְכַאֲשֶׁרשָׁאַלְנוּ קָדְקֹד שֵׂעָר

מִתְהַלֵךְ בַּאֲשָׁמָיו

עַל רָדְפוֹ בַחֶרֶב אָחִיו וְשִׁחֵת רַחֲמָיו

אָמַר הַסִּבָּה אֲשֶׁר עָלָה עָשָׁן בְּאַפִּי

וְאֵשׁ תֹּאכַל מִפִּי

וְאֶנְהֹם וְאוֹחֵז טַרְפִּי

הוּא כִּי אָנֹכִי הַדַּל בְּאַלְפִּי

הִלְוִיתִי לְזֶה מֹשֶׁה הָאִישׁ בִּימֵי חָרְפִּי

זְהָבִי וְכַסְפִּי

וְנָתַן לִי עֲרֵבִים

שְנַיִם מִבְּנֵי הַנְּדִיבִים

וְהוּא מִפָּנַי מִתְעַלֵּם

וּבוֹשׁ לֹא יֵבוֹשׁ כִּי לֹא יֵדַע הִכָּלֵם

וּבִרְאוֹתִי מִרְיוֹ וְקָשְׁיוֹ

וְכִי לֹא יְשַׁלֵּם נִשְׁיוֹ

הוֹרַדְתִּי מֵעָלָיו עֶדְיוֹ

אֲחַזְתִּיו וְלֹא אַרְפֶּנוּ

וְלַעֲפַר מָוֶת אֶשְפְּתֶנוּ

וַיְהִי כִּרְאוֹת הַשָׂר אֲדֹנִי

נִכְמְרוּ רַחֲמָיו עַל מֹשֶׁה הֶעָנִי

וַיֹּאמֶר אֶל הָאִיש הָאוֹחֵז בּוֹ לֹא נָכוֹן לַעֲשׂוֹת כֵּן בְּנִי

כִּי הַכַּעַס לֹא יֵחָשֵׁב לְךָ לְחָכְמָה

וְלָכֵן הֶרֶף מֵאַף וַעֲזֹב חֵמָה

וְאַל תִּתְחַר לְהָרַע בַּמְּרֵעִים

כִּי דִבְרֵי חֲכָמִים בְּנַחַת נִשְׁמָעִים

וְעַתָּה חֲקֹר וְּדְרֹשׁ וְאִם יוּכַל לְשַׁלֵּם נִשְׁיוֹ מֹשֶׁה

אָז תִּהְיֶה לוֹ כְּנוֹשֶׁה

וְאִם יְצָאוּהוּ הַכְּסָפִים וְהַזְּהוּבִים

רִיבְךָ רִיב אֶת הַעֲרֵבִים

וַיִּפֶן אֵלַי הַשָּׂר וַיֹּאמַר אַחֲרֵי אֲשֶר עָשִׁיתָ בִּמְלֶאכֶת הַשִּיר נִפְלָאוֹת

וְרָאִינוּ מִמְּךָ נוֹרָאוֹת

הָרִימָה פְּעָמֶיךָ לְמַשֻּׂאוֹת

וְתָקִים אֶת דְּבָרַי בְּשִׁיר עָרֵב

וְנִפָּרֵד מִזֶּה הַמַּעֲמָד כִּי הָרָעָב הוֹלֵךְ וְקָרֵב

וְיַּכֶּה בִצְדִיָּה לְמוֹ אָרֶב

כִּי פָנָה יוֹם וְכִי יִנָּטוּ צִלְלֵי עָרֶב

וְאִישׁ לֹא יוֹסִיף עוֹד לִשְׁאוֹל

פֶּן אֵרֵד אֶל רַעֲבוֹנִי אָבֵל שְׁאוֹל

וְעַתָּה גְאַל לְךָ אַתָּה אֶת גְּאֻלָּתִי כִּי לֹא אוּכַל לִגְאֹל

וָאָקוּם בְּמִצְוַת הַשָּׂר אֲדֹנִי וָאֲחַבֵּר שִׁיר חָדָשׁ אֵין לוֹ שֵׁנִי

לְהַצִּיל הֶעָשׁוּק מִמּוֹקְשׁוֹ

וּלְהוֹשִׁיעַ מִשֹּׁפְטֵי נַפְשׁוֹ


  


וָאֶשָּׂא מְשָׁלִי וָאֹמַר:


  


עֲזֹב אַף וְחֵמָה נְדִיב לֵב וְקוּמָה

וְדַבֵּר בְּחָכְמָה בְּמִלִּים עֲרֵבִים

וְאִם יֵש לְמֹשֶׁה תְּהִי לוֹ כְּנוֹשֶׁה

וְאִם אֵין בְּיָדוֹ מְעָה רִיב עֲרֵבִים


  


וַיְהִי כִּשְׁמֹעַ הָאֲנָשִׁים הַשִּׁיר הַנָּעִים

מַעֲשֶׂה צַעֲצוּעִים

צָעֲקוּ בְּקוֹל גָּדוֹל כְּאִלוּ אֵל הַכָּבוֹד הִרְעִים

וַיֹּאמְרוּ מַה מְאֹד נִכְבָּד זֶה הַמַּעֲמָד וּמָה רַב טוּבוֹ

כִּי רוּחַ אֲדֹנָי נוֹסְסָה בּוֹ

וְהָיוּ כֻּלָּם מְסַפְּרִים בִּשְׁבָחָי

וַיֹּאמְרוּ כֹּה לֶחָי

הַיּוֹם הַזֶּה רָאִינוּ כִּי יְדַבֵּר אֱלֹהִים אֶת הָאַָדָם וָחָי

וָאֵלֵךְ עִם הַשָׂר לִשְׁכּן תַּחַת צֶאֱלָיו

וְכָל הָעָם שָׁבוּ אַישׁ לְאֹהָלָיו


  






“הִלְבִּישֵׁנִי”– לא ברור בטקסט אם השי"ן מנוקדת בצירה או בפתח (הערת פב"י).  ↩
