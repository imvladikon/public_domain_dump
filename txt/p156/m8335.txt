


	
	מחברות עמנואל: מחברת חמישית
	



מחברת חמישית


אָמַר עִמָּנוּאֵל הַמְּחַבֵּר:

הָיִיתִי עִם הַשָּר בְּיוֹם פּוּרִים

עִם עֲדַת נִכְבָּדִים וְשָׂרִים

וְעִמָּהֶם הָיוּ חֶבְרַת מְשׁוֹרְרִים

בָּאוּ מֵאֶרֶץ מֶרְחָק

וְלִבָּם לְכָל מִבְצָר יִשְׂחָק

וַנְּהִי בְעֵינֵיהֶם כַּחֲגָבִים

חָשְׁבוּ שִׁירֵינוּ שִׁירֵי עֲגָבִים

וַיֹּאמֶר אֵלַי הַשָּר נַעֲשֶה לָנוּ שֵׁם כַּיּוֹם הַזֶּה לְמַעַן יֵדְעוּ כָל הַמְּשׁוֹרְרִים כִּי נִיבָם נִבְזֶה וְרָזֶה

וְשִׂכְלָם שׁוֹכֵב וְהוֹזֶה

וְאֵין עֲרֹך אֵלֵינוּ בְּכָל הָאָרֶץ

וּפָרַצְנוּ בִּגְבוּל הַמְּלִיצוֹת פָּרֶץ

וְזֶה הַדָּבָר אֲשֶׁר נַעֲשֶׂה נְדַבֵּר בְּעִנְיָנִים שׁוֹנִים

אֲנִי אֲדַבֵּר בַּתְּחִלָּה וּדְבָרֶיךָ יִהְיוּ לִדְבָרַי עוֹניִם

וְכָל אֶחָד מִמֶּנוּ חִידָתוֹ יִזְכֹּר

פַּעַם נְדַבֵּר בְּעִנְיָן הֶעָצֵל וּפַעַם בְּעִנְיַן הַשִּׁכּוֹר

וּפַעַם נְדַבֵּר בְּעִנְיַן בַּעַל הַקְּרָנַיִם

וּפַעַם בְּעִנְיַן הַקְּדֵשָׁה הִיא בָעֵינַיִם

וּפַעַם נְדַבֵּר בְּעִנְיַן הַכִּילַי וּפַעַם בְּעִנְיַן הָאֹרֶךְ

וּפַעַם בְּעִנְיַן הַיִּרְאָה וּפַעַם בְּעִנְיַן הַצֹּרֶךְ

וּפַעַם בְּעִנְיַן הָאַהֲבָה וְהַשּׂנְאָה

וּפַעַם בְּעִנְיַן הַקִּנְאָה

לְמַעַן יֵדְעוּ בְּנֵי אַסְנָה וּבְנֵי סְנָאָה

וְהָעָם וּמְשׁוֹרְרֵיהֶם

כִּי עֵירֻמִּים הֵם

וְלָנוּ הַחִידוֹת הַנַּעֲלָמוֹת

וְהַמְּלִיצוֹת הָרָמוֹת

קָרְאוּ בִשְׁמוֹתָם עֲלֵי אֲדָמוֹת

וָאַעַן וָאֹמַר אָמֵן הִנְנִי אֲשַׁנֵּס מָתְנַי

וְלָצֵאת אֶל הַמַּעֲרָכָה אֲעוֹרֵר רַעְיוֹנַי

וְאֶעֱנֹד עַל גַּרְגְּרוֹת אֲמָרֶיךָ פְּנִינַי

לְרֶגֶל הַמְּלָאכָה אֲשֶׁר לְפָנַי

וַיֹּאמֶר הַשַּר לְעֵינֵי הַזְּקֵנִים וְהַחוֹרִים[

שִׁמְעוּ נָא הַמֹּרִים

יֶשׁ לִי בֵּן עָצֵל

כְּעֶבֶד יִשְׁאַף צֵל

וּבְכָל אֲשֶׁר יַחֱטָא יִתְנַצֵּל

וְחָפַצְתִּי לְגַנּוֹת פְּעֻלּוֹתָיו

הֲיֵש פֹּה אִישׁ יִתְנַדֵּב לַעֲנוֹת עַל מַה שֶׁאַזְכִּיר מֵאֹדוֹתָיו

וָאַעַן וָאֹמַר הִנְנִי אֲנִי לַעֲנוֹת אֶל מַאֲמָרֶיךָ נִכְסָף

דַּבֵּר וְלוּ אַתָּה אָסָף

וַיֹּאמֶר הַשָּׁר: בְּנִי יִקְרָאוּהוּ בְּקוֹל הֲמֻלָּה כְּקוֹל מַחֲנֶה

וָאַעַן וָאֹמַר: מִקּוֹלָם לֹא יֵחָת וּמֵהֲמוֹנָם לֹא יַעֲנֶה.

וַיֹּאמֶר הַשָּׁר: בְּנִי בְּכָל אֲשֶׁר יֶחֱטָא יִתְנַצֵּל

וָאַעַן וָאֹמַר: וְאִם כָּל עֵצוֹ וּפְרִי אַדְמָתוֹ יְיָרֵש הַצְּלָצַל.

וַיֹּאמֶר הַשָּׁר: אִם הַשֵּׁנָה הִיא לַמִּיתָה אֶחָד מִשּׁלֹשִׁים

וָאַעַן וָאֹמַר: הַמִּיתָה לִשְׁנַת בִּנְךָ אֶחָד מִשִּׁשִׁים.

וַיֹּאמֶר הַשָּׁר: בְּנִי יִפֹּל אֶל הַפַּחַת וּמֵעֲלוֹת מִמֶּנוּ יִתְיָאֵש

וָאַעַן וָאֹמַר: לְדִמְיוֹן הַחֲמוֹר שֶׁיִּשָּׂרֵף וְלֹא יִבְרַח מִן הָאֵשׁ.

וַיֹּאמֶר הַשָּׁר: בְּנִי יוֹתֵר עָצֵל מִגַּלְגַּל שַׁבְּתַי

וָאַעַן וָאֹמַר: וְעַל יְמֵי הַחֹל יֹאמַר אֵלֶּה הֵם שַׁבְּתוֹתַי.

וַיֹּאמֶר הַשָּׁר: אֹמַר לִשְׁנַת בְּנִי עַל כָּל דָּבָר עָלִית

וָאַעַן וָאֹמַר: כִּי הִיא מֵהַדְּבָרִים שֶׁאֵין לָהֶם תַּכְלִית.

וַיֹּאמֶר הַשָּׁר: אֵלּוּ הַדְּבָרִים שֶׁאֵין לָהֶם שִׁעוּר הַפֵּאָה וְהַבִּכּוּרִים וְהָרַאֲיוֹן

וָאַעַן וָאֹמַר: וּשְׁנַת בִּנְךָ עַל כָּל דָּבָר עֶלְיוֹן.

וַיֹּאמֶר הַשָּׁר: שְׁנַת בְּנִי מִיתָה מִקְרִית

וָאַעַן וָאֹמַר: כִּי יֶשׁ-לָהּ רֵאשִׁית וְאֵין לָהּ אַחֲרִית.

וַיֹּאמֶר הַשָּׁר: אִלּו הָיָה קֵץ הַמָּשִׁיחַ תָּלוּי בְּקֵץ שְׁנַת בְּנִי וְתַרְדֵּמָתוֹ

וָאַעַן וָאֹמַר: הִתְיָאֲשׁוּ הַיְּהוּדִים כְּבָר מִבִּיאָתוֹ.

וַיֹּאמֶר הַשָּׁר: בְּנִי יֹאכַל וְשָׂבֵעַ וְדָשֵׁן

וָאַעַן וָאֹמַר: וְהוּא עֵר וְלִבּוֹ יָשֵׁן.

וַיֹּאמֶר הַשָּׁר: בְּנִי יָשֵׁן עַד שָׁעָה רְבִיעִית בְּתוֹךְ הַתַּכְרִיכִים

וָאַעַן וָאֹמַר: וְיֹאמַר כָּל יִשְׂרָאֵל בְּנֵי מְלָכִים.

וַיֹּאמֶר הַשָּׁר: בְּנִי יִרְאֶה הַסְּחוֹרוֹת וְיָפְיָם

וָאַעַן וָאֹמַר: וְיֹאמַר מִי יַעֲבָר לָנוּ אֶל עֵבֶר הַיָּם.

וַיֹּאמֶר הַשָּׁר: בְּנִי יֶחְסַר בֶּגֶד לִלְבֹּש וְלֶחֶם לֶאֱכֹל

וָאַעַן וָאֹמַר: וְיֹאמַר חַנַּנִי אֱלֹהִים וְיֶשׁ-לִי-כֹל.

וַיֹּאמֶר הַשָּׁר: בְּנִי שִׂמְחָתוֹ בְחַגִּים וּבְמוֹעֲדִים

וָאַעַן וָאֹמַר: לִהְיוֹת הַשִּׂמְחָה וְהַמְּנוּחָה רֹכְבִים צְמָדִים.

וַיֹּאמֶר הַשָּׁר: בְּנִי יִרְאֶה אִשְׁתוֹ נְתוּנָה בְיַד חשֵׁק

וָאַעַן וָאֹמַר: וְלֹא יַצִּיל עָשׁוּק מִיַּד עשֵׁק.

וַיֹּאמֶר הַשָּׁר: שְׁנַת בְּנִי אֲרֻכָּה מִיּוֹם הַזִּכָּרוֹן

וָאַעַן וָאֹמַר: וְיַחֲרִישׁ לְקוֹרְאָיו וְאִם יִקְרְאוּ בְגָרוֹן.

וַיֹּאמֶר הַשָּׁר: יֹאמַר בְּנִי יָשַׁנְתִּי אָז יָנוּחַ לִי

וָאַעַן וָאֹמַר: וְזֶה הָיָה חֶלְקִי מִכָּל עֲמָלִי.

וַיֹּאמֶר הַשָּׁר: יֹאמַר בְּנִי אִם אֵין בְּעֵדֶן גַּן אֱלֹהִים שֵׁנָה

וָאַעַן וָאֹמַר: לֹא תִהְיֶה נַפְשִׁי בוֹ דְשֵׁנָה.

וַיֹּאמֶר הַשָּׁר: בְּנִי יִרְאֶה הַסְּחוֹרָה וּשְׁבָחֶיהָ

וָאַעַן וָאֹמַר: יֹאמַר מִי יַעֲלֶה לָנוּ הַשָּׁמַיְמָה וְיִקָּחֶהָ.

וַיֹּאמֶר הַשָּׁר: בְּנִי יִרְאֶה שִׂפְתֵי חֲשׁוּקָתוֹ יַאְדִימוּ כַתוֹלָע

וָאַעַן וָאֹמַר: וְלֹא יָבוֹא לִרְאוֹת אֶת-הַקֹּדֶשׁ כְּבַלַּע.

וַיֹּאמֶר הַשָּׁר: כְּבָר נִסּוּ חַכְמֵי הָאֱמֶת

וָאַעַן וָאֹמַר: כִּי לוּ יַנִּיחוּ הֶעָצֵל בִּשְׁנָתוֹ יִמְצָאוּהוּ מֵת.

וַיְהִי כַּהֲתִימֵנוּ לְתָאֵר עִנְיַן הָעַצְלוּת

וְזָכַרְנוּ מִמֶּנוּ הוּלְלוּת וְסִכְלוּת

וְשָמְעוּ הַשָּׂרִים וְהַמְּשׁוֹרְרִים דִּבְרֵי הַשָּׂר וַאֲשֶׁר אֶעֱנֵהוּ

וַיִּתְמְהוּ הָאֲנָשִׁים אִישׁ אֶל רֵעֵהוּ

וַיֹּאמֶר הַשָּׂר אֵלָי הִנֵּה נָא הוֹאַלְתִּי לְדַבֵּר

עֲזַרְתַּנִי אַתָּה הֶחָבֵר

לְגַנּוֹת הָעַצְלוּת

וְיֵשׁ-לִי אֶת בַּעַל הוֹלֵלוּת

וְהוּא שֹׁתֶה שִׁכּוֹר

הֲתַעַזְרֵנִי עוֹד גּנוּתוֹ לִזְכּוֹר

וָאַעַן וָאֹמַר חַיֶיךָ הַשּׁר לֹא יָדַעְתִּי אֲכַנֶה

תִּקְרָא וְאָנֹכִי אֶעֱנֶה

וּמַטָּרָא לְחִצֵּי דְּבָרֶיךָ תַצִּיבֵנִי

אוֹ אֲדַבֵּר וַהֲשִׁיבֵנִי

וַיֹּאמֶר הַשּׂר דַבֵּר אַתָּה וַאֲנִי אַקְשִׁיב

וּלְמַאֲמָרֶיךָ אֶהְיֶה מֵשִׁיב

וָאֹמְרָה: אָחִיךָ הַשּׂר יִשְׁתֶּה לִצְמָאוֹ גַלֵּי יָם

וַיֹּאמֶר הַשָּר: וְהֵנִיף יָדוֹ עַל הַנַּהַר בַּעְיָם.

וָאֹמְרָה: חַי אֲנִי אִם אֲדַבֵּר עַל אָחִיךָ כְּזָבִים

וַיֹּאמֶר הַשָּר: אִם לֹא רְאִיתִיו מִתְפַּלֵּל בְּעֵת הַבָּרָד אַל תִּשְׁכַּח עֲנָבִים.

וָאֹמְרָה: אֵין שִׂמְחָה כִּרְאוֹת אָחִיךָ פִּיו מְנַבֵּל

וַיֹּאמֶר הַשָּר: הוּא כְּשֹׁכֵב בְּלֶב-יָם וּכְשֹׁכֵב בּרֹאשׁ חִבֵּל.

וָאֹמְרָה: אֵין טוֹב בַּזְּמָן כִּרְאוֹת אָחִיךָ שִׁכּוֹר

וַיֹּאמֶר הַשָּר: יִשְׁתֶּה וְיִשְׁכַּח רִישׁוֹ וַעֲמָלוֹ לֹא יִזְכּוֹר.

וָאֹמְרָה: אָחִיךָ כַּאֲשֶׁר יִיקַץ מִיֵּינוֹ

וַיֹּאמֶר הַשָּר: יֹאמַר כִּי נָפְלָה דְּלֵקָה בִגְרוֹנוֹ.

וָאֹמְרָה: אָחִיךָ כָּל עֲמָלוֹ וּבֵית אָבִיו יַשֶּׁה

וַיֹּאמֶר הַשָּר: וְלוּ הָיָה יַם-סוּף יָיִן לֹא הֻצְרַכְנוּ לְמַטֵּה משֶׁה.

וָאֹמְרָה: יֹאמַר אָחִיךָ יָשָׁר בָּאָדָם אָיִן

וַיֹּאמֶר הַשָּר: כִּי אִם הַשּׁתִים בּמִזְרְקֵי יָיִן.

וָאֹמְרָה: הֶחָכָם יִצְמָא וְהוּא יוֹצֵא הַמָּיְמָה

וַיֹּאמֶר הַשָּר: וְאָחִי שִׁכּוֹר מֻצָּב אַרְצָה וְרֹאשׁוֹ מַגִּיעַ הַשָּׁמָיְמָה.

וָאֹמְרָה: יֹאמַר אָחִיךָ הַיַּיִן תַּעֲנוּג הָעוֹלָם וְהָפְכּוֹ

וַיֹּאמֶר הַשָּר: וְלָאָרֶץ לֹא יְכֻפָּר לַיַּיִן אֲשֶר שֻׁפַּךְ בָּהּ כִּי אִם בְּדַם שֹׁפְכוֹ.

וָאֹמְרָה: יֹאמַר אָחִיךָ מִי יִתֵּן בְּחַיַּי אֶשְׁתֶּה וְאֶשְׂבַּע

וַיֹּאמֶר הַשָּר: וּבְמוֹתִי בְּתוֹך הַיַּיִן אֶטְבַּע.

כִשְׁמֹעַ הַשָּׂרִים חִידוֹתֵינוּ תָמָהוּ

וְהַמְּשֹׁרְרִים פָּחֲדוּ וְרָהוּ

וּלְשׁוֹנָם לְחִכָּם דָּבֵקָה הַעֲוֵה נִלְאוּ

כִּי אֲשֶׁר לֹא סֻפַּר לָהֶם רָאוּ

וַיֹּאמֶר הַשָּׂר חֵי אֲנִי כִּי נְדַבֵּר מִן הַסָּרִיס

אֲשֶׁר הוּא מַקְרִין מַפְרִיס

וּמֵאִשְׁתוֹ אֲשֶׁר תִּטְחַן לְאַחֵר

וַאֲנִי אֶהְיֶה בִמְלִיצוֹתַי פְּנֵיהֶם מְשַחֵר

וָאֹמְרָה אֶל הַשָּׂר הָחֵל

כִּי הִנְנִי לַעֲנוֹת אֶל מַאֲמָרֶיךָ מְיַחֵל

וַיֹּאמֶר הַשָּׂר: פְּלוֹנִית יִשְׁאֲלוּ בַעֲדָהּ אַיֵּה הַקְּדֵשָׁה הִיא בָעֵינָיִם

וַיֹּאמֶר הַשָּׂר: נְדִיבוּת פְּלוֹנִית עַל כָּל נְדִיבוֹת עוֹדְפָה

וָאַעַן וָאֹמַר: כִּי תִּתְנַדֵּב לַכֹּל בְּנַפְשָׁה וּבְגוּפָה.

וַיֹּאמֶר הַשָּׂר: פְּלוֹנִית אָסְפָה כָּל מִדָּה מְגֻנָּה

וָאַעַן וָאֹמַר: ואַחֲרֶיהָ לֹא זוּנָה.

וַיֹּאמֶר הַשָּׂר: אֵיךְ פְּלוֹנִית מֵחֵשֶׁק תִּתְיָאֵשׁ

וָאַעַן וָאֹמַר: וְתַחְתֶּיהָ נֶהְפַּךְ כְּמוֹ אֵשׁ.

וַיֹּאמֶר הַשָּׂר: פְּלוֹנִי לוּ הָיָה מְנַגֵּחַ בְּקַרְנָיו

וָאַעַן וָאֹמַר: כָּל חַיּוֹת לֹא יַעַמְדוּ לְפָנָיו.

וַיֹּאמֶר הַשָּׂר: פְּלוֹנִי יְדַעְנוּהוּ מִתְמוֹל שִׁלְשֹׁם שׁוֹר נַגָּח

וָאַעַן וָאֹמַר: וְקַרְנֵי רְאֵם קַרְנָיו בָּהֶם עַמִּים יְנַגַּח.

וַיֹּאמֶר הַשָּׂר:פְּלוֹנִי לוּ רָאָה קַרְנָיו הִתְאַבָּל

וָאַעַן וָאֹמַר: כִּי לֹא יִשּׂאוּם אַרְבָּעִים אֶלֶף נוֹשֵׂא סַבָּל.

וַיֹּאמֶר הַשָּׂר: פְּלוֹנִי בֵּן חֲמִשּׁים שָׁנָה וְלֹא יַעֲבֹד

וָאַעַן וָאֹמַר: וְאִשְׁתּוֹ דָרְשָׁה קַרְנוֹ תָּרוֹם בְּכָבוֹד.

וַיֹּאמֶר הַשָּׂר: פְּלוֹנִי יוֹשֵׁב בְּאֹהֶל הַלִּמּוּד וּבְמִשְׁכְּנוֹתָיו

וָאַעַן וָאֹמַר: וְאִשְׁתּוֹ לוֹמֶדֶת וְעָשִׂיתָ קַרְנֹתָיו עַל אַרְבַּע פִּנּוֹתָיו.

וַיֹּאמֶר הַשָּׂר: פְּלֹנִית מְדֻשּׁתִי וּבֶן גָּרְנִי

וָאַעַן וָאֹמַר: וּבַעְלָהּ גָּנַב מִשִּׁירַת חַנָּה רָמָה קַרְנִי.

וַיֹּאמֶר הַשָּׂר: כַּמָּה הָיָה פְּלוֹנִי שָׁמֵן וְדָשֵׁן

וָאַעַן וָאֹמַר: אִילוּ הָיוּ קַרְנוֹתָיו קַרְנוֹת שֵׁן.

וַיֹּאמֶר הַשָּׂר: לִפְלוֹנִי שָׁכֵן יִשְׁכֹּן בְּסֵתֶר אָהֳלוֹ

וָאַעַן וָאֹמַר: וְקַרְנַיִם מִיָּדוֹ לוֹ.

וַיֹּאמֶר הַשָּׂר: קַרְנֵי פְלוֹנִי לוּ רָאָה צִדְקִיָּה בֶן כְּנַעֲנָה אוֹתָם

וָאַעַן וָאֹמַר: אָמַר לְאַחְאָב בְּאֵלֶּה תְּנַגַּח אֶת אֲרָם עַד-כַּלּוֹתָם.

וַיֹּאמֶר הַשָּׂר: פְּלוֹנִי הָיָה אָדָם בְּצוּרַת פָּנָיו

וָאַעַן וָאֹמַר: לוּלֵי אֲשֶׁר קֶרֶן חָזוּת בֵּין עֵינָיו.

וַיֹּאמֶר הַשָּׂר: בְּעַד קַרְנֵי פְּלוֹנִי יֹאמְרוּ הָעֵדָה

וָאַעַן וָאֹמַר: אֵלֶּה הַקְּרָנוֹת אֲשֶׁר זֵרוּ אֶת-יְהוּדָה.

וַיֹּאמֶר הַשָּׂר: פְּלוֹנִי אִלּוּ בָא בַיַעַר וְעָבַר בֵּין אִילָנָיו

וָאַעַן וָאֹמַר: נֶאֱחַז בַּסְבַך בְּקַרְנָיו.

וַיֹּאמֶר הַשָּׂר: פְּלוֹנִית בְּמַעֲשֶׂיהָ בַּעְלָהּ תַּצְדִיק

וָאַעַן וָאֹמַר: לְקַיֵּם תְּרוֹמַמְנָה קַרְנוֹת צַדִּיק.

וַיֹּאמֶר הַשָּׂר: דֶּרֶךְ הַשְּׁוָרִים לְנַגַּח הָעַמִּים

וָאַעַן וָאֹמַר: וּפְלוֹני יְנַגַּח לְבָנוֹן וְשִׂרְיוֹן כְּמוֹ בֶּן-רְאֵמִים.

וַיֹּאמֶר הַשָּׂר: פְּלוֹנִי לוּ רָאוּ קַרְנָיו הָעַתּוּדִים בֵּרְכוּהוּ

וָאַעַן וָאֹמַר: וְלוּ פָגְעוּ בּוֹ הָאֵילִים הִמְלִיכוּהוּ.

וַיְהִי כַאֲשֶׁר הִשְׁלַמְנוּ לְגַנּוֹת בַּעַל הַקְּרָנָים

וְאִשְׁתּוֹ גֹּמֶר בַּת דִּבְלָיִם

בִּמְלִיצוֹת נֶחְמָדוֹת לְהַשְׂכִּיל וְתַאֲוָה הֵם לָעֵינָים

נִמְלָא פִּי הַשָּׂרִים וְהָעָם שְׂחוֹק

וּפָעֲרוּ פִּיהֶם לְנֶגֶד הַמְּשֹׁרְרִים לִבְלִי חֹק

וַיִרְאוּ הַמְּשֹׁרְרִים וַיָנֻעוּ וַיַּעַמְדוּ מֵרָחוֹק

וַיֹּאמֶר אֵלַי הַשַּׂר עוֹד הֲלֹא נְדַבֵּר מֵאֹדוֹת הַכִּילַי

וָאַעַן וָאֹמַר מִי לַייָ אֵלָי.

וָאֹמְרָה: הַכִּילַי כֹּל אֲשֶׁר בְּתוֹכוֹ יִטְמָא

ויען השר: כִּי הוּא יָרֵא לְהָסֵךְ אֶת-רַגְלָיו פֶּן יִצְמָא.

ואמרה: כַּמָּה הַכִּילַי נִתְעָב

ויען השר: כִּי הוּא יְפַחֵד לִבְדֹּק נְקָבָיו פֶּן יִרְעָב.

וַיְהִי כְּהַשְׁלִימֵנוּ דְּבַר הַכִּילַי

וַיֹּאמֶר הַשָּׂר אֵלָי

יֶשׁ-לִי שָׁכֵן אָרֹךְ שׁוֹטֶה וּכְמַדְקְרוֹת חֶרֶב בּוֹטֶה

וְהִנְנִי לְגַנּוֹתוֹ מְלִיצוֹתַי אֵלָיו נוֹטֶה

עַד כַּמְּעִיל בֹּשֶׁת יַעֲטֶה

וָאַעַן וָאֹמַר אָנָּא הַשַּׂר הָחֵל כִּי הִנְנִי מַכֶּה בַּמַּטֶּה

וּמֵהַשְׁלִים דְּבָרֶיךָ יָמִין וּשְׂמֹאל לֹא אֶטֶּה.

וַיֹּאמֶר הַשּׂר: אֹרֶךְ פְּלוֹנִי אֹרֶךְ עוֹג מֶלֶךְ הַבָּשָׁן

וָאַעַן וָאֹמַר: אֹרֶךְ הַחָכְמוֹת וּפֵרוּשָׁן.

וַיֹּאמֶר הַשּׂר: אֹרֶךְ פְּלוֹנִי אֹרֶךְ וָו וַיְזָתָא

וָאַעַן וָאֹמַר: אֹרֶךְ עֵץ הָמָן בֶּן הַמְּדָתָא.

וַיֹּאמֶר הַשּׂר: אֹרֶךְ פְּלוֹנִי אֹרֶךְ קֵץ הַיְּהוּדִים

וָאַעַן וָאֹמַר: אֹרֶךְ יְמֵי הַחֲרֵדִים.

וַיֹּאמֶר הַשּׂר: אֹרֶךְ פְּלוֹנִי אֹרֶךְ סִפּוּרֵי הַנָּשִׁים

וָאַעַן וָאֹמַר: אֹרֶךְ עֲצֵי הַבְּרוֹשִׁים.

וַיֹּאמֶר הַשּׂר: אֹרֶךְ פְּלוֹנִי אֹרֶךְ יְמֵי הָאֲסוּרִים

וָאַעַן וָאֹמַר: אֹרֶךְ יוֹם הַכִּפּוּרִים.

וַיֹּאמֶר הַשּׂר: אֹרֶךְ פְּלוֹנִי אֹרֶךְ קוֹמַת הָעֲנָקִים

וָאַעַן וָאֹמַר: אֹרֶךְ אַנְחַת הַחֹשְׁקִים

וַיֹּאמֶר הַשּׂר: אֹרֶךְ פְּלוֹנִי אֹרֶךְ יְמֵי הַפְּרִידָה

וָאַעַן וָאֹמַר: אֹרֶךְ כַּנְפֵי הַחֲסִידָה.

וַיֹּאמֶר הַשּׂר: אֹרֶךְ פְּלוֹנִי אֹרֶךְ חִבּוּרֵי שְׁמוּאֵל אִבְּן תִּבּוֹן

וָאַעַן וָאֹמַר: אֹרֶךְ יְמֵי הָרְעָבוֹן.

וַיֹּאמֶר הַשּׂר: אֹרֶךְ פְּלוֹנִי אֹרֶךְ סִפְרֵי רַבֵּנוּ סַעַדְיָה

וָאַעַן וָאֹמַר: אֹרֶךְ קְלָלוֹת שֶׁבְּיִרְמְיָה.

וַיֹּאמֶר הַשּׂר: אֹרֶךְ פְּלוֹנִי אֹרֶךְ חַיֵּי הָעוֹלָם הַבָּא

וָאַעַן וָאֹמַר: אֹרֶךְ יְדִיעַת לְכָל הַנִּמְצָאוֹת סִבָּה.

וַיֹּאמֶר הַשּׂר: אֹרֶךְ פְּלוֹנִי אֹרֶךְ הַצְּלָלִים לִפְנוֹת הַיּוֹם

וָאַעַן וָאֹמַר: אֹרֶךְ קֵץ הָאֲסוּרִים מְקַוִּים לְפִדְיוֹם.

וַיֹּאמֶר הַשּׂר: אֹרֶךְ פְּלוֹנִי אֹרֶךְ מִגְדַּל דּוֹר הַפְלָגָה

וָאַעַן וָאֹמַר: אֹרֶךְ יְמֵי הַדְּאָגָה.

וַיֹּאמֶר הַשּׂר: אֹרֶךְ פְּלוֹנִי אֹרֶךְ יְמֵי מְתוּשֶׁלַח

וָאַעַן וָאֹמַר: אֹרֶךְ יַם הַמֶּלַח.

וַיֹּאמֶר הַשּׂר: אֹרֶךְ פְּלוֹנִי אֹרֶךְ צוֹם הַרְבִיעִי

וָאַעַן וָאֹמַר: אֹרֶךְ הַגַּלְגַּל הַתְּשִׁיעִי.

וַיֹּאמֶר הַשָּר: יֵשׁ-לִי אִשָּׁה מְאֹד יְרֵאָה

וְחָפַצְתִּי לְתָאֵר בַּעֲבוּרָה הַיִּרְאָה

וַיֹּאמֶר הַשּׂר: צֹרֶךְ הַסְּפִינָה לְרַב הַחוֹבֵל

וַיֹּאמֶר הַשּׂר: צֹרֶךְ הַמֵת אֶל הַקֶּבֶר

וָאַעַן וָאֹמַר: צֹרֶךְ הָרָעֵב אֶל הַשֶּׁבֶר.

וַיֹּאמֶר הַשּׂר: צֹרֶךְ

אַתָּה תְּתָאֵר הַשִּׂנְאָה

וְהִנֵּה תְּשׁוּבָתִי אַחֲרֶיךָ בָאָה.

וָאֹמְרָה: פְּלוֹנִי שְׂנֵאתִיו שִׂנְאַת הַחַלָּשׁ אֶל הַסִּבּוּב

וַיֹּאמֶר הַשָּר: שִׂנְאַת הַזְּאֵב אֶל הַכֶּלֶב.

וָאֹמְרָה: שִׂנְאַת הֶעָשִׁיר אֶל הֶעָנִי

וַיֹּאמֶר הַשָּר: שִׂנְאַת הַמְּזַנֶּה אֶל אִישׁ-מְלָשְׁנִי.

וָאֹמְרָה: שִׂנְאַת הַשִּׁכּוֹר לַנָּזִיר

וַיֹּאמֶר הַשָּר: שִׂנְאַת הַיִּשְׁמְעֵאלִי אֶל הֶחָזִיר.

וָאֹמְרָה: שִׂנְאַת בַּעַל הַגָּרָב לַזְּבוּבִים

וַיֹּאמֶר הַשָּר: שִׂנְאַת הַכִּילַי לַנְּדִיבִים.

וָאֹמְרָה: שִׂנְאַת הַמַּיִם אֶל הָאֵשׁ

וַיֹּאמֶר הַשָּר: שִׂנְאַת הֶחָרוּץ אֶל הַמִּתְיָאֵשׁ.

וָאֹמְרָה: שִׂנְאַת הָאִשּׁה לְמָעוּך וְכָתוּת

וַיֹּאמֶר הַשָּר: שִׂנְאַת הַמְּגֹרֶשֶׁת לְגֵט כְּרִיתוּת.

וָאֹמְרָה: שִׂנְאַת הַחֲמוֹר אֶל הַבָּצִיר

וַיֹּאמֶר הַשָּר: שִׂנְאַת הֶעָצֵל אֶל עֵת הַקָּצִיר.

וָאֹמְרָה: שִׂנְאַת הָאָסִיר אֶל הָכֶּבֶל

וַיֹּאמֶר הַשָּר: שִׂנְאַת הַגַּנָב אֶל הַחֶבֶל.

וָאֹמְרָה: שִׂנְאַת הָעַכְבָּר אֶל הָעַקְרַבִּים

וַיֹּאמֶר הַשָּר: שִׂנְאַת הַיַיִּן אֶל הָעֲצֵבִים.

וָאֹמְרָה: שִׂנְאַת הָאַף אֶל הַזָּבֶל

וַיֹּאמֶר הַשָּר: שִׂנְאַת הַמִתְאַבְּלִים לַפֹּרְטִים עַל פִּי הַנָּבֶל.

וָאֹמְרָה: שִׂנְאַת הֶעָרֹם אֶל הַסְּתָיו

וַיֹּאמֶר הַשָּר: שִׂנְאַת הַמְּהַלֵּך בַּשָּׁרָב לְרֹב כְּסוּתָיו.

וָאֹמְרָה: שִׂנְאַת הַכְּעוּרָה אֶל הַמַּרְאָה

וַיֹּאמֶר הַשָּר: שִׂנְאַת הַשֹּׁבָב לְדִבְרֵי הַנְּבוּאָה.

וָאֹמְרָה: שִׂנְאַת הָאִשָּׁה לַסָּרִיס

וַיֹּאמֶר הַשָּר: שִׂנְאַת הַשּׁוֹר לָאָרִיס וְהָרֹאשׁ לַגָּרִיס.

וָאֹמְרָה: שִׂנְאַת הַחשֵׁק לַמִּתְאַמֶּצֶת

וַיֹּאמֶר הַשָּר: שִׂנְאַת חַג הַמַּצּוֹת לַמַּחְמֶצֶת.

וָאֹמְרָה: שִׂנְאַת הַשׁוֹר לַדָּרְבָן

וַיֹּאמֶר הַשָּר: שִׂנְאַת עַם הָאָרֶץ לְרַבָּן.

וָאֹמְרָה: שִׂנְאַת הַזּוֹלֵל וְסוֹבֵא אֶל הַצּוֹמוֹת

וַיֹּאמֶר הַשָּר: שִׂנְאַת בַּעַל מָרָה שְׁחוֹרָה אֶל הַחֲלוֹמוֹת.

וָאֹמְרָה: שִׂנְאַת הֶעָנִי אֶל הַבַּת

וַיֹּאמֶר הַשָּר: שִׂנְאַת הַכִּנִּים לַשֶּׁמֶש וְשַׁבָּת.

וָאֹמְרָה: שִׂנְאַת הַנִּכְבָּד לִבְנֵי פִּרְחָח

וַיֹּאמֶר הַשָּר: שִׂנְאַת הַדָּג אֶל הֶחָח.

וָאֹמְרָה: שִׂנְאַת הַנָּדִיב אֶל הַנָּבָל

וַיֹּאמֶר הַשָּר: שִׂנְאַת הָרְשָעִים אֶל הַר עֵיבָל.

וָאֹמְרָה: שִׂנְאַת בַּעַל הַגָּרָב אֶל הַטָּרִית

וַיֹּאמֶר הַשָּר: שִׂנְאַת אַבְרָהָם לְמֵפֵר בְּרִית.

וָאֹמְרָה: שִׂנְאַת הֶחָתוּל אֶל הָעַכְבָּרִים

וַיֹּאמֶר הַשָּר: שִׂנְאַת בְנוֹת הַחֵן לִבְנֵי הַנְּזִירִים.

וָאֹמְרָה: שִׂנְאַת הַחֲכָמִים לְאִיש נָבוּב וָרֵק

וַיֹּאמֶר הַשָּר: שִׂנְאַת הַנְעָרִים לִרְחִיצַת הָרֹאשׁ וְהַמַּסְרֵק.

וָאֹמְרָה: שִׂנְאַת הָעֲלָמוֹת לְאַנְשֵׁי הַשֵּׂיבָה

וַיֹּאמֶר הַשָּר: שִׂנְאַת אַנְשֵׁי סוֹד לְמוֹצִיאֵי דִּבָּה.

וָאֹמְרָה: שִׂנְאַת הַתַּרְנְגוֹלִים לְיוֹם הַתְּרוּעָה

וַיֹּאמֶר הַשָּר: שִׂנְאַת הָעֶבֶד וְהַתַּלְמִיד לְמַקֵּל וּרְצוּעָה.

וַיֹּאמֶר הַשָּׂר אֵלַי: חַי אֲנִי בְּזֹאת תִּבָּחֵן

אִם בְּעֵינֵי הַמְּשֹׁרְרִים תִּמְצָא חֵן

אֲנִי אֲתָאֵר הָאַהֲבָה בִּמְשָׁלָי

וְאַתָּה לְתַאֵר הַשִּׂנְאָה תָּשִׁיב אֵלָי

וָאַעַן וָאֹמַר כָּל מַחְסוֹרְךָ עָלָי

וַיֹּאמֶר הַשּׂר: אַהֲבַת הָאָב אֶל הַבֵּן

וָאַעַן וָאֹמַר: שִׂנְאַת הַתּוֹפֵר אֶל הַגִּבֵּן.

וַיֹּאמֶר הַשּׂר: אַהֲבַת הֶעָצֵל אֶל הַשֵּׁנָה

וָאַעַן וָאֹמַר: שִׂנְאַת הַבָּחוּר לַזְּקֵנָה.

וַיֹּאמֶר הַשּׂר: אַהֲבַת הַיִשְׁמְעֵלִים אֶל הַיִּצְהָר

וָאַעַן וָאֹמַר: שִׂנְאַת הַזְּקֵנִים עֲלוֹת בַּהָר.

וַיֹּאמֶר הַשּׂר: אַהֲבַת הָאָדָם לִרְאוֹת פֶּלֶא

וָאַעַן וָאֹמַר: שִׂנְאַת הָאָסִיר לְבֵית כֶּלֶא.

וַיֹּאמֶר הַשּׂר: אַהֲבַת הָאִישׁ אֶל הָעֹשֶׁר

וָאַעַן וָאֹמַר: שִׂנְאַת הָרַמַאי אֶל הַיֹּשֶׁר.

וַיֹּאמֶר הַשּׂר: אַהֲבַת הַסּוֹבֵא אֶל הַיָּיִן

וָאַעַן וָאֹמַר: שִׂנְאַת הַחֲסִידִים לִכְלֵי זָיִן.

וַיֹּאמֶר הַשּׂר: אַהֲבַת הַחֲשׁוּקָה לְדוֹדָהּ

וָאַעַן וָאֹמַר: שִׂנְאַת הַכְּסִילִים לְתוֹרָה וּלִתְעוּדָה.

וַיֹּאמֶר הַשּׂר: אַהֲבַת הַקֵּרֵחַ אֶל הַכֹּבַע

וָאַעַן וָאֹמַר: שִׂנְאַת בַּעַל הַבָּר אֶל הַשֹּׂבַע.

וַיֹּאמֶר הַשּׂר: אַהֲבַת הַשִּכּוֹר לְפוּרִים

וָאַעַן וָאֹמַר: שִׂנְאַת הַזּוֹלֵל וְסוֹבֵא לְיוֹם הַכִּפּוּרִים.

וַיֹּאמֶר הַשּׂר: אַהֲבַת הָאִשָּׁה לְבִתָּהּ

וָאַעַן וָאֹמַר: שִׂנְאַת הַכַּלָּה לַחֲמוֹתָהּ.

וַיֹּאמֶר הַשּׂר: אַהֲבַת הַשֻּׁעָל לַעֲלוֹת בַּפְרָצוֹת

וָאַעַן וָאֹמַר: שִׂנְאַת הַיָּחֵף לְטִיט חוּצוֹת.

וַיֹּאמֶר הַשָּׂר: יֶשׁ-לִי שְׁכֵנָה תַּחְפִּיר הַשֶּׁמֶש בְּהִלָּהּ

וּבַעְלָהּ מְעוּף צוּקָה וַאֲפֵלָה

נְדַבֵּר עַל שְׁנֵיהֶם מֵעֵין הַמְּאֹרָע

וּנְדַבֵּר עַל הַטּוֹב טוֹב וְעַל הָרָע רָע.

וָאֹמַר אֶל הַשָּׂר הִנֵּנִי

וּלְכָל אֲשֶׁר תַחְפֹּץ תַּטֵּנִי

וַיֹּאמֶר הַשּׂר: פְּלוֹנִית תַחְפִּיר הַיָרֵחַ וּבַעֲלָהּ כְּתוֹא מִכְמָר

וָאַעַן וָאֹמַר: וּבִשְׁנֵיהֶם מַרְאֵה הָעֶרֶב וְהַבֹּקֶר אֲשֶׁר נֶאֱמָר.

וַיֹּאמֶר הַשּׂר: פְּלוֹנִית מַעֲשֶׂה יְדֵי אָמָּן

וָאַעַן וָאֹמַר: וּבַעֲלָהּ לֹא יַעֲשֶׂה בּוֹ הַפֶּחָם סִמָֹן.

וַיֹּאמֶר הַשּׂר: פְּלוֹנִית לֹא תֻכָּר בֵּין הַמַּלְאָכִים אִם בָּהֶם תִּתְעָרֵב

וָאַעַן וָאֹמַר: וּבַעֲלָהּ אִלּוּ הָיוּ לוֹ כְּנָפַיִם חֲשַׁבְנוּהוּ לְעֹרֵב.

וַיֹּאמֶר הַשּׂר: פְּלוֹנִית לֹא יָבוּזוּ לַנּוֹתֵן בְּאַהֲבָתָהּ הַנֶּפֶשׁ וְהָרְכוּשׁ

וָאַעַן וָאֹמַר: וּבַעֲלָהּ יֹאְמרוּ בַעֲדוֹ יְגִיעַ מִצְרַיִם וּסְחַר כּוּשׁ.

וַיֹּאמֶר הַשּׂר: כִּמְעַט שֶׁלֹּא עָשִׂיתִי שְׁתִי וָעֶרֶב

וָאַעַן וָאֹמַר: בִּרְאוֹתִי הַגְּבִירָה יוֹם וּבַעֲלָהּ עֶרֶב

וַיֹּאמֶר הַשּׂר: כִּמְעַט שֶׁלֹּא עָשִׂיתִי עֶרֶב וָשֶׁתִי

וָאַעַן וָאֹמַר: בִּרְאוֹתִי הַגְּבִירָה חֲכָמָה וּבַעֲלָהּ פֶּתִי

וַיֹּאמֶר הַשּׂר: יֶשׁ-לִי חָבֵר תִּנְעַם אֵלַי חֶבְרָתוֹ וְלִבִּי בּוֹעֵר בְּאָהֲבָתוֹ

וְעַתָּה נַמְשִׁיל מֹתֶק הַחֶבְרָה אֲשֶׁר לֹא יֻתַּן סְגוֹר תַּחְתֶיהָ וְלֹא יִשָּׁקֵל כֶּסֶף מְחִירָהּ

וָאֹמַר אֶל הַשָּׂר: חוּדָה חִידָתְךָ וְאֶשְׁמָעֶנָה

וַאֲנִי אֶעֱרֹךְ תְּשׁוּבוֹתַי וְאֶעֱנֶנָה

וַיֹּאמֶר הַשּׂר: מֹתֶק הַצּוּף וְהַחֶמְאָה

וָאַעַן וָאֹמַר: מֹתֶק שִׂפְתֵי הַנָּאָה

וַיֹּאמֶר הַשּׂר: מֹתֶק חָשׁוּק עִם חֲשׁוּקָתוֹ

וָאַעַן וָאֹמַר: מֹתֶק רְאוֹת סִיסְרָא וְהַיָּתֵד בְּרַקָּתוֹ.

וַיֹּאמֶר הַשּׂר: מֹתֶק מַיִם קָרִים עַל נֶפֶשׁ עֲיֵפָה

וָאַעַן וָאֹמַר: מֹתֶק חֶבְרַת הַיָּפָה.

וַיֹּאמֶר הַשּׂר: יָדַעְתִּי כִּי כָּל הָאֲנָשִׁים אֲשֶׁר מַעֲלוֹתֵינוּ רוֹאִים

בָּנוּ מְקַנְּאִים

וְאַתָּה נְתָאֵר הַקִּנְאָה

וָאַעַן וָאֹמַר: חוּדָה חִידָתְךָ וְהִנֵּה מְלִיצָתִי אַחֲרֶיהָ בָאָה.

וַיֹּאמֶר הַשּׂר: קִנְאַת הַבְּתוּלָה אֶל הַנְּשׂוּאָה

וָאַעַן וָאֹמַר: קִנְאַת הַכְּעוּרָה אֶל הַנָּאָה.

וַיֹּאמֶר הַשּׂר: קִנְאַת הַחוֹלֶה אֶל הַבָּרִיא

וָאַעַן וָאֹמַר: קִנְאַת הָרָעֵב לַזּוֹבֵחַ שׁוֹר וּמְרִיא

וַיֹּאמֶר הַשּׂר: קִנְאַת עַם הָאָרֶץ אֶל הָרַבָּנִים

וָאַעַן וָאֹמַר: קִנְאַת הָעֲקָרָה לְרַבַּת בָּנִים.

וַיֹּאמֶר הַשּׂר: קִנְאַת אֵשֶׁת הַפֶּתִי לְאֵשֶׁת רַב פְּעָלִים

וָאַעַן וָאֹמַר: קִנְאַת אַשֶׁת הַחַלָּשׁ לְאֵשֶׁת מְגַבֵּר חֲיָלִים.

וַיְהִי כִּשְׁמֹעַ הַשָּרִים נֹעַם מְשָלֵינוּ

נָסוּ לְקוֹלֵנוּ

וְכִמְעַט שְׂרָפוּם חַשְמַלֵּינוּ

וְיֶתֶר הָעָם וְהַשָּרִים

וְהַסְּגָנִים וְהַחוּרִים

אָמְרוּ הַיּוֹם הַזֶּה רָאִינוּ נוֹרָאוֹת

וְשָׁמָעְנוּ נִפְלָאוֹת

אֵין אֵלוּ אֶלָא דִּבְרֵי נְבוּאוֹת

וַיְבָרְכוּנוּ בְּשֵׁם יְיָ צְבָאוֹת

וְאַחֲרֵי כֵן נִפְרְדוּ הַקְּהִלּוֹת

מְדַבְּרִים רַק מְחִידוֹתֵינוּ הַמְּהֻלָּלוֹת

וָאֵלֵךְ עִם הַשָּׂר אֶל מְקוֹם תַּחֲנֹתוֹ

וְכָל הָעָם הָלְכוּ אִישׁ לְבֵיתוֹ.
