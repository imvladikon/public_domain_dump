לאנשי לבבות בני מחשבות

לְאַנְשֵׁי לְבָבוֹת בְּנֵי מַחֲשָׁבוֹת / ר' משה אבן-עזרא

לְאַנְשֵׁי לְבָבוֹת / בְּנֵי מַחֲשָׁבוֹת / בְּיָדָם נְעִמוֹת / לְמִחְיַת בְּרוּאִים

וְעֵינֵי שְׂעִפִּים / עֲתִידוֹת מְצַפִּים / וְאִסַּר זְמַנִּים / בְּלִי פֶה מְנִיאִים

וְדֵעוֹת כְּמַרְאוֹת / רְאוֹת הַפְּלִיאוֹת / וְעֵצוֹת כְּמַרְאוֹת / וְחֶזְיוֹן נְבִיאִים

יְשׁוּרוּן עֲצֻמוֹת / וְסוֹד תַּעֲלוּמוֹת / וְהֵמָּה בְּמִסְגַּר / פְּגָרִים כְּלֻאִים

וּמַה טּוֹב נְפָשִׁים / בְּמִצְעַר אֲנָשִׁים / בְּמַחְנוֹת קְדֹשִׁים / וְאֶל עָל קְרֻאִים

כְּעֶצֶם שְׁחָקִים / וְזֹהַר בְּרָקִים / עֲלֵי עָשׁ עֲמֻסִים / וְכִימָה נְשֻׂאִים

וְלוּלֵי יְצוּעִים / בְּלִבּוֹת לְדֵעִים / אֲזַי גָּבְרוּ מִ-/ גְּבָרִים לְבָאִים

וְיֵצֶר יְשִׂיחֵם / וְשֵׂכֶל יְרִיחֵם / עֲדֵי נִבְדְּלוּ מֵ-/ עֲיָרֵי פְרָאִים

וְיֵשׁ בַּעֲלָמִים / מְסַכֵּל חֲכָמִים / וְצָעִיר לְיָמִים / מְבָאֵר פְּלָאִים

וּבִקְרָב מְלִיצִים / לְלַחֵם חֲלוּצִים / לְהַפִּיל נְפִילִים / וְלַחְלשׁ רְפָאִים

כְּגוּרֵי נְמֵרִים / בְּיַלְדֵי שְׂעִירִים / וְשַׁאְגַת כְּפִירִים / בְּעֶדְרֵי טְלָאִים

לְאָחוֹר מְשִׁיבִים / עֲלָמִים וְשָׂבִים / וְאֹתָם בְּאֹתוֹת / אֱמֶת מַחֲטִיאִים

וְכַמָּה יְשִׁישִׁים / כְּגֹבַהּ בְּרוּשִׁים / גְּדֹלוֹת מְבַקְשִׁים / וְהֵם אַךְ נְכָאִים

תְּמוּנַת גְּבָרִים / תְּבוּנַת עֲיָרִים / וְדֵעוֹת גְּמוּלִים / וְגִשְׁמֵי מְרִיאִים

מְתֵי הוֹד וּמִדּוֹת / צְנִיפִים וּמִדּוֹת / וּמִסִּיג אֲבוּסִים / וְעָוֶל בְּרִיאִים

וּמִפּתַ פְּתַיּוּת / שְׂבֵעִים וְסִכְלוּת / וּמִיֵין כְּסִילוּת / יְמוֹתָם סְבוּאִים

וּמֵהֶם אֲלָפִים / אֲבָל כַּאֲלָפִים / עֲלֵי מַאֲבוּסִים / בְּכָל טוּב מְלֵאִים

גְּבֹהוֹת יְשִׂיחוּן / וְיוֹם רִיב יְרִיחוּן / לְהַשְׂכֵּל יְבֹאוּן / בְּתוֹךְ מַחֲבֹאִים

וּפֹצְחִים בְּמִדְבָּר / כְּרוּחַ בְּמִדְבָּר / וְאַךְ לֹא לְחָבַר / כְּרוּחַ שְׁפָאִים

וּמָשָׁל לְפָנִים / הֲכִי הוֹן עֲוֹנִים / מְסַתֵּר וְְעשֶׁר / מְכַסֶּה חֲטָאִים

קְחוּ לִי מְסַפֵּר / בְּפִי עֵט וְסֵפֶר / לְצַפְצֵף בְּשֶׁפֶר / כְּעוֹף בֵּין עֳנָפִים

לְכַתֵּב אֲמָרִים / לְשֹׁכְנֵי עֲבָרִים / בְּסַפִּיר סְפָרִים / בְּנֹפֶךְ מְלֵאִים

לְאַחִים נְפוּצִים / בְּלִבִּי קְבוּצִים / צְפוּיִם בְּעֵינֵי / זְמָמַי וְנִרְאִים

וּבֵין בֹּעֲרֵיהֶם / יְבַלּוּ יְמֵיהֶם / בְּהֶבֶל כְּגֵרִים / וְגָרֵי כְלָאִים

הֱיוּ נָא נְכֹנִים / לְשֻׁלְחַן אֱמוּנִים / וְגשׁוּ רְעֵבִים / וְהוֹי כָּל צְמֵאִים

לְרַוֹּת אֲמָרִים / כְּנוֹזְלֵי שְׂעִירִים / בְּגַנֵּי כְפָרִים / וְיֶרֶק דְּשָׂאִים

כְּנֹפֶת לְפיּוֹת / וְעַל חֹם כְּלָיוֹת / כְּפֶגֶ בְּצִיּוֹת / וּמַלְקוֹשׁ נְשִׂיאִים

וְלָעוּת יְעֵפִים / דְּבָרִים כְּצוּפִים / וְקֹר מֵי עֲדָנִים / וְלֹא מֵי גְבָאִים

וְאֶל יָם מְזִמּוֹת / מְשָׁלִים וְחָכְמוֹת / וְגָבִישׁ וְרָאמוֹת / בּחוֹפוֹ מְצוּאִים

בְּמֵימֵי מְצוּלוֹת / פְּנִינֵי תְהִלּוֹת / וּמֹר וַאֲהָלוֹם / נְבָכָיו מְקִיאִים

לְבָרִים כְּצוּפִים / וְרוֹשׁ לַחֲנֵפִים / וְעֹשֶר לְחָפִים / וְחֹסֶר מְשַׂנְאִים

וּמִכָּל עֲבָרָיו / עֲצֵי מֹר לְאֹרָיו / וְטַרְפֵּי אֱמִירָיו / תְּרוּפַת חֳלָאִים

נְגִידֵי אֲמָרִים / לְעִבְרִים וְהַגְרִים / כְּאֵלִים לְעִמְקֵי / שְׂפָתוֹת וְתָאִים

וְלֶקַח לְרוֹגְנִים / וְצַחוֹת לְשׁוֹנִים / וְעֵצָה וְאָמְצָה / לְשָׂרֵי צְבָאִים

פְּרֻדֵי לְשֹׁנוֹת / לְאֹסֵף תְּבוּנוֹת / וְחֶמְאַת אֲמָרוֹת / לְחֶמְדַּת נְשִׂיאִים

לְחֹרֶשׁ בְּאִבּוֹ / בְּגַן אֵל מְסִבּוֹ / וְכָל עֵץ אֲשֶׁר בּוֹ / הֲדָרוֹ מְקַנְאִים

מְגַבֵּר חֲיָלִים / בְּפִי עֵט וּמִלִּים / וְלוֹ עֹז שְׁחָלִים / וְחֵן הַצְּבָאִים

וְלָשׁוֹן תְּשׁוֹבֵב / מְלִיצוֹת לְנוֹבֵב / נְפָשִׁים תְּשׁוֹבֵב / וְתַחְכִּים פְּתָאִים

קְרָסִים לְמַעְנָיו / בְּנֵי רַעֲיוֹנָיו / בְּלֻלְאוֹת אֲמָרָיו / בְּלִי יָד מְבִיאִים

בְּאֹזְנֵי כְסִילִים / נְתָנָם עֲגִילִים / וְעַל צַוְּרוֹנֵי / לְבָנָה חֲלָאִים

וּבִידֵי חֲסָדִים / נְאֻמָיו צְמִידִים / וְחַיָּיו בְּחַיֵּי/ בְּנֵי עָשׁ תְּלוּאִים

וְהִנֵּה יְלָדִים / לְהָקִיץ עֲתִידִים / יְשֵׁנֵי עֲפָרוֹת / וְעוֹרֵר רְפָאִים

וְנֹשְׂאִים בְּפִיהֶם / וְלֹא עַל יְדֵיהֶם / גְּדֻלַּת אֲבִיהֶם / עֲדֵי גוֹי שְׁבָאִים

וְכוֹכְבֵי אֲמָרוֹת / לְהָאִיר וְהוֹרוֹת / אֲנָשִׁים בְּמִסְגַּר / אֱפֵלָה יְרֵאִים.
