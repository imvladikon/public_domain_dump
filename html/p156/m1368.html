מחברות עמנואל – מחברת ארבע-עשרה / עמנואל הרומי

אָמַר עִמָּנוּאֵל הַמְחַבֵּר: בִּהְיוֹתִי בְּבֵית הַשָּׂר שָׁמֵן וְדָשֵׁן • שׁוֹכֵב עַל מִטּוֹת שֵׁן • וּבְמַעֲדַנָּיו אֶשְׁלַח הַשֵּׁן • הָיִיתִי בְחוּצוֹת הַשְּׂמָחוֹת מְגַמֵּא • וְסוֹבֵב עַל מַעְיְנֵי הַשְּׂשׂוֹנִים סִבּוּב הַצָּמֵא • וַיְהִי הַיּוֹם וַאֲנִי אֶת רַגְלַי חָלִיתִי • וְעָמַדְתִּי יָמִים שִׁבְעָה וּפְנֵי הַשָּׂר לֹא רָאִיתִי • וְיַיִן וְשֵׁכָר לֹא שָׁתִיתִי • וְכַאֲשֶׁר הִתְחַזַּקְתִּי וְיָצָאתִי מִפֶּתַח בֵּיתִי • וָאֶשָּׂא עֵינַי וָאֶרְאֶה וְהִנֵּה אֵם הַשָּׂר בָּאָה לִקְרָאתִי • וָאָרוּץ לִקְרָאתָהּ וָאֹמַר הֲשָׁלוֹם לָךְ גְּבִרְתִּי • הֲשָׁלוֹם לַשָּׂר הֲשָׁלוֹם לַיְלָדִים כִּי בָאת הֲלוֹם • וַתֹּאמַר שָׁלוֹם • אָמְנָם הַשָּׂר שְׁמוּעָה שָׁמַע וְחָרַד חֲרָדָה • וְצִירִים אֲחָזוּהוּ כְּצִירֵי יוֹלֵדָה • סוֹעֵר כַּהֲמוֹת הַיָּם לְגַלָּיו • וּשְׁנָתוֹ נִהְיְתָה עָלָיו • וְעַתָּה הַדּוֹד הַנֶּאֱמָן • הַחוֹמֵל עָלָיו וְהָרַחֲמָן • לָמָּה תַעֲמֹד בְּרָחוֹק וְתַעְלִים לְעִתּוֹת בַּצָּרָה • וְהַיְגוֹנִים • חוֹנִים • עַל הַשָּׂר אִישׁ עַל דִּגְלוֹ בְּאוֹתוֹת • בּוֹאָה אֵלָיו הַחַדְרָה וְנַהֲלֵהוּ עַל מֵי מְנוּחוֹת • וְיִהְיֶה לוֹ בִרְאוֹתוֹ פָנֶיךָ שֹׂבַע שְׂמָחוֹת • וְשָׂא אֵלָיו חִידוֹתֶיךָ הַמְשַׂמְּחוֹת • יְכַבּוּ אֵשׂ הַיְגוֹנִים • וְיָאִירוּ נֵרוֹת הַשְּׂשׂוֹנִים • וָאַעַן וָאֹמַר הִנְנִי גְבִרְתִּי • וָאָקוּם קַל חִישׁ נִשְׁעָן עַל מִשְׁעַנְתִּי • וָאֵלֵךְ אֶל הַשָּׂר חֶדֶר בְּחָדֶר וּמְצָאתִיו הוֹמֶה וְסוֹעֵר • וּבְכָל מַחֲמַדֵּי הַזְּמָן גּוֹעֵר • וְלִבּוֹ בְּאֵשׁ הַיְגוֹנִים בּוֹעֵר • וְעִמּוֹ חֶבְרַת מְיֻדָּעָיו • וְגוֹאֲלָיו וְרֵעָיו • וּבִרְאוֹתָם אוֹתִי מֵרָחוֹק • וּבְיָדִי הַמִּשְׁעֶנֶת הֵחֵלּוּ לִשְׂחוֹק • וַיֹּאמְרוּ מִי הֱבִיאֲךָ הֲלוֹם וְאַתָּה צָרִיךְ לְמִשְׁעֶנֶת • וָאַעַן וָאֹמַר אַהֲבַת הַשָּׂר הַנֶּאֱמֶנֶת • וַיֹּאמְרוּ הֲיָדַעְתָּ חִידָה מְשַׂמַּחַת • תִּהְיֶה אֶל הַשָּׂר בִּשְׁמוֹעַ אוֹתָהּ נַחַת • וְלֹא יָמוּת לַשַּׁחַת • וָאַעַן וָאֹמַר יָדַעְתִּי וְלֹא אוּכַל לְהַגִּידָהּ • וְעַתָּה חוּסוּ עַל נַפְשִׁי הָאוֹבְדָה • לְהָכִין אוֹתָהּ וּלְסַעֲדָהּ • הֲטֶרֶם תֵּדְעוּ כִּי אָבְדָה • הֲלֹא תִרְאוּ עֲצָמַי אֲשֶׁר הִתְדַּלְדָּלוּ • וְהָעֲגָבוֹת אֲשֶׁר הִתְקַלְקָלוּ • וְאַתֶּם אֶת הַחִידָה תִשְׁאָלוּ • לָמָּה זֶה הֶבֶל תֶּהְבָּלוּ • הָבוּ שְׂכָרִי וְאִם לֹא חֲדָלוּ • וַיִּשְׂחַק הַשָּׂר וַיֹּאמַר וּמַה יִהְיֶה הַשָּׂכָר • וָאַעַן וָאֹמַר אַטִּיף לְךָ לַיַּיִן וְלַשֵּׁכָר • וַיְצַו הַשָּׂר הַנֶּעֱרָב • וַיָּבִיאוּ יֵין מַלְכוּת רָב • וָאֹמַר בָּרוּךְ יְיָ צוּרִי הַמְלַמֵּד יָדַי לַקְרָב • וָאָקוּם וָאֶשְתֶּה יַיִן לְשָׂבְעָה • כִּי לֹא שָׁתִיתִי מִמֶּנּוּ יָמִים שִׁבְעָה • וַיְהִי כַּאֲשֶׁר שָׂבַעְתִּי מִן הַיַּיִן • הָיוּ יְגוֹנֵי הַשָּׂר בְּעֵינָיו אַיִן • וַיֹּאמַר עַד מָתַי חִידָתְךָ בְּתוֹךְ חִכְּךָ תִמְנָעֶנָּה • חוּדָה חִידָתְךָ וְנִשְׁמָעֶנָּה • וָאַעַן וָאֹמַר הַאֲזִינוּ מְיֻדָּעַי וַאֲדַבֵּרָה • וְזֶה חָזִיתִי וַאֲסַפֵּרָה • אִישׁ הָיָה בְּאֶרֶץ רוֹמָ“א מֹשֶׁה שְׁמוֹ • רָצוּי לְרֹב אֶחָיו וְדוֹרֵשׁ טוֹב לְעַמּוֹ • וְשֵׁם אִשְתּוֹ רוֹשׂוֹלָה • אֲשֶׁר יָרַד אֵלֶיהָ אָבֵל שְׁאוֹלָה • וְשֵׁם בְּנוֹ דָנִיֵּאל אֱוִיל שְׂפָתָיִם • חֲמֹר גָּרֶם רוֹבֵץ בֵּין הַמִּשְׁפְּתָיִם • לֹא מָצָא בְּבֵיתוֹ נַחַת • רַק פַּח וָפַחַד וָפַחַת • בָּרָד וְאֵשׁ מִתְלַקַּחַת • כִּי הָיוּ מִדֵּי יוֹם יוֹם נִצִּים • הִנֵּה הָאֵשׁ וְהָעֵצִים • יָקוּמוּ זֶה עַל זֶה כְּהַר פְּרָצִים • וַיִּצְעַק הָאִישׁ אֶל יְיָ מִפְּנֵי לוֹחֲצִים • כִּי בְנוֹ מָשַׁךְ יָדוֹ אֶת לוֹצְצִים • וְאִשְׁתּוֹ תוֹרֶה בְלִבּוֹ מִן הַכְּלִמָּה חִצִּים • וְהַנּוֹגְשִׂים אָצִים • וַיֹּאמֶר הָאִישׁ בְּלִבּוֹ מִשֶּׁבֶת עוֹד אִתָּם נִבְעַתִּי • פֶּן תִּדְבָּקַנִי הָרָעָה וּמַתִּי • אֵין לִי טוֹב רַק לְהַנִּיחַ אֵלֶּה הָאֲמֵלָלִים • וְלִגְלוֹת בְּרֹאשׁ גּוֹלִים • אוּלַי יִקָּרֶה יְיָ לִקְרָאתִי • וְאֶמְצָא מָקוֹם תִּהְיֶה שָׁם מְנוּחָתִי • וַיְהִי הַיּוֹם וַיִּקַּח הָאִישׁ הוֹנוֹ וּרְכוּשׁוֹ • וַיָּקָם וַיֵּלֶךְ אֶל נַפְשׁוֹ • וַיִּקֶּר מִקְרֵהוּ בְּאֶרֶץ מֵאַרְצוֹת יָוָן אֲשֶׁר שָׁם אַנְשֵׁי אֱמוּנָה • הָיוּ עִם הַשֵּׂכֶל בְּאָמְנָה • בְּשִׁבְטֵי יִשְׂרָאֵל מוּדַעַת נֶאֱמָנָה • וְשֵׁם הָעִיר שִׂבְמָה • וַיַּרְא הָאִישׁ מְנֻחָה כִּי טוֹב וְאֶת הָאָרֶץ כִּי נָעֵמָה • וַיִּטַּע שָׁם אָהֳלֵי אַפַּדְנוֹ • וַיִּשְׁכַּח שָׁם צָרַת אִשְׁתּוֹ וּבְנוֹ • וַיֹּאמַר זֹאת הָאָרֶץ אֲהַבְתִּיהָ • אִוִּיתִיהָ חֲמַדְתִּיהָ • זֹאת מְנוּחָתִי עֲדֵי עַד פֹּה אֵשֵׁב כִּי אִוִּיתִיהָ • וַיֶּאֱהָבוּהוּ אַנְשֵׁי הַמָּקוֹם עַל יָשְׁרוֹ • וְעַל נֹעַם פְּעָלָיו וּכְבוֹד עָשְׁרוֹ • כִּי כָלִיל הוּא בַהֲדָרוֹ • וַיֹּאמְרוּ בּוֹא יָבוֹא בְרִנָּה נֵזֶר הוֹדֵנוּ • הַמַּשְׂבִּיעַ בַּטּוֹב עֶדְיֵנוּ • הַמְחַדֵּשׁ כְּבוֹדֵנוּ • זֶה יְנַחֲמֵנוּ מִמַּעֲשֵׂנוּ וּמֵעִצְּבוֹן יָדֵינוּ • וַיּוֹאֶל הָאִישׁ לָשֶׁבֶת אִתָּם • וְלִבּוֹ עָלֵז בְּחֶבְרָתָם • וַיְהִי שָׁם כְּעֶשְׂרִים שָׁנָה גָדוֹל לַיְּהוּדִים וְנֶאֱהָב • כָּבֵד מְאֹד בַּמִּקְנֶה בַּכֶּסֶף וּבַזָּהָב • כִּי הִצְלִיחוֹ הָאֵל עֹשָׂיו • וְהִרְוִיחוֹ בְכָל מַעֲשָׂיו • וַיְהִי הַיּוֹם חָלָה אֶת חָלְיוֹ אֲשֶׁר מֵת בּוֹ • וּבִרְאוֹתוֹ כִּי גָדַל צִירוֹ וּמַכְאוֹבוֹ • קָרָא אֵלָיו אַנְשֵׁי מְקוֹמוֹ • אֲשֶׁר יָדַע כִּי הֵם דּוֹרְשֵׁי שְׁלוֹמוֹ • וַיֹּאמַר שְׁמָעוּנִי אַנְשֵׁי שְׁלוֹמִי • הִנְנִי נֶאֱסָף אֶל עַמִּי • וְאֵין לִי יוֹרֵשׁ וְגוֹאֵל • רַק בֵּן אֶחָד בְּרוֹמָ”א וּשְׁמוֹ דָנִיֵּאל • אֲחַלֶּה לִפְנֵיכֶם הָאֲצִילִים לִהְיוֹת הוֹנִי בְיֶדְכֶם לְמִשְׁמֶרֶת • וְהוֹדִיעוּ לִבְנִי מִיתָתִי בְאִגֶּרֶת • וּבְבוֹאוֹ הֵנָּה בְּעֵדוּת מבֹאֶרֶת • תְּנוּ לוֹ אֶת נַחֲלָתִי הַנִּשְׁאֶרֶת • וַאֲשֶׁר יָבוֹא בְּלֹא עֵדוּת וּמַזְכֶּרֶת • יֵחָשֵׁב כְּגַנָּב נִמְצָא בַּמַּחְתֶּרֶת • וַיַּעֲנוּ הָאֲנָשִׁים וַיֹּאמְרוּ אֲדֹנֵנוּ נֵזֶר הַיְקוּם • כַּאֲשֶׁר צִוִּיתָ אֶת עֲבָדֶיךָ כֵּן יָקוּם • וַיְהִי כְּכַלּוֹתוֹ לְדַבֵּר עִם אַנְשֵׁי שְׁלוֹמָיו • וַיֶּאֱסֹף רַגְלָיו אֶל הַמִּטָּה וַיִּגְוַע וַיָּמָת וַיֵּאָסֶף אֶל עַמָּיו • וַיָּבוֹאוּ כָל אַנְשֵׁי מְקוֹמוֹ לָנוּד לוֹ וּלְסָפְדוֹ • הוֹי אָדוֹן וְהוֹי הוֹדֹה • וַיִּקְבְּרוּהוּ בְמַעֲלֵה קִבְרֵיהֶם עַל צִדְקָתוֹ • וְכָבוֹד עָשׂוּ לוֹ בְּמוֹתוֹ • וַיִּזְכְּרוּ הָאֲנָשִׁים אֶת מִצְוַת וַיְהִי כַּאֲשֶׁר תַּם כָּל הָעָם לִשְׂחֹק לִרְצוֹנוֹ • וַיֹּאמֶר הַשָּׂר הֲיָדַעְתָּ מֶה עָשָׂה אֶפְרַיִם הַבּוֹגֵד מֵהוֹנוֹ • וָאַעַן וָאֹמַר זֶה פִּתְרוֹנוֹ • אֶפְרַיִם הָיָה עֻגָה בְּלִי הֲפוּכָה אֵיךְ לֹא יִשׁתּוֹק אִישׁ סָכָל וְּמ‑

חַרְחַר רִיב וּמְאַזֵּר זִיקוֹת

אִלּוּ שָׁתַק אִישׁ לֹא יֵדַע

אָז בָּטְלוּ כָּל הַמַּחְלוֹקוֹת •


כִּשְׁמֹעַ הַשָּׂר שִׁירַי וּמַעֲנָיו • אוֹרוּ עֵינָיו וְצָהֲלוּ רַעְיוֹנָיו • וַיֹּאמֶר אֶל הָעוֹמְדִים לְפָנָיו • הַרְאִיתֶם דָּבָר דָּבוּר עַל אָפְנָיו • חֵי אֲנִי כִּי הוּא יְפַזֵּר לְשָׁמָיו וּפְנִינָיו • כְּאִלּוּ יִינַק שְׁדֵי הַתְּבוּנָה • וְהָיְתָה הַחָכְמָה לוֹ לְמָנָה • וְהִנֵּה בִּהְיוֹת הַצָּרָה אוֹמַנְתּוֹ • וְיָקוּם וְיִתְהַלֵּךְ בַּחוּץ עַל מִשְׁעַנְתּוֹ • נִכְאָב וְדָוֶה • וְחִסֵּר נַפְשׁוֹ מִכָּל אֲשֶׁר יִתְאַוֶּה • הָרֵי הַתְּבוּנָה נָמַסּוּ לְפָנָיו כַּדּוֹנָג • וְהֵאִירוּ בְרָקָיו תֵּבֵל וְעַל שְׁדֵי הַבִּינָה יִתְעַנָּג • וּמַה יַּעֲשֶׂה אִם הָיוּ לוֹ חִטֵּי מִנִּית וּפַנַּג • חֵי אֲנִי כִּי אָז יַעֲשֶׂה בַשִּׁירִים נִפְלָאוֹת • וְתוֹרֵנוּ יְמִינוֹ נוֹרָאוֹת • וְיִהְיוּ דְבָרָיו דִּבְרֵי נְבוּאוֹת • וָאַעַן וָאֹמַר אֶל אֲדֹנִי • אַל יֹאמַר כֵּן נֵזֶר זְמַנִּי • כִּי לֹא נִבְרֵאת מַעֲלַת הָעֹשֶׁר וְהָעָצְמָה • לְצֹרֶךְ אַנְשֵׁי הַחָכְמָה • כִּי לֹא נִבְרְאוּ הָעֲטָרוֹת וְהַמְּעִילִים • רַק לְכַסּוֹת עֶרְוַת הָאֱוִילִים • כִּי בִלְעֲדֵי הָעֹשֶׁר טוֹב לָהֶם הַמַּחֲנָק • וְאֵין צֹרֶךְ אֶל הָעֹשֶׁר לַאֲשֶׁר שְׁדֵי בִינָה יִינָק • כִּי בִהְיוֹת הַצָּוָּאר לָבָן אֵין צֹרֶךְ לַעֲנָק •


וָאֶשָּׂא מְשָׁלִי וָאֹמַר:


יוֹם נִפְזָרוּ מִפִּי לִשְׁמֵי

בִּינָה וּשְׁדֵי חָכְמָה אִינָק

אָמְרוּ אִלּוּ יַעְשִׁיר זֶה אוֹ

יִתְנַשֵּׂא וּכְאַרְיֵה זָנָק

עָנִיתִי בִּהְיוֹת הַצַּוָּאר

לָבָן אֵין צֹרֶךְ אֶל עָנָק •


וַיֹּאמֶר הַשָּׂר אֶרְאֶה אוֹתְךָ הַיּוֹם מִן הַדְּבֵקִים • בַּתּוֹרָה וּבַמִּצְוֹת וּבַחֻקִּים • וְהָיִיתָ כְבָר נוֹשֵׂא דֶגֶל הַחֹשְׁקִים • וְנוֹטֶה אֹהֶל הַחֵשֶׁק וִירִיעוֹתָיו מֵקִים • וַתְּשַׂדֵּד אַחֲרָיו הָעֲמָקִים • וָאַעַן וָאֹמַר כֵּן דִּבַּרְתָּ וְהַדְּבָרִים עַתִּיקִים • אָמְנָם זָקַנְתִּי וָשַׂבְתִּי • וְלוּ תַאֲרִיכוּן יָמִים עוֹד תְּשׁוּבוּן מֵאֲשֶׁר שַׁבְתִּי • וְלִרְאוֹת בְּמַה שֶּׁיִּהְיֶה אַחֲרַי חָשַׁבְתִּי • כִּי בָא הַקֵּץ לְהָקִיץ מִשֵּׁנַת הַשִּׁכְּרוּת • וּלְאַבֵּד בָּמֳתֵי הַבַּחְרוּת • וּלְהָסִיר אוֹפַנֵּי הַחֵשֶׁק וְהַנָּשׂוּא מֵהָרַעְיוֹנִים וְהַמַּחֲשָׁב • וְאֵימָתַי אִם לֹא עַכְשָׁיו •


וָאֶשָּׂא מְשָׁלִי וָאֹמַר:


בָּא הַקֵּץ הָקִיץ מִשֵּׁנַת

שַׁחְרוּת וּלְאַבֵּד בָּמוֹתָי

וּלְהָסִיר אוֹפְנֵי הַחֵשֶׁק

כִּי אִם לֹא עַכְשָׁיו אֵימָתָי •


וַיֹּאמֶר הַשָּׂר אַשְׁבִּיעֲךָ בְּיוֹצֵר רוּחַ אָדָם בְּקִרְבּוֹ • לְהוֹדִיעֵנִי מָה הַפֶּלֶא אֲשֶׁר רָאִיתָ בְּיַם הַחֵשֶׁק בִּהְיוֹתְךָ בוֹ • וָאַעַן וָאֹמַר חַטָּאתִי אוֹדִיעֲךָ וַעֲוֹנִי לֹא כִסִּיתִי • הַפֶּלֶא הַגָּדוֹל אֲשֶׁר בּוֹ רָאִיתִי • הוּא אֲשֶׁר בּוֹ שָׁלָטְתִּי • וְצָלַלתִּי כַעוֹפֶרֶת בְּמֵימָיו הָאַדִּירִים וְנִמְלָטְתִּי •


וָאֶשָּׂא מְשָׁלִי וָאֹמַר:


יוֹם צֵאתִי מִיַּם הַחֵשֶׁק

אָמְרוּ מַה פֶּלֶא נִרְאָה בוֹ

עָנִיתִי הַפֶּלֶא גָּדוֹל

הוּא שֶׁנִּמְלַטְתִּי מִקִּרְבּוֹ •


וַיֹּאמֶר הַשָּׂר הִנְנִי בִגְאוֹנִי נִשְׁבָּע • כִּי יָרֵא אֲנִי פֶּן בְּיַם הַחֵשֶׁק אֶטְבָּע • כִּי יוֹדֵעַ אֲנִי הֱיוֹתִי מֻרְכָּב מִגּוּף וּמִנֶּפֶשׁ • וְהַנֶּפֶשׁ רוּחָנִית וְהַגּוּף טִיט וָרֶפֶשׁ • וְאִם הַנֶּפֶשׁ אַחַר הַשֵּׂכֶל יִטֶּה הַגּוּף אַחַר הַחֹמֶר אוֹזֵל • גּוֹרָל אֶחָד לַאדֹנָי וְגוֹרָל אֶחָד לַעֲזָאזֵל • וּבְנוֹת הַחֵן יִמֱשְׁכוּנִי כִּמְשֹׁךְ הָאֶבֶן הַשֹּׁאֶבֶת אֶת הַבַּרְזֶל • וּבַמָּה אוֹצִיא כַּסְפִּי וּזְהָבִי • וַאֲכַבֶּה חֵשֶׁק לְבָבִי • אִם לֹא בִּבְנוֹתִי בֵית מִדּוֹת • וַעֲלִיּוֹת מְרֻוָּחִים וְנִכְבָּדוֹת • וּבְתַעֲנֻגוֹת בְּנֵי הָאָדָם שִׁדָּה וְשִׁדּוֹת • וָאַעַן וָאֹמַר הִשָּׁמֵר וְהַשְׁקֵט הַשָּׂר פֶּן יְשִׁיבוּךָ הַתַּאֲוֹת דַּךְ וְנִכְלָם • וְאִם נָשְׂאוּ נַהֲרוֹת הַבַּחֲרוּת קוֹלָם • וַיִּשְׂאוּ נַהֲרוֹת הַטְּבָעִים דָּכְיָם • אַל תָּמִיר בַּכָּלֶה הַקַּיָּם • וְתָנִיף יַד שִׂכְלְךָ עַל נְהַר הַתַּאֲוֹת בַּעְיָם • וְיֵרְדְּ מִיָּם עַד יָם • וְאַל תַּאֲמֵן אֶל הַכְּסִילִים אִם בַּמְּרוֹמִים יִתְּנוּ הַסָּכֶל • וְהוֹצֵא כְסָפֶיךָ בִּדְבַר חָכְמָה וְתִמְצָא תְמוּרָתָם זְהַב הַשָּׂכֶל •


וָאֶשָּׂא מְשָׁלִי וָאֹמַר:


קָרְבָה לְעַם בִּינוֹת וְסוּרָה מִבְּנוֹת

חֵן כִּי פְּאֵר הוֹדָם מְהֵרָה יִכֶל

מִתַּעֲנוּגֵי הַזְּמָן סוּר נֶאֱמָן

רַק בֶּאֱמֶת תִּדְבַּק וְסוּר מִנָּכֶל

אַל תַּאֲמֵן אֶל הָעֲוִילִים נוֹאֲלִים

אִם בַּמְּרוֹמִים יִתְּנּוּ הַסָּכֶל

הוֹצֵא כַּסְפְּךָ בְּסִפְרֵי הַיְּקָר

תִּמְצָא תְמוּרָתוֹ זְהַב הַשָּׂכֶל •


וּמִמַּעֲלַת הַחָכְמָה שֶׁהִיא תַעֲנוּג הַנְּשָׁמוֹת וְיִנְחֲלוּ נוֹחֲלֶיהָ שְׁנֵי עוֹלָמוֹת • וּבָהּ יִהְיוּ קְדוֹשִׁים אַחֲרֵי מוֹת • הִיא הַסֻּלָּם לַעֲלוֹת אֶל אֲדֹנָי וְאֶל טוּבוֹ • אֲשֶׁר מַלְאֲכֵי אֱלֹהִים עוֹלִים וְיוֹרְדִים בּוֹ • וְהִיא הַנּוֹתֶנֶת לַיָּעֵף כֹּחַ וּלְאֵין אוֹנִים תַּרְבֶּה עָצְמָה •

וְלֹא הִנְחִיל הָאָב לַבֵּן דָּבָר טוֹב מִן הַחָכְמָה •


וָאֶשָּׂא מְשָׁלִי וָאֹמַר:


אִם יוֹרִישׁוּ אָבוֹת הוֹן רַב

לִבְנֵיהֶם עִם נַחְלָה רָמָה

אוֹ יִבְנוּ לָהֶם מִגְדּוֹל עוֹז

אוֹ בֵּית מוֹשָׁב תּוֹךְ עִיר חוֹמָה

תֵּן אַתָּה לַבָּנִים עָרְמָה

וּלְנַעַר דַּעַת וּמְזִמָּה

כִּי לֹא הִנְחִיל הָאָב לַבֵּן

דָּבָר יָקָר מִן הַחָכְמָה •


וְלָכֵן הַשָּׂר רִיבָה אֶת יִצְרֶךָ • אֲשֶׁר בִּקֵּשׁ לְהַדִּיחֲךָ מֵעַל יוֹצְרֶךָ • וְתָבִיא צַוָּארְךָ בַּעֲבוֹדַת הָאֱלֹהִים • גָּבֹהַּ מֵעַל גָּבֹהַּ וְנִשָּׂא עַל כָּל גְּבוֹהִים • לְבָרֵךְ אֶת שְׁמוֹ בְמַקְהֵלוֹת • הַנֶּאְדָּר בַּקֹּדֶשׁ וְנוֹרָא תְהִלּוֹת • לְהַגִּיד בַּבֹּקֶר חַסְדּוֹ וֶאֱמוּנָתוֹ בַּלֵּילוֹת • עֲלֵי עָשׂוֹר וַעֲלֵי נָבֶל • כִּי הַיַּלְדוּת וְהַשַּׁחֲרוּת הָבֶל • וְהוּא יַעַלְךָ מִבּוֹר שָׁאוֹן וְטִיט הַיָּוֵן • אִם תָּסִיר מִתּוֹכְךָ מוֹטָה שְׁלַח אֶצְבַּע וְדַבֶּר‑אָוֶן •


וָאֶשָּׂא מְשָׁלִי וָאֹמַר:


בַּמֶּה אֱלֹהִים יִפְדְּךָ מִיַּד שְׁאוֹל

מָטָה וּבוֹר שָׁאוֹן וְטִיט הַיָּוֵן

אִם לֹא בְּהָסִיר שִׂכְלְךָ מִתּוֹכְךָ

מוֹטָה שְׁלַח אֶצְבַּע וְדַבֶּר אָוֶן •


וְהַמַּעֲלָה אֲשֶׁר עַל כָּל מַעֲלוֹת לָהּ גַּבְהוּת • הִיא לְהִתְהַפֵּךְ מִן הַבַּהֲמוּת אֶל הָאֱלֹהוּת • וְלָכֵן מַעֲלַת הָאֱנוֹשִׁית הִיא לֶאֱחֹז בְּמִדַּת הַפְּרִישׁוּת • אֲשֶׁר הַמַּחֲזִיק בָּהּ יִשְׁלָם • וְאָכַל וָחַי לְעֹלָם • וּלְהַנִּיחַ תַּעֲנוּגֵי הַזְּמָן • וְלָשִׂים יְתַד הַשֵּׂכֶל תְּקוּעָה בְּמָקוֹם נֶאֱמָן • כִּשְׁמֹעַ הַשָּׂר וַחֲבֻרָתוֹ אֶת דְּבָרָי • חָרְדוּ לְקוֹלִי וַיִּנָּהוּ אַחֲרָי • וַיֹּאמְרוּ לִי בְּחַסְדֶּךָ הוֹדִיעֵנוּ הַפְּרִישׁוּת וּמֶבָּטוֹ • וְחֻקּוֹ וּמִשְׁפָּטוֹ • כִּי דְרָכָיו נֶעֶלְמוּ מִמֶּנּוּ וְאֹדוֹתָם • וְעַתָּה הוֹצִיאֵם אֵלֵינוּ וְנֵדְעָה אוֹתָם • וְנָשִׂים אוֹתָם עַל לִבֵּנוּ חוֹתָם • וָאַעַן וָאֹמַר דַּרְכֵי הַפְּרִישׁוּת נִכְבָּדוֹת מְדֻבָר בָּם • מַה יָּפְיָם וּמַה טּוּבָם • כִּי הַפְּרִישׁוּת הִיא מַעֲלַת הַחֲסִידִים וְהַנְּבִיאִים • הָעוֹמְדִים בִּלְבָבוֹת נִדְכָּאִים • הַמּוֹאֲסִים בְּקִרְבַת הַבְּרוּאִים • הַלָּנִים בֵּין שְׁפָאִים • הַמִּתְרַחֲקִים מִכָּל גְּנוּת • הָאוֹכְלִים לֶחֶם בְּמִסְכֵּנוּת • וְלֹא יִבְזוּ וְלֹא יְשַׁקְּצוּ עֱנוּת • הַפּוֹשְׁטִים מְעִילֵי הַהֲמוֹנוּת • וְהַלּוֹבְשִׁים שַׂקֵּי הַנְּזִירוּת • הַנִּכְסָפִים אֶל הָאוֹר הָעֶלְיוֹן וְהַמֵּתִים בַּגֵּרוּת • הַמַּתִּירִים עֲבוֹתוֹת הַחֶבְרָה • וְיוֹבִילוּ נַפְשׁוֹתָם שַׁי לַמּוֹרָא •


וָאֶשָּׂא מְשָׁלִי וָאֹמַר:


אִם בַּפְּרִישׁוּת יְדֵיכֶם אֱמוּנוֹת

חִגְרוּ יְגוֹנוֹת וְתֵשְׁבוּ נְכָאִים

לִפְשׁוֹט מְעִילֵי הֲמוֹנוֹת וְלִלְבּוֹשׁ

שַׂקֵּי נְזִירוּת וְלִין בֵּין שְׁפָאִים

כִּי הַפְּרִישׁוּת לְהַתִּיר עֲבוֹתוֹת

חֶבְרָה וְלִמְאֹס בְּקִרְבַת בְּרוּאִים •


וַיֹּאמֶר הַשָּׂר אַחֲרֵי אֲשֶׁר מִדַּרְכֵי הַפְּרִישׁוּת לִמְאֹס בְּחֶבְרַת בְּרוּאִים בְּתוֹךְ לְבָאִים • וְנָפַל בַּעֲצוּמָיו חֵלְכָּאִים • אִם נִצְטָרֵךְ לְהַנִּיחַ חֶבְרַת הַיָּפוֹת • וְתַעֲנוּגֵי הַנָּשִׁים הַנּוֹאֲפוֹת • וְעַתָּה הוֹדִיענִי אִם יֵשׁ מִדַּרְכֵי הַפְּרוּשִׁים • לְהִמָּנַע מִלְּדַבֵּר בְּעִנְיַן הַנָּשִׁים • וָאַעַן וָאֹמַר יֵשׁ וָיֵשׁ • וַיִּשְׂחַק הַשָּׂר וַיֹּאמַר וְאֵיךְ לֹא תִתְבַּיֵּשׁ • כִּי בְּבֹא נָתָן רֵעֶךָ יָד לְיָד שְׁאִלְתּוּ • עַל דְּבַר שָׂרָה אִשְׁתּוֹ • וָאַעַן וָאֹמַר חַיֶּיךָ מָצָאתִי לְפִידְךָ פִדְיוֹם • אֲשֶׁר תִּפְקֹד עָלַי עֲוֹן הָאִשָּׁה הַיּוֹם • כִּי אִם שְׁאִלְתִּיו עַל דְּבַר שָׂרָה אִשְׁתּוֹ לֹא אֶהְיֶה כְּאִישׁ נִדְהָם • וּבָזֶה כּוֹכְבֵי שִׂכְלִי לֹא אָסְפוּ נָגְהָם • כִּי בָאוּ הַמַּלְאָכִים אֶל אַבְרָהָם • וַיֹּאמְרוּ אֵלָיו אַיֵּה שָׂרָה אִשְׁתֶּךָ • וְלָכֵן חֲדַל לְךָ הַשָּׂר וְתִירָא בְגִשְתֶּךָ • כִּי נִלְכַּדְתָּ בְּרִשְתֶּךָ •


וָאֶשָּׂא מְשָׁלִי וָאֹמַר:


בִּי גָעֲרוּ דּוֹדִים בְבֹא נָתָן

אֶל שָׁאֲלִי מֶנּוּ שְׁלוֹם עָפְרָה

אֵיךְ יִגְעֲרוּ בִי הַפְתָאִים אוֹ

עָלַי יְדַבְּרוּ דְבַר סָרָה

בָּאוּ בְּנֵי אֵלִים לְאַבְרָהָם

וַיִּשְׁאֲלוּהוּ עַל דְּבַר שָׂרָה •


כִּשְׁמֹעַ הָאֲנָשִׁים שִׁירִי זָחֲלוּ מֵעִנְיָנָיו • וְלֹא יָכְלוּ לַעֲנוֹת אוֹתוֹ כִּי נִבְהֲלוּ מִפָּנָיו • וַיֹּאמֶר הַשָּׂר חַיֶּיךָ וְחַיֵּי הָאוֹהֲבִים הַנְּעִימִים • כִּי אַתָּה נֵזֶר הַיָּמִים • וּבִלְעָדֶיךָ בְּעִנְיַן הַמְּלִיצוֹת וְהַשִּׁירִים • אִישׁ אֶת יָדוֹ וְאֶת רַגְלוֹ לֹא יָרִים • בָּרוּךְ יִהְיֶה אֲשֶׁר שְׁלָחֲךָ הַיּוֹם פֹּה • לְהַעֲמִיד יַם הַיְגוֹנִים מִזַּעְפּוֹ • וַיִּפְשֹׁט הַשָּׂר אֶת מְעִילוֹ • וְאֶת כְּתֹנֶת הַפַּסִּים אֲשֶׁר לוֹ • וָאֶלְבַּשׁ אוֹתָם כְּמִצְוָתוֹ • וָאַפְקִיד בְּיַד הָאֵל רוּחוֹ וְנִשְׁמָתוֹ • וַיְבָרְכוּ הָאֲנָשִים אוֹתִי בְּשֵׁם אֵל עֵילוֹם • וָאֵלֵךְ מֵאִתָּם בְּשָׁלוֹם •
